23/// header with the macro COMPILE_LAYERED_WINDOW_STUBS defined.

44// Define COMPILE_LAYERED_WINDOW_STUBS to compile the stubs;
45// otherwise, you get the declarations.
46//
47#ifdef COMPILE_LAYERED_WINDOW_STUBS
48
49// Implement API stubs.
50//
51
52BOOL(WINAPI* g_pfnSetLayeredWindowAttributes)(HWND, COLORREF, BYTE, DWORD) { nullptr };
53BOOL(WINAPI* g_pfnUpdateLayeredWindow)(HWND, HDC, POINT*, SIZE*, HDC, POINT*, COLORREF, BLENDFUNCTION*, DWORD) { nullptr };
54BOOL g_fLayeredWindowInitDone { FALSE };
55BOOL g_enableLayeredWindows { TRUE };
56
57BOOL InitLayeredWindowStubs(void) {
58 HMODULE hUser32;
59
60 if (g_fLayeredWindowInitDone)
61 return g_pfnSetLayeredWindowAttributes != nullptr;
62
63 hUser32 = GetModuleHandle(TEXT("USER32"));
64
65 if ((hUser32 != nullptr) && g_enableLayeredWindows &&
66 (*(FARPROC*)&g_pfnSetLayeredWindowAttributes = GetProcAddress(hUser32, "SetLayeredWindowAttributes")) != nullptr &&
67 (*(FARPROC*)&g_pfnUpdateLayeredWindow = GetProcAddress(hUser32, "UpdateLayeredWindow")) != nullptr
68 ) {
69 g_fLayeredWindowInitDone = TRUE;
70 return TRUE;
71 } else {
72 g_pfnSetLayeredWindowAttributes = nullptr;
73 g_pfnUpdateLayeredWindow = nullptr;
74 g_fLayeredWindowInitDone = TRUE;
75 return FALSE;
76 }
77}
78
79BOOL HaveLayeredWindows(void) {
80 return InitLayeredWindowStubs();
81}
82
83BOOL WINAPI
84xSetLayeredWindowAttributes(HWND hwnd, COLORREF crKey, BYTE bAlpha, DWORD dwFlags) {
85 if (InitLayeredWindowStubs()) {
86 return g_pfnSetLayeredWindowAttributes(hwnd, crKey, bAlpha, dwFlags);
87 }
88 return TRUE;
89}

91BOOL WINAPI
92xUpdateLayeredWindow(HWND hwnd, HDC hdcDst, POINT* pptDst, SIZE* psize, HDC hdcSrc, POINT* pptSrc, COLORREF crKey,
93 BLENDFUNCTION* pblend, DWORD dwFlags) {
94 if (InitLayeredWindowStubs()) {
95 return g_pfnUpdateLayeredWindow(hwnd, hdcDst, pptDst, psize, hdcSrc, pptSrc, crKey, pblend, dwFlags);
96 }
97 return TRUE;
98}
