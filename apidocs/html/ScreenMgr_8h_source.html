31/// Provides information on the display monitor(s). One of the primary
32/// jobs of this manager is to properly handle multiple display monitor
33/// environments.
34///
35class MeaScreenMgr : public MeaScreenProvider, public MeaSingleton_T<MeaScreenMgr> {
36
37public:
38 static constexpr bool kDefUseManualRes { false }; ///< Use manual resolution by default.

65 /// Returns the number of display screens attached to the system.

69 int GetNumScreens() const override { return static_cast<int>(m_screens.size()); }

72 /// @return Constant iterator over all screens.

85 /// Obtains an iterator pointed at the screen that contains the

92 ///
93 ScreenIter GetScreenIter(const POINT& point) const override;
94
95 /// Obtains an iterator pointed at the screen that contains the
96 /// specified rectangle.
97 ///
98 /// @param rect [in] Rectangle whose screen is desired.
99 ///
100 /// @return Constant iterator pointed at screen containing the
101 /// specified rectangle.
102 ///
103 ScreenIter GetScreenIter(const RECT& rect) const override;
104
105 /// Indicates if the screen iterator is at the end of the
106 /// screen collection.
107 ///
108 /// @param iter [in] Screen iterator to test.
109 ///
110 /// @return <b>true</b> if the specified iterator is pointed to
111 /// the end of the collection of screens.
112 ///
113 bool AtEnd(const ScreenIter& iter) const override { return iter == m_screens.end(); }
114
115 /// Determines the center of the screen that contains the

117 ///
118 /// @return Center point of the screen containing the app's

121 const CPoint& GetCenter() const override;
122
123 /// Returns a point guaranteed to be located off any screen.
124 ///
125 /// @return Point located off any display screen.
126 ///
127 CPoint GetOffScreen() const override;
128
129 /// Returns the virtual screen rectangle. The virtual rectangle
130 /// is the rectangle containing all display screen attached to
131 /// the system.
132 ///
133 /// @return Rectangle encompassing all display screens attached
134 /// to the system, in pixels.
135 ///
136 const CRect& GetVirtualRect() const override { return m_virtualRect; }
137
138 /// Returns the screen rectangle for the screen pointed to by
139 /// the specified iterator.

141 /// @param iter [in] Screen iterator pointed at the screen

144 /// @return Screen rectangle, in pixels.
145 ///
146 const CRect& GetScreenRect(const ScreenIter& iter) const override;
147
148 /// Returns the manually set screen resolution for the screen
149 /// pointed to by the specified iterator.
150 ///
151 /// @param iter [in] Screen iterator pointed at the screen
152 /// whose manual resolution is desired.
153 /// @param useManualRes [out] <b>true</b> if the resolution was set
154 /// manually.
155 /// @param manualRes [out] Manually set resolution, in pixels per inch.
156 ///
157 void GetScreenRes(const ScreenIter& iter, bool& useManualRes, MeaFSize& manualRes) const override;
158
159 /// Returns the resolution for the screen pointed to by the specified iterator.
160 ///
161 /// @param iter [in] Screen iterator pointed at the screen whose
162 /// resolution is desired.
163 ///
164 /// @return Screen resolution, in pixels per inch. If the resolution has been
165 /// set manually, it is returned. Otherwise, the operating system
166 /// reported resolution is returned.
167 ///
168 const MeaFSize& GetScreenRes(const ScreenIter& iter) const override;
169
170 /// Indicates if the resolution for the specified screen has been set manually.
171 ///
172 /// @param iter [in] Screen iterator pointed at the screen whose
173 /// resolution setting is being queried.
174 ///
175 /// @return <b>true</b> if the screen's resolution was set manually.
176 ///
177 bool IsManualRes(const ScreenIter& iter) const override;
178
179 /// Indicates whether any screen is using the operating system reported

183 ///
184 bool AnyOSRes() const override;
185
186 /// Sets the resolution for the screen pointed to by the specified iterator.
187 ///
188 /// @param iter [in] Iterator pointing to the screen whose resolution is
189 /// to be set.

191 /// @param manualRes [in] Manually calibrated screen resolution, in pixels.
192 ///
193 void SetScreenRes(const ScreenIter& iter, bool useManualRes, const MeaFSize* manualRes = nullptr) const;
194
195 /// Indicates if the screen pointed to by the specified iterator is the primary.
196 ///
197 /// @return <b>true</b> if the specified screen is the primary.
198 ///
199 bool IsPrimary(const ScreenIter& iter) const override;
200
201 /// Returns the name for the specified screen.
202 ///
203 /// @param iter [in] Iterator pointing to the screen whose name is desired.
204 ///
205 /// @return Name of the specified screen.
206 ///
207 CString GetScreenName(const ScreenIter& iter) const override;
208
209 /// Indicates if the specified screen was calibrated in inches or centimeters.
210 ///
211 /// @param iter [in] Points at the screen whose calibration units are desired.

214 ///
215 bool IsCalInInches(const ScreenIter& iter) const;
216
217 /// Specifies the calibration units for the specified screen.
218 ///
219 /// @param iter [in] Points at the screen whose calibration units are to be set.

221 ///
222 void SetCalInInches(const ScreenIter& iter, bool calInInches) const;
223
224 /// Indicates whether the virtual screen rectangle has changed
225 /// since the last time the application was run. If the virtual screen
226 /// size has changed, re-calibration is recommended.
227 ///
228 /// @return <b>true</b> if the virtual screen rectangle has changed.
229 ///
230 bool SizeChanged() const { return m_sizeChanged; }
231
232 /// Ensures that the specified window rectangle is visible on
233 /// the nearest screen. If the window rectangle is already visible,
234 /// the rectangle is not changed.
235 ///
236 /// @param windowRect [in] Window rectangle that is to be tested
237 /// for visibility. The rectangle is altered if
238 /// necessary to ensure at least a part of the
239 /// window is visible.
240 ///
241 /// @return New rectangle that is guaranteed to be visible.
242 ///
243 CRect EnsureVisible(const RECT& windowRect) const;
244
245 /// Ensures that the specified point is on a screen.
246 ///
247 /// @param pt [in] Point to check.
248 ///
249 /// @return New point based on the specified point that is guaranteed
250 /// to be on a screen.
251 ///
252 CPoint LimitPosition(const CPoint& pt) const override;
253
254private:

263 /// @param userData [in] this pointer.
264 ///
265 /// @return TRUE to continue enumeration of monitors.
266 ///
267 static BOOL CALLBACK CreateScreens(HMONITOR hMonitor, HDC hdcMonitor, LPRECT monitorRect, LPARAM userData);
268
269 /// Obtains the screen containing the specified point.

271 /// @param point [in] Point whose screen is desired.
272 ///
273 /// @return Screen containing the specified point or nullptr if
274 /// the point is not contained by any screen.
275 ///
276 Screen* GetScreen(const POINT& point) const;
