31/// Provides information on the display monitor(s). One of the primary
32/// jobs of this manager is to properly handle multiple display monitor
33/// environments.
34///
35class MeaScreenMgr : public MeaSingleton_T<MeaScreenMgr>
36{
37private:
38 /// Represents a single display screen.
39 ///
40 class Screen
41 {
42 public:
43 /// Constructs a display screen object.
44 ///
45 /// @param mgr [in] Parent manager.
46 /// @param rect [in] Rectangle representing the screen, in pixels.

50 Screen(MeaScreenMgr& mgr, LPCRECT rect);

52 /// Destroys a display screen object.
53 ///

56
57 /// Returns the screen rectangle. Note that in multiple
58 /// monitor environments, the screen rectangle can have
59 /// negative coordinates.
60 ///
61 /// @return Screen rectangle, in pixels.
62 ///
63 const CRect& GetRect() const { return m_rect; }
64
65 /// Determines the center of the screen.
66 ///
67 /// @return Center point of the screen.
68 ///
69 const CPoint& GetCenter() const { return m_center; }
70
71
72 /// Sets the resolution of the screen, in pixels per inch.
73 ///
74 /// @param useManualRes [in] Specifies whether the manually calibrated
75 /// resolution should be used (true) or the operating
76 /// system resolution (false).
77 /// @param manualRes [in] If non-NULL, specifies the manually calibrated
78 /// resolution.
79 ///
80 void SetScreenRes(bool useManualRes, const FSIZE *manualRes = NULL);
81
82 /// Obtains the currently set manual screen resolution, if set.
83 ///
84 /// @param useManualRes [out] Indicates whether the manually calibrated
85 /// resolution is used (true) or the operating
86 /// system resolution (false).
87 /// @param manualRes [out] Manually calibrated resolution, in pixels
88 /// per inch.
89 ///
90 void GetScreenRes(bool& useManualRes, FSIZE& manualRes) const {
91 useManualRes = m_useManualRes;
92 manualRes = m_manualRes;
93 }
94
95 /// Obtains the current screen resolution. If the resolution has been
96 /// manually calibrated, it is returned. Otherwise the operating system
97 /// reported resolution is returned.
98 ///
99 /// @return Screen resolution, in pixels per inch.
100 ///
101 const FSIZE& GetScreenRes() const { return m_currentRes; }
102
103 /// Indicates if the screen resolution has been manually set.
104 ///
105 /// @return <b>true</b> if the screen resolution has been set manually.
106 ///
107 bool IsManualRes() const { return m_useManualRes; }
108
109
110 /// In multiple monitor environments, one of the monitors is
111 /// designated as the primary. This method designates the screen
112 /// as the primary.
113 ///
114 /// @param primary [in] Specify <b>true</b> if this screen
115 /// is the primary.
116 ///
117 void SetPrimary(bool primary) { m_primary = primary; }
118
119 /// Indicates if this is the primary screen.
120 ///
121 /// @return <b>true</b> if this is the primary screen.
122 ///
123 bool IsPrimary() const { return m_primary; }
124
125
126 /// Specifies whether the resolution calibration has been performed
127 /// in inches or centimeters.
128 ///
129 /// @param calInInches [in] <b>true</b> if the calibration has been
130 /// performed in inches.
131 ///
132 void SetCalInInches(bool calInInches) { m_calInInches = calInInches; }
133
134 /// Indicates whether the resolution calibration has been performed
135 /// in inches or centimeters.
136 ///
137 /// @return <b>true</b> if the calibration has been performed in inches.
138 ///
139 bool GetCalInInches() const { return m_calInInches; }
140
141
142 /// Sets a descriptive name for the screen.
143 ///
144 /// @param name [in] Name for the screen.
145 ///
146 void SetName(LPCTSTR name) { m_name = name; }
147
148 /// Returns the descriptive name for the screen.
149 ///
150 /// @return Name for the screen or the empty string if
151 /// there is no name.
152 ///
153 CString GetName() const { return m_name; }
154
155 private:
156 MeaScreenMgr& m_mgr; ///< Parent manager.
157 CRect m_rect; ///< Screen rectangle.
158 CPoint m_center; ///< Center point of the screen.
159
160 FSIZE m_currentRes; ///< Current screen resolution, pixels per inch.
161 FSIZE m_manualRes; ///< Manually calibrated resolution, pixels per inch.
162 bool m_useManualRes; ///< Indicates if manually calibrated resolution is used.
163 bool m_calInInches; ///< Indicates if calibration in inches or centimeters.
164 bool m_primary; ///< Indicates if this is the primary screen.
165 CString m_name; ///< Descriptive name for the screen.
166 };

174 /// Used by the STL collections for ordering comparisons.
175 ///
176 struct less {
177 /// Compares two screen iterators.
178 /// @param lhs [in] Left hand side of the comparison.
179 /// @param rhs [in] Right hand side of the comparison.
180 /// @return <b>true</b> if lhs < rhs.
181 bool operator()(const ScreenIter& lhs, const ScreenIter& rhs) const
182 {
183 return (*lhs).first < (*rhs).first;
184 }

330 /// Indicates whether any screen is using the operating system reported
331 /// resolution. In other words, are there any screen that require calibration.

333 /// @return <b>true</b> if there are any screens requiring manual calibration.
334 ///
335 bool AnyOSRes() const {
336 for (ScreenIter iter = m_screens.begin(); iter != m_screens.end(); ++iter) {
337 if (!(*iter).second->IsManualRes()) {
338 return true;

341 return false;
