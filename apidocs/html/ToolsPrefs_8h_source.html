30/// Property page for specifying the following tool preferences:
31/// <ul>
32/// <li>Line color</li>
33/// <li>Crosshair normal background</li>
34/// <li>Crosshair highlight color</li>

36/// <li>Crosshair opacity</li>
37/// <li>Display of popup data window</li>
38/// <li>Display of the origin marker</li>
39/// </ul>

112 /// @fn OnChangeLineColor()
113 /// Called when the line color button is pressed. Displays a
114 /// color picker dialog for selecting a new line color. The
115 /// selected color is shown in the line color swatch.
116
117 /// @fn OnDefLineColor()
118 /// Called when the default line color button is pressed. Resets
119 /// the line color to its default value.
120
121 /// @fn OnChangeBorderColor()
122 /// Called when the crosshair border color button is pressed.
123 /// Displays a color picker dialog for selecting a new border
124 /// color and displays the selected color on the sample crosshairs.
125
126 /// @fn OnDefBorderColor()
127 /// Called when the default border color button is pressed. The
128 /// default border color is set and displayed on the sample crosshairs.
129
130 /// @fn OnChangeBackColor()
131 /// Called when the crosshair background color button is pressed.
132 /// Displays a color picker dialog for selecting a new background
133 /// color and displays the selected color in the normal crosshair
134 /// sample.
135
136 /// @fn OnDefBackColor()
137 /// Called when the default background color button is pressed.
138 /// The default background color is set and displayed on the normal
139 /// crosshair sample.
140
141 /// @fn OnChangeHiliteColor()
142 /// Called when the highlight color button is pressed. Displays a
143 /// color picker for selecting a new highlight color and displays the
144 /// selected color in the highlight crosshair sample.
145
146 /// @fn OnDefHiliteColor()
147 /// Called when the default highlight color button is pressed. The
148 /// default highlight color is set and displayed on the highlight crosshair
149 /// sample.
150
151 /// @fn OnHScroll(UINT nSBCode, UINT nPos, CScrollBar* pScrollBar)
152 /// Called when the crosshair opacity slider is moved. Sets the new
153 /// opacity on the crosshair samples.
154
155 /// @fn OnChange()
156 /// Called whenever a value is changed on the property page. Sets
157 /// the modified flag.
158
159private:
160 MeaSwatch m_line; ///< Swatch to display the line color.
161 MeaCrossHair m_normalCH; ///< Sample crosshair to preview the normal background color.
162 MeaCrossHair m_hiliteCH; ///< Sample crosshair to preview the highlight color.
163 MeaDataWin m_dataWin; ///< Sample data popup window.
164};
