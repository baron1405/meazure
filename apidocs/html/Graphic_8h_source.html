26/// Base class for all graphic elements. Classes derived from this base
27/// class are used by the measurement tools to perform their function.
28///
29class MeaGraphic : public CWnd {
30
31public:
32 /// Base class constructor. Prior to displaying the graphic with
33 /// the Show() method, the Create() method must be called to create
34 /// the graphic's window.
35 ///
36 MeaGraphic();
37
38 /// Base class destructor
39 ///
40 virtual ~MeaGraphic();
41
42 /// Creates the window for the graphic. This method must be called
43 /// before the graphic can be displayed and before any operation that
44 /// attempts to manipulate the graphic's window.
45 ///
46 /// @param classname [in] Class name for the window. Typically, a derived
47 /// class obtains a window class name from a call to the
48 /// AfxRegisterWndClass function.
49 /// @param size [in] Initial size for the window.
50 /// @param parent [in] Parent window or <b>nullptr</b> if the graphic does
51 /// not have an immediate parent. If a parent window is
52 /// specified, the graphic is created as a child window
53 /// (<b>WS_CHILD</b>). Otherwise the graphic is created
54 /// as a popup window (<b>WS_POPUP</b>) and is assigned
55 /// the main window as a parent.
56 /// @param id Unique ID for this graphic element. Typically, the ID
57 /// is used to identify the graphic element in a callback.
58 ///
59 /// @return <b>true</b> if the window is created successfully.
60 ///
61 bool Create(PCTSTR classname, const SIZE& size, const CWnd* parent, UINT id = 0xFFFF);
62
63 /// Displays the graphic. Call the Create() method prior
64 /// to calling this method for the first time.
65 ///
66 virtual void Show();
67
68 /// Hides the graphic. Call the Create() method prior
69 /// to calling this method for the first time.
70 ///
71 virtual void Hide();
72
73 /// Changes the z-order of the graphic window so that it
74 /// if above the specified window. Call the Create() method
75 /// prior to calling this method for the first time.
76 ///
77 /// @param winBelow [in] Window to place below this class's window
78 ///
79 void SetAbove(const CWnd* winBelow) { SetWindowPos(winBelow, 0, 0, 0, 0, SWP_NOMOVE | SWP_NOSIZE); }
80
81protected:
82 static constexpr int kInitCoord { -1000 }; ///< An arbitrary initial coordinate for placing new graphics
83
84
85 UINT m_id; ///< ID for the graphic
86 bool m_visible; ///< Indicates whether the graphic is shown or hidden
87 const CWnd* m_parent; ///< Graphic's parent window or nullptr if graphic is a popup window
88};
