30/// The frame class for the application. The application frame is the
31/// root window for the application's window hierarchy.
32///
33class AppFrame : public CFrameWnd {
34
35public:
36 static constexpr bool kDefAlwaysVisible { true }; ///< Indicates whether the application's window is always the
37 ///< topmost window by default.
38
39 /// Constructs the application's frame window.
40 ///
41 AppFrame();
42
43 /// Destroys the application's frame window.
44 virtual ~AppFrame();
45
46#ifdef _DEBUG
47 /// Verifies that the frame window is valid.
48 ///
49 virtual void AssertValid() const override;
50
51 /// Dumps the state of the frame window.
52 ///
53 /// @param dc [in] Device context for debugging output.
54 ///
55 virtual void Dump(CDumpContext& dc) const override;
56#endif
57
58 /// Returns the view window associated with the frame.
59 /// Since this application is a single document style
60 /// program, there is only one view held by the frame.
61 ///
62 /// @return The application's view window.
63 ///
64 AppView* GetView() { return &m_wndView; }
65
66 /// Returns the keyboard accelerator table for the application.
67 ///
68 /// @return Handle for the application keyboard accelerator table.
69 ///
70 HACCEL GetAccelTable() { return m_hAccelTable; }
71
72 /// Sets the window to either the topmost (i.e. always visible)
73 /// position or a normal stacking order.
74 ///
75 void UpdateVisibility();
76
77 /// Sets the filename specified on the command line, if any.
78 ///
79 /// @param fileName [in] Filename specified on the application
80 /// command line, or the empty string if no filename
81 /// was specified.
82 ///
83 void SetCommandLineFile(const CString& fileName) { m_cmdLineFile = fileName; }
84
85 /// Persists the frame's state to the specified profile object
86 /// and initiates the persisting of the application view. Ultimately,
87 /// this method causes the persistence of the entire state of the
88 /// application.
89 ///
90 /// @param profile [in] Profile object into which the state
91 /// of the application is persisted.
92 ///
93 void SaveProfile(MeaProfile& profile);
94
95 /// Restores the frame's state from the specified profile object
96 /// and initiates the restoring of the application view. Ultimately,
97 /// this method causes the restoration of the entire state of the
98 /// application.
99 ///
100 /// @param profile [in] Profile object from which the state
101 /// of the application is restored.
102 ///
103 void LoadProfile(MeaProfile& profile);
104
105 /// Called at startup right before the application windows are shown.
106 /// Shows/hides the portions of the view according to the registry
107 /// profile.
108 ///
109 void InitView();
110
111 /// Indicates whether this is the first time the application has
112 /// been run.
113 ///
114 /// @return <b>true</b> if this is the first time the application
115 /// has been run.
116 ///
117 bool IsNewInstall() const { return m_newInstall; }
118
119 /// Forwards command messages to the view.
120 /// @param nID [in] ID of control issuing the message.
121 /// @param nCode [in] Message code.
122 /// @param pExtra [in] Pointer to extra data.
123 /// @param pHandlerInfo [in] Handler information structure.
124 ///
125 virtual BOOL OnCmdMsg(UINT nID, int nCode, void* pExtra, AFX_CMDHANDLERINFO* pHandlerInfo) override;
126
127protected:
128 DECLARE_DYNAMIC(AppFrame)
129
130 static UINT m_indicators[]; ///< Status bar indicators. Only one field is used.
131
132
133 /// Called before the window is created to set the appearance
134 /// of the application including setting the program's icon.
135 /// @param cs [in] Window creation structure.
136 ///
137 virtual BOOL PreCreateWindow(CREATESTRUCT& cs) override;
138
139 /// Creates the main frame window, the toolbar and the view.
140 /// @param lpCreateStruct [in] Window creation parameters structure.
141 /// @return 0 if creation successful.
142 ///
143 afx_msg int OnCreate(LPCREATESTRUCT lpCreateStruct);
144
145 /// Called when the frame gets the focus and forwards it on to the view.
146 /// @param pOldWnd [in] Window that had the focus previously.
147 ///
148 afx_msg void OnSetFocus(CWnd* pOldWnd);
149
150 /// Called when the Always Visible item is selected on the View menu.
151 ///
152 afx_msg void OnAlwaysVisible();
153
154 /// Called right before the View menu is displayed to update the
155 /// Always Visible menu item.
156 /// @param pCmdUI [in] UI command class to update the menu item.
157 ///
158 afx_msg void OnUpdateAlwaysVisible(CCmdUI* pCmdUI);
159
160 /// Called when the windows session is ending. Saves the state of the
161 /// application to the registry.
162 /// @param bEnding [in] TRUE if the windows session is ending.
163 ///
164 afx_msg void OnEndSession(BOOL bEnding);
165
166 /// Called when the application window is closing. Saves the state of
167 /// the application to the registry.
168 ///
169 afx_msg void OnClose();
170
171 /// Changes the window geometry according to whether the toolbar is
172 /// displayed.
173 ///
174 afx_msg void OnViewToolbar();
175
176 /// Updates the view toolbar menu item.
177 /// @param pCmdUI [in] UI command class to update the menu item.
178 ///
179 afx_msg void OnUpdateViewToolbar(CCmdUI* pCmdUI);
180
181 /// Called when the status bar view menu item is selected. Toggles
182 /// view / hide of the status bar.
183 ///
184 afx_msg void OnViewStatusbar();
185
186 /// Updates the view status bar menu item.
187 /// @param pCmdUI [in] UI command class to update the menu item.
188 ///
189 afx_msg void OnUpdateViewStatusbar(CCmdUI* pCmdUI);
190
191 /// Called before the Help menu is displayed. Tests to see
192 /// if the help file exists.
193 /// @param pCmdUI [in] UI command class to update the menu item.
194 ///
195 afx_msg void OnUpdateHelpMenu(CCmdUI* pCmdUI);
196
197 /// Called to open the Meazure GitHub releases web page.
198 ///
199 afx_msg void OnMeazureReleases();
200
201 /// Called to open the Meazure GitHub issue web page.
202 ///
203 afx_msg void OnReportIssue();
204
205 /// Called when the Apply button is pressed on the preferences
206 /// dialog.
207 /// @param wParam [in] Not used.
208 /// @param lParam [in] Not used.
209 /// @return Always returns 0.
210 ///
211 afx_msg LRESULT OnPrefsApply(WPARAM wParam, LPARAM lParam);
212
213 /// Called to display the preferences dialog from the
214 /// resolution calibration warning dialogs.
215 /// @param wParam [in] Not used.
216 /// @param lParam [in] Not used.
217 /// @return Always returns 0.
218 ///
219 afx_msg LRESULT OnShowCalPrefs(WPARAM wParam, LPARAM lParam);
220
221 /// Called when a second copy of the application is launched with
222 /// a file specified on the command line. This method receives
223 /// the pathname of the file and loads it.
224 /// @param wParam [in] Not used.
225 /// @param lParam [in] COPYDATASTRUCT
226 /// @return Always returns TRUE.
227 ///
228 afx_msg LRESULT OnCopyData(WPARAM wParam, LPARAM lParam);
229
230 /// Called when a master reset has been requested from the
231 /// advanced page on the preferences dialog.
232 /// @param wParam [in] Not used.
233 /// @param lParam [in] Not used.
234 /// @return Always returns TRUE.
235 ///
236 afx_msg LRESULT OnMasterReset(WPARAM wParam, LPARAM lParam);
237
238 DECLARE_MESSAGE_MAP()
239
240
241 CToolBar m_wndToolBar; ///< Application toolbar.
242 MeaStatusBar m_wndStatusBar; ///< Application status bar.
243 AppView m_wndView; ///< Application view window.
244 bool m_alwaysVisible; ///< Indicates if the application window is always top window on the desktop.
245 bool m_profileToolbarVisible; ///< Indicates if the persisted preference is for the toolbar to be displayed.
246 bool m_profileStatusbarVisible; ///< Indicates if the persisted preference if for the status bar to be displayed.
247 bool m_toolbarVisible; ///< Indicates if the toolbar is currently visible.
248 bool m_statusbarVisible; ///< Indicates if the status bar is currently visible.

250private:

255 bool CreateToolbar();
256
257 /// Selects whether to show the toolbar.
258 /// @param enable [in] Indicates whether the toolbar should be visible.
259 ///
260 void ViewToolbar(bool enable);
261
262 /// Selects whether to show the status bar.
263 /// @param enable [in] Indicates whether the status bar should be visible.
264 ///
265 void ViewStatusbar(bool enable);
266
267 /// Opens the specified URL in a web browser.
268 ///
269 /// @param urlId String resource ID for the URL
270 /// @return true if the browser was opened.

272 bool OpenUrl(int urlId);

274
275 bool m_newInstall; ///< Indicates if this is the first time the application has been run.

278 int m_appWidth; ///< Total width of the application, in pixels.
279 CString m_cmdLineFile; ///< Filename specified on the command line, if any.
