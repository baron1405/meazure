35/// Grid tool configuration dialog. This dialog allows the user to
36/// set the grid spacing. The grid spacing can be set independently
37/// in the vertical and horizontal directions.
38///
39class MeaGridDialog : public CDialog {
40
41public:
42 /// Constructs a new instance of the Grid tool dialog. To display
43 /// the dialog call its DoModal method.
44 ///
45 /// @param tool [in] Grid tool being configured by the dialog.
46 ///
47 explicit MeaGridDialog(MeaGridTool* tool);
48
49 /// Destroy the Grid tool dialog.
50 ///
51 virtual ~MeaGridDialog();
52
53 /// Called immediately before the dialog is displayed in response
54 /// to a DoModal call.
55 ///
56 /// @return <b>TRUE</b> to indicate Windows should set the input focus.
57 ///
58 virtual BOOL OnInitDialog() override;
59
60protected:
61 /// Called when the user clicks the checkbox that controls whether
62 /// the vertical and horizontal grid spacings are set independently.
63 ///
64 afx_msg void OnGridLink();
65
66 /// Called when the value in the horizontal grid spacing text field
67 /// is changed.
68 ///
69 afx_msg void OnHSpaceChange();
70
71 /// Called when the value in the vertical grid spacing text field
72 /// is changed.
73 ///
74 afx_msg void OnVSpaceChange();
75
76 DECLARE_MESSAGE_MAP()
77
78 /// Called by Windows to perform Dynamic Data Exchange between
79 /// the dialog's controls and their corresponding member variables.
80 ///
81 /// @param pDX [in] Object containing the dialog data exchange
82 /// context.
83 ///
84 virtual void DoDataExchange(CDataExchange* pDX) override;
85
86 /// Called when the dialog's OK button is pressed. The method
87 /// verifies that the spacing fields contain valid numerical
88 /// values. If not, a dialog is displayed, the erroneous field
89 /// is selected and the dialog is not dismissed.
90 ///
91 virtual void OnOK() override;
92
93 /// Called when the dialog's Cancel button is pressed. The method
94 /// restores the original spacing values.
95 ///
96 virtual void OnCancel() override;
97
98private:
99 /// Updates the dialog's controls to reflect the member variable
100 /// values.
101 ///
102 void UpdateDisplay();
103
104 /// Reads the content of the specified text field and attempts
105 /// to convert it to a numerical value. In order to succeed, the
106 /// contents must represent a number and the number must be in the
107 /// range:
108 ///
109 /// MeaGridTool::kDefMinSpacing \f$\leq\f$ value \f$\leq\f$ MeaGridTool::kDefMaxSpacing
110 ///
111 /// @param fieldId [in] Species the text field to read
112 /// @param value [out] Field contents converted to a numerical value
113 ///
114 /// @return <b>true</b> if the text field's contents were successfully converted
115 /// and the value was within the allowable range. The value parameter
116 /// is changed only if the method returns <b>true</b>.
117 ///
118 bool GetFieldValue(UINT fieldId, long& value) const;
119
120 MeaGridTool* m_tool; ///< Grid tool being configured by the dialog
121
122 MeaNumberField m_hSpaceField; ///< Horizontal grid spacing text field
123 MeaNumberField m_vSpaceField; ///< Vertical grid spacing text field
124
125 CSpinButtonCtrl* m_hSpaceSpin; ///< Horizontal spacing increment / decrement control
126 CSpinButtonCtrl* m_vSpaceSpin; ///< Vertical spacing increment / decrement spacing
127
128 bool m_origLinked; ///< State of the lock checkbox before the dialog was displayed
129 bool m_linked; ///< Current state of the lock checkbox
130 SIZE m_origSpacing; ///< Grid spacing before the dialog was displayed
131 SIZE m_spacing; ///< Current grid spacing
132};

135/// Grid overlay tool. This tool overlays a rectangular grid on
136/// the screen. If there are multiple monitors, the grid is overlayed
137/// continuously across all monitors. The grid spacing can be changed
138/// using the MeaGridDialog.
139///
140class MeaGridTool : public MeaTool {
141
142public:
143 // Defaults
144 //
145 static constexpr int kDefMinSpacing { 10 }; ///< Minimum allowable grid spacing in pixels
146 static constexpr int kDefMaxSpacing { 4000 }; ///< Maximum allowable grid spacing in pixels
147 static constexpr int kDefDefaultSpacing { 100 }; ///< Default grid spacing in pixels
148 static constexpr bool kDefGrid { false }; ///< Default grid display state
149 static constexpr bool kDefLinked { true }; ///< Default spacing link state
150
151 static const CString kToolName; ///< "GridTool"
152
153 /// Constructs a new instance of the Grid overlay tool. To use
154 /// a newly constructed Grid tool, the Enable() method must be called.
155 ///
156 /// @param mgr [in] Tool manager parent of the Grid tool.
157 /// @param screenProvider [in] Screen information provider
158 /// @param unitsProvider [in] Units information provider
159 ///
160 explicit MeaGridTool(MeaToolMgr& mgr, const MeaScreenProvider& screenProvider,
161 const MeaUnitsProvider& unitsProvider);
162
163 /// Hides and then destroys the Grid tool.
164 ///
165 virtual ~MeaGridTool();
166
167 /// Displays the Grid tool.
168 ///
169 virtual void Enable() override;
170
171 /// Hides the Grid tool.
172 ///
173 virtual void Disable() override;
174
175 /// Persists the state of the tool to the specified profile object.
176 ///
177 /// @param profile [in] The destination for the state information.
178 /// Typically an MeaFileProfile or MeaRegistryProfile
179 /// object.
180 ///
181 virtual void SaveProfile(MeaProfile& profile) override;
182
183 /// Restores the state of the tool from the specified profile object.
184 ///
185 /// @param profile [in] The source for the state information. Typically
186 /// an MeaFileProfile or MeaRegistryProfile object.
187 ///
188 virtual void LoadProfile(MeaProfile& profile) override;
189
190 /// Resets the tool to its default state.
191 ///
192 virtual void MasterReset() override;
193
194 /// Draws the horizontal and vertical grid lines, if the tool
195 /// is enabled.
196 ///
197 /// @param reason [in] Reason why the update has been requested
198 ///
199 virtual void Update(MeaUpdateReason reason) override;
200
201 /// Draws the horizontal grid lines.
202 ///
203 void UpdateH();
204
205 /// Draws the vertical grid lines.
206 ///
207 void UpdateV();
208
209 /// Returns the name of the tool. Each tool has a unique name
210 /// which is used to identify the tool in profiles and position
211 /// logs.
212 ///
213 /// @return Name of the tool ("GridTool").
214 ///
215 virtual CString GetToolName() const override;
216
217 /// Returns the tool's current position which, for the Grid tool,
218 /// is always the position (0, 0).
219 ///
220 /// @return The position (0, 0)
221 ///
222 virtual const POINT& GetPosition() const override;
223
224 /// Called to notify the tool that the colors in the MeaColors class
225 /// have been changed. The grid lines are redrawn in their new colors.
226 ///
227 virtual void ColorsChanged() override;
228
229 /// Sets the horizontal and vertical grid spacing.
230 ///
231 /// @param spacing [in] Horizontal and vertical grid spacing, in pixels
232 ///
233 void SetGridSpacing(const SIZE& spacing);
234
235 /// Gets the horizontal and vertical grid spacing.
236 ///
237 /// @return Horizontal and vertical grid spacing, in pixels.
238 ///
239 const SIZE& GetGridSpacing() const { return m_gridSpacing; }
240
241 /// Indicates if the grid spacing is linked so that the
242 /// horizontal and vertical grid spacing is the same.
243 ///
244 /// @return <b>true</b> if the horizontal and vertical
245 /// grid spacing are linked.
246 ///
247 bool IsLinked() const { return m_linked; }
248
249 /// Sets the grid spacing link state.
250 ///
251 /// @param linked [in] <b>true</b> to link the spacing
252 ///
253 void SetLinked(bool linked) { m_linked = linked; }
254
255private:
256 typedef std::list<MeaLine*> LineList; ///< List of grid lines
257
258 /// Grid line direction
259 ///
260 enum LineDir {
261 HDir = 0,
262 VDir = 1
263 };
264
265
266 /// Creates the vertical or horizontal lines for the grid as needed
267 /// and positions them on the grid according to the currently set
268 /// grid spacing. Since there are many lines on a grid and line
269 /// creation is expensive, lines are created on demand and
270 /// reused. The lines are destroyed only when the tool is
271 /// destroyed.
272 ///
273 /// @param dir [in] Direction of the lines to create
274 ///
275 void SetLines(LineDir dir);
276
277 /// Displays the vertical or horizontal grid lines.
278 ///
279 /// @param dir [in] Direction of the lines to display
280 ///
281 void ShowLines(LineDir dir);
282
283 /// Hides the vertical or horizontal grid lines
284 ///
285 /// @param dir [in] Direction of the lines to display
286 ///
287 void HideLines(LineDir dir) const;
288
289 /// Hides and then destroys the lines in the specified line list.
290 /// The line list is then cleared.
291 ///
292 /// @param list [in] List of lines whose contents is to be
293 /// destroyed. The list is empty following the
294 /// call to this method.
295 ///
296 void DeleteLines(LineList& list) const;
297
298
299 CSize m_gridSpacing; ///< Vertical and horizontal grid spacing, in pixels
300 bool m_linked; ///< <b>true</b> if the vertical and horizontal grid spacings are linked (i.e. kept equal)
301 LineList m_hlineList; ///< List of horizontal grid lines
302 LineList m_vlineList; ///< List of vertical grid lines
303};
