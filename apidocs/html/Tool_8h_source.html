45/// Abstract base class for all measurement tools. Both radio and non-radio
46/// measurement tools inherit from this base class.
47///
48class MeaTool {
49
50public:
51 /// Base class constructor. Typically, to use a newly constructed tool,
52 /// the Enable() method must be called.
53 ///
54 /// @param mgr [in] Tool manager parent of the tool.
55 ///
56 explicit MeaTool(MeaToolMgr* mgr);
57
58 /// Destroys the tool.
59 ///
60 virtual ~MeaTool();
61
62 /// Typically used by derived classes to show the tool. In addition,
63 /// the first time this method is called, it typically creates the
64 /// components that comprise the tool.
65 ///
66 virtual void Enable();
67
68 /// Typically used by derived classes to hide the tool.
69 ///
70 virtual void Disable();
71
72 /// Used by derived classes to update the display of their data.
73 ///
74 /// @param reason [in] Reason why the update has been requested
75 ///
76 virtual void Update(MeaUpdateReason reason);
77
78 /// Indicates whether the derived tool is a radio tool and
79 /// cannot be enabled until any other enabled Radio tool
80 /// is disabled.
81 ///
82 /// @return Always <b>false</b> indicating that this class
83 /// and its derived classes are not radio tools unless
84 /// they have overridden this method.
85 ///
86 virtual bool IsRadioTool();
87
88 /// Sets the position of the tool. This method sets the position
89 /// one location component at a time.
90 ///
91 /// @param which [in] Position component
92 /// @param pixels [in] Position to set for the component, in pixels
93 ///
94 virtual void SetPosition(MeaFields which, int pixels);
95
96 /// Returns the tool's current position. The current position is
97 /// the position of the last crosshair entered or moved. The default
98 /// position is the location of point 1.
99 ///
100 /// @return Current position
101 ///
102 virtual const POINT& GetPosition() const = 0;
103
104 /// Increments the specified position field.
105 ///
106 /// @param which Position field to increment
107 ///
108 virtual void IncPosition(MeaFields which);
109
110 /// Decrements the specified position field.
111 ///
112 /// @param which Position field to decrement
113 ///
114 virtual void DecPosition(MeaFields which);
115
116 /// Returns the name of the tool. Each tool has a unique name
117 /// which is used to identify the tool in profiles and position
118 /// logs.
119 ///
120 /// @return Name of the tool
121 ///
122 virtual CString GetToolName() const = 0;
123
124 /// Persists the state of the tool to the specified profile object.
125 /// This base class implementation does nothing.
126 ///
127 /// @param profile [in] The destination for the state information.
128 /// Typically an MeaFileProfile or MeaRegistryProfile
129 /// object.
130 ///
131 virtual void SaveProfile(MeaProfile& profile);
132
133 /// Restores the state of the tool from the specified profile object.
134 /// This base class implementation does nothing.
135 ///
136 /// @param profile [in] The source for the state information. Typically
137 /// an MeaFileProfile or MeaRegistryProfile object.
138 ///
139 virtual void LoadProfile(MeaProfile& profile);
140
141 /// Resets the tool to its default state.
142 ///
143 virtual void MasterReset();
144
145 /// Called to notify the tool that the colors in the MeaColors class
146 /// have been changed. Typically a derived class redraws its components
147 /// with the new colors and opacities.
148 ///
149 virtual void ColorsChanged();
150
151 /// Indicates whether the tool is enabled. A tool is enabled when the
152 /// Enable() method is called and is disabled when the Disable() method
153 /// is called.
154 ///
155 /// @return <b>true</b> if the tool is enabled.
156 ///
157 bool IsEnabled() const { return m_enabled; }
158
159protected:
160 static POINT m_defaultPos; ///< A default position used by tools that do not have a defined position. This
161 ///< position is returned by the GetPosition() method
162
163 MeaToolMgr* m_mgr; ///< Tool manager
164 bool m_enabled; ///< Indicates whether the tool is enabled
165};
