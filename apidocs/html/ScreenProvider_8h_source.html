29/// Provides information on the display monitor(s).
30///
31struct MeaScreenProvider {
32
33 class Screen;
34
35 typedef std::map<HMONITOR, Screen*> Screens; ///< Maps a monitor handle to a screen object.
36 typedef Screens::const_iterator ScreenIter; ///< Constant iterator over the screens.
37
38 /// Used by the STL collections for ordering comparisons.
39 ///
40 struct less {
41 /// Compares two screen iterators.
42 /// @param lhs [in] Left hand side of the comparison.
43 /// @param rhs [in] Right hand side of the comparison.
44 /// @return <b>true</b> if lhs < rhs.
45 bool operator()(const ScreenIter& lhs, const ScreenIter& rhs) const {
46 return (*lhs).first < (*rhs).first;
47 }
48 };
