31/// Rectangle measurement tool. This tool provides a rectangle to measure
32/// an arbitrary rectangular area of the screen. Two crosshairs are provided
33/// one at the either end of the rectangle's diagonal (points 1 and 2).
34/// Each crosshair can be arbitrarily positioned, which results in moving
35/// the rectangle.
36///
37class MeaRectTool : public MeaRadioTool, public MeaCrossHairCallback
38{
39public:
40 static const CString kToolName; ///< "RectTool"
41
42 /// Constructs a new instance of a Rectangle measurement tool. To use
43 /// a newly constructed Rectangle tool, the Enable() method must be called.
44 /// Enable() will call the Create() method the first time it is called
45 /// if Create() has not previously been called.
46 ///
47 /// @param mgr [in] Tool manager parent of the Rectangle tool.
48 ///
49 explicit MeaRectTool(MeaToolMgr* mgr);
50
51 /// Hides and then destroys the Rectangle tool.

63 virtual void Disable();

70 virtual void Update(MeaUpdateReason reason);

73 /// Indicates whether the tool can define a rectangular region that

79 virtual bool HasRegion();

95 virtual void SaveProfile(MeaProfile& profile);

97 /// Restores the state of the tool from the specified profile object.

99 /// @param profile [in] The source for the state information. Typically

102 virtual void LoadProfile(MeaProfile& profile);
103
104
105 /// Displays the tool's crosshairs. If the Enable() method has not

107 ///
108 virtual void EnableCrosshairs();
109
110 /// Hides the tool's crosshairs.
111 ///
112 virtual void DisableCrosshairs();
113
114
115 /// Visually flashes the tool. The tool crosshairs are cycled
116 /// between a hilite color and their normal color a number of times.
117 /// Flashing the tool helps the user find it on the screen.
118 ///
119 virtual void Flash();
120
121 /// Visually strobes the tool. The tool crosshairs are cycled
122 /// between a hilite color and their normal color once. In addition,
123 /// the data windows attached to the crosshairs are cycled between
124 /// hidden and visible. Strobing the tool is done to visually confirm
125 /// region capture and position recording.
126 ///
127 virtual void Strobe();
128
129
130 /// Sets the position of the tool. This method sets the position
131 /// one location component at a time. For example, the x position
132 /// of the point 1 crosshair is specified by calling this method
133 /// with the new position and specifying the MeaX1Field for the
134 /// <i>which</i> parameter.
135 ///
136 /// @param which [in] Position component
137 /// @param pixels [in] Position to set for the component, in pixels
138 ///
139 virtual void SetPosition(MeaFields which, int pixels);
140
141 /// Sets the position of the tool. This method is called by the
142 /// position log manager to play back a tool's position that was
143 /// previously recorded using the GetPosition method.
144 ///
145 /// @param points [in] Map of positions for the tool's crosshairs.
146 ///
147 virtual void SetPosition(const PointMap& points);
148
149 /// Returns the tool's current position. The current position is
150 /// the position of the last crosshair entered or moved. The default
151 /// position is the location of point 1.
152 ///
153 /// @return Current position
154 ///
155 virtual const POINT& GetPosition() const;
156
157 /// Records the position of the tool. This method is called by the

159 ///
160 /// @param position [in] The tool's current position is recorded into
161 /// the position log manager's position object.
162 ///
163 virtual void GetPosition(MeaPositionLogMgr::Position& position) const;
164
165
166 /// Returns the name of the tool. Each tool has a unique name

169 ///
170 /// @return Name of the tool ("RectTool").
171 ///
172 virtual CString GetToolName() const;
173
174 /// Returns a resource ID for the string used to label this

177 /// @return Resource ID for the label.

179 virtual UINT GetLabelId() const;

181
182 /// Called to notify the tool that the colors in the MeaColors class

186 virtual void ColorsChanged();
187
188
189 /// Called when the pointer enters one of the tool's crosshairs.
190 /// The crosshair entered is identified by the MeaCrossHairCallback::CHInfo
191 /// structure passed to the method.

193 /// @param info [in] Crosshair information structure
194 ///
195 virtual void OnCHEnter(const CHInfo *info);
196
197 /// Called when the pointer leaves one of the tool's crosshairs.
198 /// The crosshair exited is identified by the MeaCrossHairCallback::CHInfo
199 /// structure passed to the method.
200 ///
201 /// @param info [in] Crosshair information structure
202 ///
203 virtual void OnCHLeave(const CHInfo *info);
204
205 /// Called when the user attempts to drag one of the tool's
206 /// crosshairs. The crosshair to be moved is identified by the
207 /// MeaCrossHairCallback::CHInfo structure passed to the method.
208 /// The info parameter indicates where the crosshair is to be
209 /// moved and the tool actually moves the crosshair to the new
210 /// position. This allows the tool to perform vertical or horizontal
211 /// locking and move the entire tool depending on which keyboard
212 /// modifiers are pressed during the drag.
213 ///
214 /// @param info [in] Crosshair information structure
215 ///
216 virtual void OnCHMove(const CHInfo *info);
217
218private:
219 static const UINT kPoint1Id; ///< ID for the point 1 crosshair
220 static const UINT kPoint2Id; ///< ID for the point 2 crosshair
221

228 ///

234 void SetPosition();

242 MeaCrossHair m_point2CH; ///< Crosshair for one corner of the rectangle diagonally opposed to #m_point1CH
243 MeaRectangle m_rectangle; ///< Rectangle connecting #m_point1 and #m_point2
244 MeaDataWin m_dataWin1; ///< Data window tooltip for point 1
245 MeaDataWin m_dataWin2; ///< Data window tooltip for point 2
246};
