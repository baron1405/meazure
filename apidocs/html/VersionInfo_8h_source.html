28/// Convenience class for obtaining version number information for the
29/// applications.
30///
31class MeaVersionInfo : public MeaSingleton_T<MeaVersionInfo> {
32
33public:
34 /// Constructs an instance of the version number information class.
35 ///

38 /// Destroys an instance of the class.
39 ///

42 /// Retrieves the product version number.
43 /// @return Product version number expressed as string of the form
44 /// major.minor or major.minor.fix (if fix > 0).
45 CString GetProductVersion() const;
46
47 /// Returns the product major version number.
48 /// @return Major version number.
49 int GetProductMajor() const { return HIWORD(m_info->dwProductVersionMS); }
50
51 /// Returns the product minor version number.
52 /// @return Minor version number.
53 int GetProductMinor() const { return LOWORD(m_info->dwProductVersionMS); }
54
55 /// Returns the product fix number.
56 /// @return Fix number.
57 int GetProductFix() const { return HIWORD(m_info->dwProductVersionLS); }
58
59 /// Returns the product build number.
60 /// @return Build number.
61 int GetProductBuild() const { return LOWORD(m_info->dwProductVersionLS); }
62
63
64 /// Returns the version number of the profile file format.
65 /// @return Version number of the profile file format.
66 int GetProfileFileMajor() const { return 2; }
67
68 /// Returns the version number of the position log file format.
69 /// @return Version number of the position log file format.
70 int GetLogFileMajor() const { return 1; }
71
72private:
73 /// The class does not have copy semantics so this method is
74 /// purposely undefined.
75 MeaVersionInfo(const MeaVersionInfo&);
76
77 /// The class does not have assignment semantics so this method is
78 /// purposely undefined.
79 MeaVersionInfo& operator=(const MeaVersionInfo&);
80
81 void* m_data; ///< Version information buffer.
82 VS_FIXEDFILEINFO* m_info; ///< Version information.
83};
