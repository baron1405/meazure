31/// Crosshair activity callback mixin class. A class wishing to receive
32/// notification of crosshair events (e.g. movement), can inherit from
33/// this class and override the On* methods of interest.
34///
35class MeaCrossHairCallback {
36
37public:
38 /// Callback information structure. A callback information structure
39 /// is initialized and passed as the parameter in each callback
40 /// method.
41 ///
42 struct CHInfo {
43 MeaCrossHair* ch; ///< Pointer to crosshair object that initiated the callback
44 CPoint centerPoint; ///< Location of the crosshair center point, in pixels
45 UINT id; ///< User defined ID for the crosshair
46 UINT flags; ///< Virtual key flags including modifier keys (e.g. MK_CONTROL, MK_SHIFT)
47 };
48
49
50 /// Constructs a callback object.
51 ///
52 MeaCrossHairCallback() {}
53
54 /// Destroys a callback object.
55 ///
56 virtual ~MeaCrossHairCallback() {}
57
58 /// Called when the left mouse button is pressed while
59 /// the pointer is positioned over the crosshair.
60 ///
61 /// @param info [in] Crosshair information structure
62 ///
63 virtual void OnCHSelect(const CHInfo* info);
64
65 /// Called when the left mouse button is release while
66 /// the pointer is positioned over the crosshair.
67 ///
68 /// @param info [in] Crosshair information structure
69 ///
70 virtual void OnCHDeselect(const CHInfo* info);
71
72 /// Called when the user moves the mouse pointer over
73 /// the crosshair while the left mouse button is pressed
74 /// (i.e. a drag operation). It is the responsibility of
75 /// the callback method to actually moved the crosshair to
76 /// its new position as specified in the callback information
77 /// structure.
78 ///
79 /// @param info [in] Crosshair information structure
80 ///
81 virtual void OnCHMove(const CHInfo* info);
82
83 /// Called when the pointer enters the crosshair window.
84 ///
85 /// @param info [in] Crosshair information structure
86 ///
87 virtual void OnCHEnter(const CHInfo* info);
88
89 /// Called when the pointer leaves the crosshair window.
90 ///
91 /// @param info [in] Crosshair information structure
92 ///
93 virtual void OnCHLeave(const CHInfo* info);
94};

97/// A crosshair element. A crosshair consists of four triangular
98/// window segments arranged vertically and horizontally such
99/// that each triangle is aimed at a center point. The crosshair
100/// is used by a number of the measurement tools to identify their
101/// measurement points and to allow the user to perform measurements
102/// by dragging the crosshairs using the pointer.
103///
104class MeaCrossHair : public MeaGraphic {
105
106public:
107 static constexpr int kFlashCount { 9 }; ///< Number of times to flash the crosshair when Flash() is called
108 static constexpr int kStrobeCount { 1 }; ///< Number of times to flash the crosshair when Strobe() is called
109
110
111 /// Constructs a crosshair. Prior to displaying the crosshair with
112 /// the Show() method, the Create() method must be called to
113 /// create the crosshair's window.
114 ///
115 MeaCrossHair();
116
117 /// Destroys the circle.
118 ///
119 virtual ~MeaCrossHair();
120
121 /// Creates the window for the crosshair. This method must be called
122 /// before the crosshair can be displayed and before any operation that
123 /// attempts to manipulate the crosshair's window.
124 ///
125 /// @param borderColor [in] The color for the crosshair border lines. The
126 /// border lines are drawn around each triangular section
127 /// of the crosshair.
128 /// @param backColor [in] Background color for the crosshair. Each
129 /// triangular section of the crosshair is drawn in the
130 /// background color. The crosshair is displayed in the
131 /// background color when the pointer is not over the
132 /// crosshair.
133 /// @param hiliteColor [in] Background color for the crosshair when the
134 /// pointer is over the crosshair.
135 /// @param opacity [in] Opacity value for the crosshair expressed as
136 /// a value between 0 (transparent) and 255 (opaque)
137 /// inclusive.
138 /// @param callback [in] A class derived from MeaCrossHairCallback that
139 /// wants to receive notification of crosshair events.
140 /// Can be set to <b>nullptr</b> if no callback is desired.
141 /// @param parent [in] Parent window or <b>nullptr</b> if the crosshair does
142 /// not have an immediate parent. If a parent window is
143 /// specified, the crosshair is created as a child window
144 /// (<b>WS_CHILD</b>). Otherwise the crosshair is created
145 /// as a popup window (<b>WS_POPUP</b>) and is assigned
146 /// the main window as a parent.
147 /// @param toolTipId [in] String resource ID for the tooltip associated
148 /// with the crosshair.
149 /// @param id [in] Caller-defined ID for the crosshair.
150 ///
151 /// @return <b>true</b> if the window is created successfully.
152 ///
153 bool Create(COLORREF borderColor, COLORREF backColor, COLORREF hiliteColor,
154 BYTE opacity, MeaCrossHairCallback* callback,
155 const CWnd* parent = nullptr, UINT toolTipId = 0xffff,
156 UINT id = 0xffff);
157
158 /// Visually flashes the crosshair by cycling its background
159 /// between normal and hilited the number of times specified.
160 ///
161 /// @param flashCount [in] Number of times to cycle the
162 /// crosshair background.
163 ///
164 void Flash(int flashCount = kFlashCount);
165
166 /// Sets the position of the crosshair.
167 ///
168 /// @param center [in] Position for the center of the crosshair, in pixels.
169 ///
170 void SetPosition(const POINT& center);
171
172 /// Sets the colors for the crosshair.
173 ///
174 /// @param borderColor [in] Color for the crosshair border lines
175 /// @param backColor [in] Normal background color
176 /// @param hiliteColor [in] Highlight background color
177 ///
178 void SetColors(COLORREF borderColor, COLORREF backColor, COLORREF hiliteColor);
179
180 /// Sets the opacity for the crosshair.
181 ///
182 /// @param opacity [in] Opacity value between 0 (transparent)
183 /// and 255 (opaque) inclusive
184 ///
185 void SetOpacity(BYTE opacity);
186
187 /// Relays messages to the crosshair's tooltip.
188 ///
189 /// @param pMsg [in] Window message to relay
190 ///
191 /// @return TRUE if message was processed.
192 ///
193 virtual BOOL PreTranslateMessage(MSG* pMsg);
194
195protected:
196 DECLARE_MESSAGE_MAP()
197
198 /// Called when the crosshair window is about to be destroyed.
199 /// If the crosshair was created with a parent window specified,
200 /// a number of bitmaps were created to simulate an opacity effect.
201 /// This method deletes those bitmaps.
202 ///
203 afx_msg void OnDestroy();
204
205 /// Called when the crosshair needs repainting.
206 ///
207 afx_msg void OnPaint();
208
209 /// Called when the crosshair is dragged. The method
210 /// initializes the crosshair information structure and calls
211 /// the callback MeaCrossHairCallback::OnCHMove method, if a
212 /// callback has been specified.
213 ///
214 /// @param flags [in] Virtual key state
215 /// @param point [in] Location of the pointer, in pixels
216 ///
217 afx_msg void OnMouseMove(UINT flags, CPoint point);
218
219 /// Called when the pointer leaves the crosshair window. The
220 /// method initializes the crosshair information structure
221 /// and calls the callback MeaCrossHairCallback::OnCHLeave,
222 /// if a callback has been specified.
223 ///
224 /// @param wParam [in] Not used
225 /// @param lParam [in] Not used
226 ///
227 /// @return Always zero.
228 ///
229 afx_msg LRESULT OnMouseLeave(WPARAM wParam, LPARAM lParam);
230
231 /// Called when the left mouse button is pressed while the
232 /// pointer is over the crosshair window.
233 ///
234 /// @param flags [in] Virtual key state
235 /// @param point [in] Location of the pointer, in pixels
236 ///
237 afx_msg void OnLButtonDown(UINT flags, CPoint point);
238
239 /// Called when the left mouse button is released while the
240 /// pointer is captured by the crosshair window.
241 ///
242 /// @param flags [in] Virtual key state
243 /// @param point [in] Location of the pointer, in pixels
244 ///
245 afx_msg void OnLButtonUp(UINT flags, CPoint point);
246
247 /// Called when the crosshair is cycling between normal
248 /// and highlighted display as a result of a call to the Flash()
249 /// method. This method redraws the crosshair in the
250 /// appropriate background color.
251 ///
252 /// @param timerId [in] ID for the timer that has expired.
253 ///
254 afx_msg void OnTimer(UINT_PTR timerId);
255
256private:
257 /// Specifies how to draw the crosshair (i.e. normally or
258 /// inverted).
259 ///
260 enum DrawState {
261 Normal, ///< Draw the crosshair in its normal appearance.
262 Inverted ///< Draw the crosshair in its inverted appearance.
263 };
264
265
266 /// Given the center of the crosshair, returns the corresponding
267 /// upper left corner.
268 ///
269 /// @param center [in] Location of the center of the crosshair, in pixels
270 ///
271 /// @return Location of the upper left corner of the crosshair window,
272 /// in pixels.
273 ///
274 static CPoint GetLeftTop(const CPoint& center) { return center - m_halfSize; }
275
276
277 static SIZE m_size; ///< Width and height of the crosshair, in pixels
278 static SIZE m_halfSize; ///< Half the width and height of the crosshair, in pixels
279 static SIZE m_spread; ///< Half the length of the base of a triangular section of the crosshair, in pixels
280 static UINT m_flashInterval; ///< Number of milliseconds to hold each display state while flashing the crosshair
281 static int* m_numCoords; ///< Number of coordinates in each rectangle making up part of the crosshair (4)
282
283
284 /// Forms the window into the shape of the crosshair. A series of
285 /// rectangular regions are aggregated together to form the four
286 /// petals of the crosshair.
287 ///
288 void SetRegion();
289
290 /// Draws the crosshair background and border. Before this method
291 /// is called, the shape of the crosshair has already been set by
292 /// the SetRegion() method. The DrawCrossHair() method simply draws
293 /// the appropriate background color and frames the window's region.
294 /// The background color depends on whether the crosshair is in a
295 /// highlight state or not.
296 ///
297 /// @param dc [in] Device context into which the crosshair is drawn
298 ///
299 void DrawCrossHair(CDC& dc);
300
301 /// Deletes the brushes used to draw the crosshair's background and
302 /// border.
303 ///
304 void DestroyColors();
305
306 /// Fills in the crosshair information structure with the current
307 /// location of the pointer and the state of the keyboard modifiers.
308 ///
309 /// @param chs [in, out] Crosshair information structure to fill
310 /// @param flags [in] State of the keyboard modifier keys
311 /// @param point [in] Current location of the pointer, in pixels
312 ///
313 void FillInfo(MeaCrossHairCallback::CHInfo& chs, UINT flags, const CPoint& point);
314
315
316 MeaCrossHairCallback* m_callback; ///< Object to call for crosshair events
317 bool m_mouseCaptured; ///< Indicates if the pointer is captured
318 bool m_mouseOver; ///< Indicates if the pointer is over the crosshair
319 CBrush* m_backBrush; ///< Brush to paint the normal crosshair background
320 CBrush* m_borderBrush; ///< Brush to paint the crosshair border
321 CBrush* m_hiliteBrush; ///< Brush to paint the highlighted crosshair background
322 CSize m_pointerOffset; ///< Offset from the pointer location in the window to the center of the crosshair
323 DrawState m_drawState; ///< Indicates how the crosshair should be drawn
324 int m_flashCount; ///< Number of times to flash the crosshair
325 CToolTipCtrl m_toolTip; ///< Tooltip associated with the crosshair
326 BYTE m_opacity; ///< Opacity of the crosshair (0 - transparent, 255 - opaque)
327 CDC m_chDC; ///< Crosshair device context
328 CDC m_backDC; ///< Background device context for alpha blending when the crosshair is a child window
329 CBitmap m_chBitmap; ///< Bitmap into which to draw the crosshair when it is a child window
330 CBitmap* m_origCHBitmap; ///< Original bitmap for the crosshair
331 CBitmap m_backBitmap; ///< Bitmap for the background when alpha blending when the crosshair is a child window
332 CBitmap* m_origBackBitmap; ///< Original background bitmap
333};
