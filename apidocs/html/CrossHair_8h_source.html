34/// Crosshair activity callback mixin class. A class wishing to receive
35/// notification of crosshair events (e.g. movement), can inherit from
36/// this class and override the On* methods of interest.
37///
38class MeaCrossHairCallback {
39
40public:
41 /// Callback information structure. A callback information structure
42 /// is initialized and passed as the parameter in each callback
43 /// method.
44 ///
45 struct CrossHairInfo {
46
47 CrossHairInfo(MeaCrossHair* ch, UINT chId, UINT chFlags, const CPoint& centerPoint) :
48 crosshair(ch), id(chId), flags(chFlags), center(centerPoint) {}
49
50 MeaCrossHair* crosshair; ///< Pointer to crosshair object that initiated the callback
51 CPoint center; ///< Location of the crosshair center point, in pixels
52 UINT id; ///< User defined ID for the crosshair
53 UINT flags; ///< Virtual key flags including modifier keys (e.g. MK_CONTROL, MK_SHIFT)
54 };

96/// A crosshair element. A crosshair consists of four triangular
97/// window segments arranged vertically and horizontally such
98/// that each triangle is aimed at a center point. The crosshair
99/// is used by a number of the measurement tools to identify their
100/// measurement points and to allow the user to perform measurements
101/// by dragging the crosshairs using the pointer.
102///
103class MeaCrossHair : public MeaGraphic {
104
105public:
106 static constexpr int kFlashCount { 9 }; ///< Number of times to flash the crosshair when Flash() is called
107 static constexpr int kStrobeCount { 1 }; ///< Number of times to flash the crosshair when Strobe() is called
108
109
110 /// Constructs a crosshair. Prior to displaying the crosshair with
111 /// the Show() method, the Create() method must be called to
112 /// create the crosshair's window.
113 ///
114 /// @param screenProvider [in] Screen information provider
115 /// @param unitsProvider [in] Units information provider
116 ///
117 MeaCrossHair(const MeaScreenProvider& screenProvider, const MeaUnitsProvider& unitsProvider);
118
119 /// Destroys the circle.
120 ///
121 virtual ~MeaCrossHair();
122
123 /// Creates the window for the crosshair. This method must be called
124 /// before the crosshair can be displayed and before any operation that
125 /// attempts to manipulate the crosshair's window.
126 ///
127 /// @param borderColor [in] The color for the crosshair border lines. The
128 /// border lines are drawn around each triangular section
129 /// of the crosshair.
130 /// @param backColor [in] Background color for the crosshair. Each
131 /// triangular section of the crosshair is drawn in the
132 /// background color. The crosshair is displayed in the
133 /// background color when the pointer is not over the
134 /// crosshair.
135 /// @param hiliteColor [in] Background color for the crosshair when the
136 /// pointer is over the crosshair.
137 /// @param opacity [in] Opacity value for the crosshair expressed as
138 /// a value between 0 (transparent) and 255 (opaque)
139 /// inclusive.
140 /// @param callback [in] A class derived from MeaCrossHairCallback that
141 /// wants to receive notification of crosshair events.
142 /// Can be set to <b>nullptr</b> if no callback is desired.
143 /// @param parent [in] Parent window or <b>nullptr</b> if the crosshair does
144 /// not have an immediate parent. If a parent window is
145 /// specified, the crosshair is created as a child window
146 /// (<b>WS_CHILD</b>). Otherwise the crosshair is created
147 /// as a popup window (<b>WS_POPUP</b>) and is assigned
148 /// the main window as a parent.
149 /// @param toolTipId [in] String resource ID for the tooltip associated
150 /// with the crosshair.
151 /// @param id [in] Caller-defined ID for the crosshair.
152 ///
153 /// @return <b>true</b> if the window is created successfully.
154 ///
155 bool Create(COLORREF borderColor, COLORREF backColor, COLORREF hiliteColor,
156 BYTE opacity, MeaCrossHairCallback* callback,
157 const CWnd* parent = nullptr, UINT toolTipId = 0xffff,
158 UINT id = 0xffff);
159
160 /// Visually flashes the crosshair by cycling its background
161 /// between normal and hilited the number of times specified.
162 ///
163 /// @param flashCount [in] Number of times to cycle the
164 /// crosshair background.
165 ///
166 void Flash(int flashCount = kFlashCount);
167
168 /// Sets the position of the crosshair.
169 ///
170 /// @param center [in] Position for the center of the crosshair, in pixels.
171 ///
172 void SetPosition(const POINT& center);
173
174 /// Sets the colors for the crosshair.
175 ///
176 /// @param borderColor [in] Color for the crosshair border lines
177 /// @param backColor [in] Normal background color
178 /// @param hiliteColor [in] Highlight background color
179 ///
180 void SetColors(COLORREF borderColor, COLORREF backColor, COLORREF hiliteColor);
181
182 /// Sets the opacity for the crosshair.
183 ///
184 /// @param opacity [in] Opacity value between 0 (transparent)
185 /// and 255 (opaque) inclusive
186 ///
187 void SetOpacity(BYTE opacity);
188
189 /// Relays messages to the crosshair's tooltip.
190 ///
191 /// @param pMsg [in] Window message to relay
192 ///
193 /// @return TRUE if message was processed.
194 ///
195 virtual BOOL PreTranslateMessage(MSG* pMsg);

197protected:
198 DECLARE_MESSAGE_MAP()

200 /// Called when the crosshair window is about to be destroyed.
201 /// If the crosshair was created with a parent window specified,

205 afx_msg void OnDestroy();
206
207 /// Called when the crosshair needs repainting.
208 ///
209 afx_msg void OnPaint();

211 /// Called when the crosshair is dragged. The method
212 /// initializes the crosshair information structure and calls
213 /// the callback MeaCrossHairCallback::OnCHMove method, if a
214 /// callback has been specified.
215 ///
216 /// @param flags [in] Virtual key state
217 /// @param point [in] Location of the pointer, in pixels
218 ///
219 afx_msg void OnMouseMove(UINT flags, CPoint point);
220
221 /// Called when the pointer leaves the crosshair window. The
222 /// method initializes the crosshair information structure

225 ///
226 /// @param wParam [in] Not used
227 /// @param lParam [in] Not used
228 ///
229 /// @return Always zero.
230 ///
231 afx_msg LRESULT OnMouseLeave(WPARAM wParam, LPARAM lParam);
232
233 /// Called when the left mouse button is pressed while the
234 /// pointer is over the crosshair window.
235 ///
236 /// @param flags [in] Virtual key state
237 /// @param point [in] Location of the pointer, in pixels
238 ///
239 afx_msg void OnLButtonDown(UINT flags, CPoint point);
240
241 /// Called when the left mouse button is released while the
242 /// pointer is captured by the crosshair window.
243 ///
244 /// @param flags [in] Virtual key state
245 /// @param point [in] Location of the pointer, in pixels
246 ///
247 afx_msg void OnLButtonUp(UINT flags, CPoint point);
248
249 /// Called when the crosshair is cycling between normal
250 /// and highlighted display as a result of a call to the Flash()
251 /// method. This method redraws the crosshair in the
252 /// appropriate background color.
253 ///
254 /// @param timerId [in] ID for the timer that has expired.
255 ///
256 afx_msg void OnTimer(UINT_PTR timerId);

258private:
259 /// Specifies how to draw the crosshair (i.e. normally or
260 /// inverted).
261 ///
262 enum DrawState {
263 Normal, ///< Draw the crosshair in its normal appearance.
264 Inverted ///< Draw the crosshair in its inverted appearance.
265 };

270 static constexpr std::array<int, kTotalLayers> MakeNumCoords() noexcept {
271 std::array<int, kTotalLayers> arr {};
272 // The fill method is not constexpr until C++20
273 for (int i = 0; i < kTotalLayers; i++) {
274 arr[i] = 4;
275 }
276 return arr;
277 }

345 ///< is a child window

347};
