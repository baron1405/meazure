51BOOL (WINAPI* g_pfnIsThemeActive)(void) = NULL;
52HTHEME (WINAPI* g_pfnOpenThemeData)(HWND, LPCWSTR) = NULL;
53HRESULT (WINAPI* g_pfnCloseThemeData)(HTHEME) = NULL;

56BOOL g_fThemeInitDone = FALSE;

58
59BOOL InitThemeStubs(void)
60{
61 HMODULE hUXTheme;
62
63 if (g_fThemeInitDone) {
64 return g_pfnIsThemeActive != NULL;
65 }
66
67 hUXTheme = GetModuleHandle(TEXT("UXTHEME"));
68 if ((hUXTheme != NULL) &&
69 (*(FARPROC*)&g_pfnIsThemeActive = GetProcAddress(hUXTheme, "IsThemeActive")) != NULL &&
70 (*(FARPROC*)&g_pfnOpenThemeData = GetProcAddress(hUXTheme, "OpenThemeData")) != NULL &&
71 (*(FARPROC*)&g_pfnCloseThemeData = GetProcAddress(hUXTheme, "CloseThemeData")) != NULL &&
72 (*(FARPROC*)&g_pfnDrawThemeBackground = GetProcAddress(hUXTheme, "DrawThemeBackground")) != NULL &&
73 (*(FARPROC*)&g_pfnDrawThemeEdge = GetProcAddress(hUXTheme, "DrawThemeEdge")) != NULL
74 ) {
75 g_fThemeInitDone = TRUE;
76 return TRUE;
77 } else {
78 g_pfnIsThemeActive = NULL;
79 g_pfnOpenThemeData = NULL;
80 g_pfnCloseThemeData = NULL;
81 g_pfnDrawThemeBackground = NULL;
82 g_pfnDrawThemeEdge = NULL;
83 g_fThemeInitDone = TRUE;
84 return FALSE;
85 }
86}
