33/// This class writes pretty printed XML in UTF-8 encoding. Because this class serves only the needs of this
34/// application, it does not support XML namespaces. Typical usage:</p>
35/// <ol>
36/// <li>Create an MeaXMLWriter specifying the output destination.</li>
37/// <li>Begin the XML output by calling the startDocument method.
38/// <li>Call XML output methods as needed (e.g. startElement).</li>
39/// <li>Complete the XML output by calling the endDocument method to properly
40/// terminate the XML document. No XML output methods can be called following the call to endDocument.</li>
41/// <li>Optionally, reuse the XmlWriter instance by calling the reset method.</li>
42/// </ol>
43/// <p>The following is an example of using MeaXMLWriter to write a simple XML document to the standard output:</p>
44/// <pre>
45/// MeaXMLWriter writer(std::cout);
46///
47/// writer.StartDocument();
48/// writer.StartElement(_T("elem1"));
49/// writer.AddAttribute(_T("id"), _T("1234"));
50/// writer.Characters(_T("Hello World"));
51/// writer.EndElement();
52/// writer.EndDocument();
53/// </pre>
54/// <p>The following XML is displayed on the standard output:</p>
55/// <pre>
56/// &lt;?xml version="1.0" encoding="UTF-8"?&gt;
57/// &lt;elem1 id="1234"&gt;Hello World&lt;/elem1&gt;
58/// </pre>
59///
60class MeaXMLWriter {
