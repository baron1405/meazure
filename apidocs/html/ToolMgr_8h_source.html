44/// Manages the measurement tools including selection,
45/// enabling, disabling, and communicating various messages
46/// to the tools.

49/// that are mutually exclusive (i.e. only one radio tool can
50/// be used at a time).

99 /// Disables all radio tools. Among other things, disabling
100 /// a tool hides it.
101 ///
102 void DisableRadioTools() {
103 if (m_dataDisplay != NULL)
104 m_dataDisplay->ClearRegionData();
105 ToolsMap::const_iterator iter;
106 for (iter = m_tools.begin(); iter != m_tools.end(); ++iter) {
107 MeaTool *tool = (*iter).second;
108 if (tool->IsRadioTool() && tool->IsEnabled()) {
109 tool->Disable();
110 }
111 }
112 }
