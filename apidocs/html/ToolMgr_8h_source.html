45/// to the tools.

100 /// Disables all radio tools. Among other things, disabling
101 /// a tool hides it.
102 ///
103 void DisableRadioTools() {
104 if (m_dataDisplay != nullptr)
105 m_dataDisplay->ClearRegionData();
106 for (const auto& toolEntry : m_tools) {
107 MeaTool* tool = toolEntry.second;
108 if (tool->IsRadioTool() && tool->IsEnabled()) {
109 tool->Disable();
110 }
111 }
112 }
