49/// Mix-in class implementing the Singleton pattern. A Singleton
50/// object is never created or destroyed explicitly. Instead, the
51/// object is created when the Instance() method is called the first
52/// time. The object is destroy when DestroyInstance() is called.
53///
54template <class singleton_t>
55class MeaSingleton_T
56{
57public:
58 /// Destructor for the Singleton. This method is called
59 /// indirectly by the DestroyInstance() method. The delete
60 /// operator is never directly applied to a Singleton object
61 /// by the user of this class.
62 ///
63 virtual ~MeaSingleton_T() { }
64
65
66 /// The first time this method is called, a instance of the
67 /// Singleton object is created and returned. Subsequent calls
68 /// to this method return the same object instance. The method
69 /// asserts if DestroyInstance() has been called.
70 ///
71 /// @return Singleton instance. The same instance is always
72 /// returned.
73 ///
74 static singleton_t& Instance() {
75 MeaAssert(!m_destroyed);
76 if (m_instance == NULL) {
77 m_instance = new singleton_t();
78 }
79 return *m_instance;
80 }
81
82 /// When the Singleton object is no longer needed, it must
83 /// be destroyed using this method. The Instance() method must
84 /// not be called after this method has been called or an
85 /// assertion will be raised.
86 ///
87 static void DestroyInstance() {
88 delete m_instance;
89 m_instance = NULL;
90 m_destroyed = true;
91 }
92
93 /// Indicates whether the Singleton object has been destroyed.
94 /// Once a Singleton instance has been destroyed, it is illegal
95 /// to call the Instance() method.
96 ///
97 /// @return <b>true</b> if the Singleton object has been destroyed.
98 ///
99 static bool IsDestroyed() { return m_destroyed; }
100
101protected:
102 /// Constructs a Singleton object instance. A Singleton is never
103 /// created directly. Instead, the object is created the first
104 /// time the Instance() method is called. Subsequent calls to
105 /// Instance() return the same object instance.
106 ///
107 MeaSingleton_T() { }
108
109private:
110 /// Singletons have no copy semantics. Therefore, this method is
111 /// purposely left undefined to trigger a compile/link error if an
112 /// attempt is ever made to use it.
113 ///
114 MeaSingleton_T(const MeaSingleton_T<singleton_t>& src);
115
116 /// Singletons have no assignment semantics. Therefore, this method
117 /// is purposely left undefined to trigger a compile/link error if
118 /// an attempt is ever made to use it.
119 ///
120 MeaSingleton_T<singleton_t>& operator=(const MeaSingleton_T<singleton_t>& src);
121
122
123 static singleton_t* m_instance; ///< Instance of the Singleton object.
124 static bool m_destroyed; ///< Has the Singleton object been destroyed.
125};
