26/// Mix-in class implementing the Singleton pattern. A Singleton
27/// object is never created or destroyed explicitly. Instead, the
28/// object is created when the Instance() method is called the first
29/// time. The object is destroy when DestroyInstance() is called.
30///
31template <typename singleton_t>
32class MeaSingleton_T {
33
34public:
35 /// The first time this method is called, a instance of the
36 /// Singleton object is created and returned. Subsequent calls
37 /// to this method return the same object instance. The method
38 /// asserts if DestroyInstance() has been called.
39 ///
40 /// @return Singleton instance. The same instance is always
41 /// returned.
42 ///
43 static singleton_t& Instance() {
44 static singleton_t instance { token{} };
45 return instance;
46 }
47
48 /// Singletons have no copy semantics.
49 ///
50 MeaSingleton_T(const MeaSingleton_T<singleton_t>& src) = delete;
51
52 /// Singletons have no assignment semantics.
53 ///
54 MeaSingleton_T<singleton_t>& operator=(const MeaSingleton_T<singleton_t>& src) = delete;
55
56protected:
57 /// Allows the derived class constructor to be public but restricts creation to this class.
58 /// Avoids having to declare this class as a friend in the derived class.
59 ///
60 struct token {};
61
62 /// Constructs a Singleton object instance. A Singleton is never
63 /// created directly. Instead, the object is created the first
64 /// time the Instance() method is called. Subsequent calls to
65 /// Instance() return the same object instance.
66 ///
67 MeaSingleton_T() = default;
68};
