29/// Interface for units related information.
30///
31struct MeaUnitsProvider {
32
33 /// Returns the current linear measurement units.
34 ///
35 /// @return Current linear units.
36 ///
37 virtual MeaLinearUnits* GetLinearUnits() const = 0;
38
39 /// Returns the linear measurement units object corresponding
40 /// to the specified identifier string (e.g. "px").
41 ///
42 /// @param unitsStr [in] Linear units identifier string.
43 ///
44 /// @return Linear measurement units object.
45 ///
46 virtual MeaLinearUnits* GetLinearUnits(const CString& unitsStr) const = 0;
47
48 /// Returns the current angular measurement units.
49 ///
50 /// @return Current angular units.
51 ///
52 virtual MeaAngularUnits* GetAngularUnits() const = 0;
53
54 /// Returns the angular measurement units object corresponding
55 /// to the specified identifier string (e.g. "deg").
56 ///
57 /// @param unitsStr [in] Angular units identifier string.
58 ///
59 /// @return Angular measurement units object.
60 ///
61 virtual MeaAngularUnits* GetAngularUnits(const CString& unitsStr) const = 0;
62
63 /// Returns the orientation of the y-axis.
64 ///
65 /// @return If <b>true</b> is returned, the origin of the coordinate
66 /// system is located at the lower left of the primary screen,
67 /// and the positive y-axis is pointing upward. If <b>false</b>
68 /// is returned, the origin is located at the upper left corner
69 /// and the positive y-axis is pointing down. Note that the
70 /// actual location of the origin is determined by the SetOrigin
71 /// method.
72 ///
73 virtual bool IsInvertY() const = 0;
74
75 /// Returns the location of the origin of the coordinate system.
76 ///
77 /// @return Location of the origin of the coordinate system, in pixels.
78 ///
79 virtual const POINT& GetOrigin() const = 0;
80
81 /// Uses the current linear measurement units to convert the two specified
82 /// points from pixels to the current units, and then calculates the
83 /// width and height of the rectangle formed by the two points.
84 ///
85 /// @param p1 [in] One point defining the rectangle, in pixels.
86 /// @param p2 [in] The other point defining the rectangle, in pixels.
87 ///
88 /// @return The width and height of the rectangle formed by the two
89 /// specified points converted to the current linear measurement
90 /// units. Regardless of the relative locations of the two points,
91 /// the width and height are always positive. In addition, the
92 /// width and height are inclusive (e.g. p1.x = 1, p2.x = 3,
93 /// width = p2.x - p1.x + 1 = 3).
94 ///
95 virtual MeaFSize GetWidthHeight(const POINT& p1, const POINT& p2) const = 0;
96
97 /// Converts the specified angle to the current angular units
98 /// and formats the result for display.
99 /// @param angle [in] Angle to be converted and displayed, in radians.
100 /// @return String containing the converted and formatted angle data.
101 virtual CString FormatConvertAngle(double angle) const = 0;
102
103 /// Formats the specified linear measurement value for display.
104 /// @param id [in] Identifies the linear measurement for
105 /// use in determining the display precision.
106 /// @param value [in] Linear measurement value to format,
107 /// in current units.
108 /// @return String containing the formatted measurement data.
109 virtual CString Format(MeaLinearMeasurementId id, double value) const = 0;
110
111 /// Formats the specified angular measurement value for display.
112 /// @param id [in] Identifies the angular measurement for
113 /// use in determining the display precision.
114 /// @param value [in] Angular measurement value to format,
115 /// in current units.
116 /// @return String containing the formatted measurement data.
117 virtual CString Format(MeaAngularMeasurementId id, double value) const = 0;
118
119 /// Converts the specified coordinate from pixels to the desired units.
120 /// This conversion takes into account the location of the origin and the
121 /// orientation of the y-axis.
122 ///
123 /// @param pos [in] Coordinates in pixels to convert to the desired units.
124 ///
125 /// @return Coordinates converted from pixels to the desired units. The
126 /// conversion takes into account the location of the origin and
127 /// the orientation of the y-axis.
128 ///
129 virtual MeaFPoint ConvertCoord(const POINT& pos) const = 0;
130
131 /// Converts the specified position from pixels to the desired units.
132 /// This conversion does not take into account the location of the origin
133 /// nor does it compensate for the orientation of the y-axis.
134 ///
135 /// @param pos [in] Position in pixels to convert to the desired units.
136 ///
137 /// @return Position converted from pixels to the desired units. The
138 /// conversion does not take into account the location of the
139 /// origin nor does it compensate for the orientation of the y-axis.
140 ///
141 virtual MeaFPoint ConvertPos(const POINT& pos) const = 0;
142
143 /// Converts from the current units to pixels. The conversion does not take into
144 /// account the location of the origin nor the orientation of the y-axis.
145 ///
146 /// @param pos [in] Position to convert to pixels.
147 ///
148 /// @return Position converted from the current units to pixels. The conversion
149 /// does not take into account the location of the origin nor the
150 /// orientation of the y-axis.
151 ///
152 virtual POINT UnconvertPos(const MeaFPoint& pos) const = 0;
153
154 /// Converts the specified resolution in pixels/inch to the desired units.
155 ///
156 /// @param res [in] Resolution in pixels/inch to convert to the desired units.
157 ///
158 /// @return Resolution converted from pixels/inch to the desired units.
159 ///
160 virtual MeaFSize ConvertRes(const MeaFSize& res) const = 0;
161
162 /// Converts the specified angle value from its native radians
163 /// to the desired units.
164 ///
165 /// @param angle [in] Value to be converted.
166 ///
167 /// @return Angle value converted to the desired units.
168 ///
169 virtual double ConvertAngle(double angle) const = 0;
170
171 /// Converts the specified value from the specified units to pixels. A minimum
172 /// pixel value is specified in case the resolution is such that the conversion
173 /// to pixels results in a value that is too small.
174 ///
175 /// @param id [in] Linear units identifier.
176 /// @param res [in] Screen resolution to use to perform the conversion.
177 /// @param value [in] Value to be converted to pixels.
178 /// @param minPixels [in] If the converted pixel value in less than this
179 /// minimum value, the minimum value is returned.
180 ///
181 /// @return X and Y pixel values.
182 ///
183 virtual SIZE ConvertToPixels(MeaLinearUnitsId id, const MeaFSize& res, double value, int minPixels) const = 0;
184
185 /// Returns the number of minor ticks to display before a major tick mark is
186 /// displayed on the measurement rulers.
187 ///
188 /// @return Number of minor tick marks to display before displaying a major
189 /// tick mark.
190 ///
191 virtual int GetMajorTickCount() const = 0;
192
193 /// Returns the increment between minor tick marks on the measurement rulers.
194 ///
195 /// @param rect [in] Rectangle in screen coordinates, used to determine
196 /// the current screen resolution.
197 ///
198 /// @return Increment for the minor tick marks, in the current units.
199 ///
200 virtual MeaFSize GetMinorIncr(const RECT& rect) const = 0;
201};
