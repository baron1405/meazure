65/// Class to represent a validation error. An application class inherits
66/// from the IValidationHandler interface and overrides the
67/// HandleValidationError method. This method is passed a ValidationError
68/// instance whenever the validator detects an invalid construct.
69///
70class ValidationError
71{
72 friend class Validator; ///< Only a Validator is allowed to construct a
73 ///< validation error object.
74
75public:
76 /// Validation error code.
77 ///
78 enum Code {
79 MultipleDocumentElements, ///< The Document Element occurs more
80 ///< than once at the top level.
81 MultipleTopLevelElements, ///< There are multiple top level elements.
82 InvalidDocumentElement, ///< The top level element does not match
83 ///< DOCTYPE declaration.
84 UndeclaredElement, ///< Element not found in DTD.
85 UndeclaredAttribute, ///< Attribute not found in DTD.
86 InvalidElement, ///< Element not allowed by content model.
87 NoCharacterData, ///< Element does not allow PCDATA.
88 UncontainedCharacterData, ///< Character data with no containing element.
89 MismatchedFixedValue, ///< Value of \#FIXED attribute does not match DTD.
90 MissingRequiredAttribute, ///< Required attribute missing from element.
91 BadAttributeValue, ///< Attribute value not in set of possible values.
92 DuplicateId, ///< Duplicate ID values within document.
93 UndeclaredNotation, ///< Notation not found in DTD.
94 UndeclaredEntity, ///< Entity not found in DTD.
95 IdNotFound, ///< Could not find ID corresponding to IDREF.
96 InvalidElementPattern ///< Element pattern violates content model.
97 };

193/// The star of the show, this call performs XML validation for the expat
194/// parser. An application instantiates a Validator object and ties it
195/// into the expat parser via the parser's handler functions.
196///
197/// The validator constructs a DFA for each element's content model using
198/// the venerable algorithm 3.5 from the Dragon book. Optimizations bypass
199/// the algorithm and directly construct the DFA for simple content models
200/// (e.g. singletons, pure choice, pure sequence).
201///
202/// The validator is a C++ object with methods that you call from expat
203/// handlers. The method parameters are of the same types as those passed
204/// to the expat handlers.
205///
206/// To use the validator instantiate an ev::Validator object then call the
207/// following methods from the corresponding expat handlers:
208///
209/// <table align="center" cellspacing="0" cellpadding="3">
210/// <tr><th>Validator Method</th> <th>Call From expat Handler</th></tr>
211/// <tr><td>AddElementDecl</td> <td>XML_ElementDeclHandler</td></tr>
212/// <tr><td>AddAttributeDecl</td> <td>XML_AttlistDeclHandler</td></tr>
213/// <tr><td>AddNotationDecl</td> <td>XML_NotationDeclHandler</td></tr>
214/// <tr><td>AddEntityDecl</td> <td>XML_EntityDeclHandler</td></tr>
215/// <tr><td>SetDocumentElement</td> <td>XML_StartDoctypeDeclHandler</td></tr>
216/// <tr><td>StartElement</td> <td>XML_StartElementHandler</td></tr>
217/// <tr><td>EndElement</td> <td>XML_EndElementHandler</td></tr>
218/// <tr><td>CharacterData</td> <td>XML_CharacterDataHandler</td></tr>
219/// </table>
220///
221/// Encoding uses expat's XML_Char encoding based on whether XML_UNICODE is
222/// defined. exval defines the types ev::EVString and ev::EVOStream but these
223/// are just convenience wrappers around the appropriate STL string and stream
224/// classes.
225///

395 /// Indicates whether the specified element allows PCDATA.
