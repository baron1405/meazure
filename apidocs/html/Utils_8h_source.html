36/// Represents a rectangular size. Unlike the Windows SIZE
37/// structure, the dimensions of the FSIZE structure are
38/// double precision values.
39///
40struct FSIZE {
41 double cx; ///< Length in the x dimension.
42 double cy; ///< Length in the y dimension.

45 /// Adds this and the specified object and returns
46 /// a new size object.
47 /// \f[
48 /// \left[ \begin{array}{c}
49 /// f_x \\
50 /// f_y \\
51 /// \end{array} \right]
52 /// =
53 /// \left[ \begin{array}{c}
54 /// this_x \\
55 /// this_y \\
56 /// \end{array} \right]
57 /// +
58 /// \left[ \begin{array}{c}
59 /// fsize_x \\
60 /// fsize_y \\
61 /// \end{array} \right]
62 /// \f]
63 /// @param fsize [in] Size object to add.
64 /// @return New object that is the sum of this and the
65 /// specified object.
66 FSIZE operator+(const FSIZE& fsize) const {
67 FSIZE f;
68 f.cx = cx + fsize.cx;
69 f.cy = cy + fsize.cy;
70 return f;
71 }

80 /// =

256/// Calculates the log base 10 on each dimension of the specified
257/// size object and returns the results in a new size object.
258/// \f[
259/// \left[ \begin{array}{c}
260/// f_x \\
261/// f_y \\
262/// \end{array} \right]
263/// =
264/// \left[ \begin{array}{c}
265/// \log fsize_x \\
266/// \log fsize_y \\
267/// \end{array} \right]
268/// \f]
269/// @param fsize [in] Size object whose logarithm is to be obtained.
270///
271/// @return A new size object whose dimensions are the base 10 log of
272/// the specified size object's dimensions.
273///
274inline FSIZE log10(const FSIZE& fsize) {
275 FSIZE f;
276 f.cx = log10(fsize.cx);
277 f.cy = log10(fsize.cy);
278 return f;
279}
