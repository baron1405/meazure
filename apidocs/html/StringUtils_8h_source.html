37 /// Tests whether the specified string is a number. For the
38 /// purposes of this method, a number is a base 10 double
39 /// precision floating point value.
40 ///
41 /// @param str [in] String to test
42 /// @param valuep [out] Value parsed from the string. nullptr
43 /// if not interested. If the return value is
44 /// <b>false</b>, valuep's value is undefined.
45 ///
46 /// @return <b>true</b> if string represents a number.
47 ///
48 bool IsNumber(LPCTSTR str, double* valuep = nullptr);
49
50 /// Tests whether the specified string is a boolean value. For
51 /// the purpose of this method, the strings "1", "TRUE", "true"
52 /// are boolean <b>true</b> values, while "0", "FALSE", "false"
53 /// are boolean <b>false</b>.
54 ///
55 /// @param str [in] String to test
56 /// @param valuep [out] Value parsed from the string. nullptr
57 /// if not interested. If the return value is
58 /// <b>false</b>, valuep's value is undefined.
59 ///
60 /// @return <b>true</b> if string represents a boolean value.
61 ///
62 bool IsBoolean(LPCTSTR str, bool* valuep = nullptr);
63
