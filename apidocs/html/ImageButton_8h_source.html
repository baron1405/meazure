26/// A button that uses bitmap resources. Based on code from
27/// Joseph M. Newcomer at:
28///
29/// http://www.codeproject.com/buttonctrl/betterbmpbutton.asp
30///
31class MeaImageButton : public CButton
32{
33public:
34 /// Constructs a image button using the specified image(s).
35 ///
36 /// @param up [in] Resource ID of image for button in up position.
37 /// @param down [in] Resource ID of image for button in down position.
38 /// @param disabled [in] Resource ID of image to use when button is disabled.
39 ///
40 explicit MeaImageButton(UINT up, UINT down = 0, UINT disabled = 0);
41
42 /// Constructs an image button with no images specified; the LoadBitmaps
43 /// method must be called.
44 ///
45 MeaImageButton() : CButton(), m_theme(NULL), m_up(0), m_down(0),
46 m_disabled(0), m_depressed(false), m_toggle(false) { }
47
48 /// Destroys a image button object.
49 ///
50 virtual ~MeaImageButton();
51
52
53 /// Convenience version of the button create method. Using
54 /// this method allows the button to size itself to the bitmap.
55 ///
56 /// @param dwStyle [in] Button style as defined by CButton.
57 /// @param pParentWnd [in] Parent window for the button.
58 /// @param nID [in] ID for the button.
59 /// @param toolTipID [in] String resource ID for the button's tooltip.
60 ///
61 /// @return <b>true</b> if creation successful.
62 ///
63 bool Create(DWORD dwStyle, CWnd* pParentWnd, UINT nID = 0xffff, UINT toolTipID = 0xffff);
64
65 /// Relays messages to the tooltip control.
66 ///
67 /// @param pMsg [in] Message information structure.
68 ///
69 /// @return <b>TRUE</b> if message was processed.
70 ///
71 virtual BOOL PreTranslateMessage(MSG* pMsg);
72
73 /// Specifies whether the button is a push button or a toggle button.
74 ///
75 /// @param toggle [in] <b>true</b> if the button should have toggle behavior.
76 ///
77 void SetToggleType(bool toggle) { m_toggle = toggle; }
78
79
80 /// Loads the button's bitamps.
81 ///
82 /// @param up [in] Resource ID of image to use for up appearance when enabled.
83 /// @param down [in] Resource ID of image to use for down appearance when enabled.
84 /// If 0, the up image is used.
85 /// @param disabled [in] Resource ID of image to use for disabled appearance.
86 /// If 0, the up image is halftoned gray
87 ///
88 void LoadBitmaps(UINT up, UINT down, UINT disabled = 0)
89 { m_up = up; m_down = down; m_disabled = disabled; }
90
91 /// Returns the bitmaps associated with the button image.
92 ///
93 /// @param up [out] Resource ID of image to use for up appearance when enabled.
94 /// @param down [out] Resource ID of image to use for down appearance when enabled.
95 /// @param disabled [out] Resource ID of image to use for disabled appearance.
96 ///
97 void GetBitmaps(UINT &up, UINT &down, UINT &disabled) const
98 { up = m_up; down = m_down; disabled = m_disabled; }
99
100
101 /// Sets the button state: up or down.
102 ///
103 /// @param state [in] Button state (0 = up, 1 = down).
104 ///
105 void SetState(bool state) {
106 m_depressed = state;
107 CButton::SetState(state);
108 InvalidateRect(NULL);
109 }
110
111protected:
112 HTHEME m_theme; ///< Windows rendering theme for buttons
113 UINT m_up; ///< ID of bitmap for button up appearance.
114 UINT m_down; ///< ID of bitmap for button down appearance.
115 UINT m_disabled; ///< ID of bitmap for button disabled appearance.
116 bool m_depressed; ///< Indicates if the button is depressed.
117 bool m_toggle; ///< Indicates if this is a toggle button.
118 CToolTipCtrl m_toolTip; ///< Tool tip control for the button.

138};
