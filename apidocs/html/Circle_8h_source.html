31/// A circle element. The circle is positioned by specifying the center
32/// and it is sized by specifying a point on the perimeter. The circle
33/// is formed by using a series of circularly arranged polygonal regions
34/// to create a thin circular window.
35///
36class MeaCircle : public MeaGraphic {

38public:

70 /// Sets the position and size of the circle.

75 void SetPosition(const POINT& center, const POINT& perimeter);

77 /// Obtains the position and size of the circle.

79 /// @param center [out] Location of the center of the circle, in pixels

87 /// Calculates the radius of the circle based on the center and perimeter
88 /// points. The following classic equation is used:
89 /// \f[
90 /// radius=\sqrt{(center_x-perimeter_x)^2+(center_y-perimeter_y)^2}
91 /// \f]
92 /// @return The length of the circle's radius, in pixels.
93 ///
94 int GetRadius() const {
95 return static_cast<int>(MeaGeometry::CalcLength(m_center, m_perimeter));
96 }

98 /// Sets the color of the circle's line.

104protected:

107 /// Windows GDI was not optimized for doing crazy things like creating
108 /// a window that is a thin lined circle. Because of this, drawing the
109 /// window is very expensive. To minimize visual artifacts when the circle
110 /// is repositioned or resized a timer is used to effectively add damping
111 /// to the drawing process. Even if the user may be making quick changes
112 /// to the circle, the time delay reduces the number of redraws. When the
113 /// SetPosition() method is called, it just sets a timer and returns. When
114 /// the timer expires, this method is called and actually redraws the
115 /// window.
116 ///
117 /// @param wParam [in] not used
118 /// @param lParam [in] not used
119 ///
120 /// @return The return is not used so 0 is returned.
121 ///
122 afx_msg LRESULT OnHPTimer(WPARAM wParam, LPARAM lParam);
123

137 /// The circular window region is composed of single pixel rectangles
138 /// arranged in a circle. The location of each rectangle is determined
139 /// using an algorithm from the paper "A Fast Bresenham Type Algorithm
140 /// for Drawing Circles" by John Kennedy, Mathematics Dept., Santa
141 /// Monica College (http://homepage.smc.edu/kennedy_john/BCIRCLE.PDF,
142 /// rkennedy@ix.netcom.com).
143 ///
144 /// @param radius [in] radius of the circle, in pixels
145 ///
146 void PlotCircle(int radius);
147
148 /// The circular window is composed of a series of single pixel
149 /// rectangles arranged in a circle. At each point on the circle
150 /// the coordinates of a rectangle surrounding that point are added
151 /// to an array of points. This array of points will eventually be
152 /// handed to the GDI function CreatePolyPolygonRgn.
153 ///
154 /// @param x [in] X coordinate of the point on the circle
155 /// @param y [in] Y coordinate of the point on the circle
156 ///
157 void AddPoint(int x, int y) {
158 CPoint p(x, y);
159
160 // Make sure the point is somewhere on the virtual rectangle
161 // formed by all display monitors.
162 //
163 if (m_clipRect.PtInRect(p)) {
164 // Determine where in the coordinate array
165 // we will start adding the rectangle coordinates.
166 //
167 POINT* arr = &m_arr[m_count << 2];
168
169 // (x,y) (x+1, y)
170 // ---------
171 // | |
172 // | |
173 // | |
174 // ---------
175 // (x,y+1) (x+1,y+1)
176
177
178 // (x,y)
179 //
180 *arr++ = p;
181
182 // (x+1,y)
183 //
184 p.x++;
185 *arr++ = p;
186
187 // (x+1,y+1)
188 //
189 p.y++;
190 *arr++ = p;
191
192 // (x, y+1)
193 //
194 p.x--;
195 *arr = p;
196
197 // Keep track of the number of rectangles
198 //
199 m_count++;
200 }
201 }

205 CPoint m_perimeter; ///< Location of a point on the perimeter of the circle, in pixels
206 CRect m_clipRect; ///< Virtual rectangle formed by all display monitors, in pixels
207 CBrush* m_foreBrush; ///< Brush for drawing the circle
208 MeaTimer m_timer; ///< Timer for delayed drawing of circle to reduce visual artifacts
209 POINT* m_arr; ///< Array of coordinates for each rectangle making up the circular region
210 unsigned int m_count; ///< Number of rectangles in the circular region
211 static int* m_varr; ///< Number of coordinates comprising each polygon making up the circular
212 ///< region (always 4 in our case)
213};
