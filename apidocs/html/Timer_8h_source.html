31/// Implements a timer that issues timing messages at a higher priority
32/// than the standard windows timer that issues WM_TIMER messages.
33///
34class MeaTimer {
35
36public:
37 /// Instantiates a timer. To use the timer, the Create method
38 /// must be called.
39 ///
40 MeaTimer();
41
42 /// Destroys a timer.
43 ///
44 virtual ~MeaTimer();
45
46 /// Registers the specified parent window to receive the timer
47 /// messages. The timer will send the parent a MeaHPTimerMsg
48 /// message.
49 ///
50 /// @param parent [in] Parent window to receive the MeaHPTimerMsg message.
51 /// @param userData [in] Caller defined data.
52 ///
53 void Create(CWnd* parent, WPARAM userData = 0) {
54 MeaAssert(parent != nullptr);
55 m_parent = parent;
56 m_userData = userData;
57 }
58
59 /// Sets the timer to the specified interval and starts it running.
60 ///
61 /// @param elapse [in] Time interval in milliseconds.
62 ///
63 void Start(int elapse);
64
65 /// Cancels the current timer if it is set.
66 ///
67 void Stop();
68
69private:
70 /// Called when the timer expires. Simply calls the instance
71 /// method of the same name.
72 ///
73 /// @param pParam [i] Timer instance pointer.
74 ///
75 /// @return Zero indicating success.
76 ///
77 static UINT TimerProc(LPVOID pParam) {
78 return static_cast<MeaTimer*>(pParam)->TimerProc();
79 }
80
81 /// Called when the timer expires. Sends the MeaHPTimerMsg message
82 /// to the parent window.
83 ///
84 /// @return Zero indicating success.
85 ///
86 UINT TimerProc();
87
88
89 CWinThread* m_timerThread; ///< Thread in which the timer runs.
90 CCriticalSection m_critSect; ///< Critical section guard.
91 CEvent m_stopEvent; ///< Used internally to indicate the timer has been stopped.
92 CEvent m_exitEvent; ///< Used internally to indicate the timer has completed all processing.
93 int m_elapse; ///< Timer period, in milliseconds.
94 CWnd* m_parent; ///< Window to receive the timer expire message.
95 WPARAM m_userData; ///< Caller defined data.
96};
