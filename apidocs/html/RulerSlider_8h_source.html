26/// Slider control on a ruler background. Dynamic readout of pixel position is
27/// provided while the slider is dragged.
28///
29class MeaRulerSlider : public CProgressCtrl {
30 DECLARE_DYNAMIC(MeaRulerSlider)
31
32public:
33 /// Constructs a ruler slider control. Before the control can be used, the
34 /// Create method must be called.
35 ///
36 MeaRulerSlider();
37
38 /// Destroys a ruler slider control.
39 ///
40 virtual ~MeaRulerSlider();
41
42 /// Returns the position value of the slider.
43 ///
44 /// @return The slider's current position.
45 ///
46 int GetSliderPos();
47
48 /// Set the position of the slider.
49 ///
50 /// @param pos [in] Position for the slider.
51 ///
52 void SetSliderPos(int pos);
53
54public:
55 /// Creates the window for the ruler slider control.
56 ///
57 /// @param dwStyle [in] Style for the control's window as defined by CProgressCtrl.
58 /// @param rect [in] Bounding retangle for the control, in pixels.
59 /// @param pParentWnd [in] Parent window for the control.
60 /// @param nID [in] ID for the control.
61 ///
62 /// @return TRUE if the control was created successfully.
63 ///
64 virtual BOOL Create(DWORD dwStyle, const RECT& rect, CWnd* pParentWnd, UINT nID) override;
65
66 /// Called right before the base class control is subclassed. This
67 /// method is used to perform intial setup of the slider.
68 ///
69 virtual void PreSubclassWindow() override;
70
71protected:
72 /// Initializes the slider.
73 ///
74 /// @param rect [in] Client rectangle for the slider.
75 ///
76 void Init(const CRect& rect);
77
78 /// Draws the calipers that form the thumb control for the slider.
79 ///
80 /// @param pDC [in] Device context into which the calipers are drawn.
81 ///
82 void DrawCalipers(CDC* pDC);
83
84 /// Called when the left mouse button is pressed. Captures the pointer
85 /// for subsequent dragging of the slider.
86 /// @param nFlags [in] Modifier flags in effect.
87 /// @param point [in] Location of the pointer.
88 ///
89 afx_msg void OnLButtonDown(UINT nFlags, CPoint point);
90
91 /// Called when the left mouse button is released. Releases the capture
92 /// of the pointer.
93 /// @param nFlags [in] Modifier flags in effect.
94 /// @param point [in] Location of the pointer.
95 ///
96 afx_msg void OnLButtonUp(UINT nFlags, CPoint point);
97
98 /// Called when the mouse is moved over the slider. If the pointer is
99 /// captured, the slider thumb control is dragged.
100 /// @param nFlags [in] Modifier flags in effect.
101 /// @param point [in] Location of the pointer.
102 ///
103 afx_msg void OnMouseMove(UINT nFlags, CPoint point);
104
105 /// Called to repaint the slider control.
106 ///
107 afx_msg void OnPaint();
108
109 /// Called to clear the control's background.
110 /// @param pDC [in] Device context into which to draw the background.
111 /// @return Always returns TRUE.
112 ///
113 afx_msg BOOL OnEraseBkgnd(CDC* pDC);
114
115 /// Called when the size of the control's window is changed. The control
116 /// resets its position accordingly.
117 /// @param nType [in] Indicates the type of resizing.
118 /// @param cx [in] New width, in pixels.
119 /// @param cy [in] New height, in pixels.
120 ///
121 afx_msg void OnSize(UINT nType, int cx, int cy);
122
123 /// Called when the control is enabled or disabled so that it can redraw
124 /// itself to reflect the new state.
125 /// @param bEnable [in] Indicates if the control has been enabled or disabled.
126 ///
127 afx_msg void OnEnable(BOOL bEnable);
128
129 DECLARE_MESSAGE_MAP()
130
131private:
132 static constexpr int kHotSpotOffset { 4 }; ///< Offset around the caliper that is draggable.
133 static constexpr int kTickPosition { 10 }; ///< Relative position for the ruler tick marks.
134 static constexpr int kTickFactor { 4 }; ///< The tick height is the thickness of the window divided by this factor.
135
136
137 /// Indicates whether the specified slider position is in the draggable area.
138 ///
139 /// @return <b>true</b> if the position is in the draggable area for the slider.
140 ///
141 bool IsHot(int pos) const;
142
143 bool m_vertical; ///< Indicates if the slider is oriented vertically.
144 bool m_captured; ///< Indicates if the slider thumb control captured.
145 bool m_sliderActive; ///< Indicates the slider can be dragged.
146 CFont m_posFont; ///< Font used for displaying the current slider position.
147 CPen m_activePen; ///< Pen to draw the control when enabled.
148 CPen m_inactivePen; ///< Pen to draw the control when disabled.
149 CBrush m_activeBrush; ///< Brush to paint the control when enabled.
150 CBrush m_inactiveBrush; ///< Brush to paint the control when disabled.
151 CRect m_sliderRect; ///< Window rectangle for the control.
152 int m_sliderPos; ///< Current slider position.
153};
