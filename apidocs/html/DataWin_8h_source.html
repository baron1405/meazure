31/// A popup data display window. A data display window is attached to each
32/// radio tool's crosshair and a corner of the Window Tool. When the user
33/// hovers over a crosshair and while the user drags the crosshair, the
34/// data display window is shown and provides a continues readout of the
35/// measurements appropriate to the tool. The user can set a preference
36/// to not display the data windows.
37///
38class MeaDataWin : public CWnd {
39
40public:
41 static constexpr bool kDefArmed { true }; ///< Indicates whether to display data windows by default.
42 static constexpr int kNoLabelId { 0xffff }; ///< Indicates a data window label has not been specified.
43
44
45 /// Constructs a data window. To use the data window, its Create method
46 /// must be called and the window must be displayed using the Show method.
47 ///
48 /// @param screenProvider [in] Screen information provider
49 /// @param unitsProvider [in] Units information provider
50 ///
51 MeaDataWin(const MeaScreenProvider& screenProvider, const MeaUnitsProvider& unitsProvider);
52
53 /// Destroys a data window.
54 ///
55 virtual ~MeaDataWin();
56
57 /// Creates the window for the data display. This method must be called
58 /// prior to calling the Show method. A data item is displayed in the
59 /// window if its label resource ID is a value other than kNoLabelId.
60 ///
61 /// @param opacity [in] Opacity for the window, where 0 is completely
62 /// transparent and 255 is completely opaque.
63 /// @param xLabelId [in] String resource ID for the x coordinate label.
64 /// @param yLabelId [in] String resource ID for the y coordinate label.
65 /// @param wLabelId [in] String resource ID for the width label.
66 /// @param hLabelId [in] String resource ID for the height coordinate label.
67 /// @param dLabelId [in] String resource ID for the distance label.
68 /// @param aLabelId [in] String resource ID for the angle label.
69 /// @param parent [in] Parent window for the data window if it is
70 /// used as a child window. Typically, this parameter
71 /// is nullptr, which creates a floating window.
72 ///
73 /// @return <b>true</b> if the window is successfully created.
74 ///
75 bool Create(BYTE opacity, UINT xLabelId = kNoLabelId, UINT yLabelId = kNoLabelId,
76 UINT wLabelId = kNoLabelId, UINT hLabelId = kNoLabelId,
77 UINT dLabelId = kNoLabelId, UINT aLabelId = kNoLabelId,
78 const CWnd* parent = nullptr);
79
80 /// Displays the specified x and y coordinates. For the values
81 /// to be displayed, a string resource ID value must be specified
82 /// for the xLabelId and yLabelId parameters of the Create method.
83 ///
84 /// @param point [in] Coordinates to display.
85 ///
86 void ShowXY(const MeaFPoint& point) {
87 m_xData = m_unitsProvider.Format(MeaX, point.x);
88 m_yData = m_unitsProvider.Format(MeaY, point.y);
89 }
90
91 /// Displays the specified width and height. For the values
92 /// to be displayed, a string resource ID value must be specified
93 /// for the wLabelId and hLabelId parameters of the Create method.
94 ///
95 /// @param size [in] Dimensions to display.
96 ///
97 void ShowWH(const MeaFSize& size) {
98 m_wData = m_unitsProvider.Format(MeaW, size.cx);
99 m_hData = m_unitsProvider.Format(MeaH, size.cy);
100 }
101
102 /// Displays a diagonal distance based on the specified width and height.
103 /// For the value to be displayed, a string resource ID value must be
104 /// specified for the dLabelId parameter of the Create method.
105 ///
106 /// @param dist [in] Dimensions used to calculate the distance.
107 ///
108 void ShowDistance(const MeaFSize& dist) {
109 m_dData = m_unitsProvider.Format(MeaD, MeaGeometry::CalcLength(dist.cx, dist.cy));
110 }
111
112 /// Displays the specified distance. For the value to be displayed, a
113 /// string resource ID value must be specified for the dLabelId
114 /// parameter of the Create method.
115 ///
116 /// @param dist [in] Distance to display.
117 ///
118 void ShowDistance(double dist) {
119 m_dData = m_unitsProvider.Format(MeaD, dist);
120 }
121
122 /// Displays the specified angle. For the value to be displayed, a
123 /// string resource ID value must be specified for the aLabelId
124 /// parameter of the Create method.
125 ///
126 /// @param angle [in] Angle to display.
127 ///
128 void ShowAngle(double angle) {
129 m_aData = m_unitsProvider.FormatConvertAngle(angle);
130 }
131
132 /// Displays the currently set data in the window.
133 ///
134 /// @param centerWin [in] The data window is displayed
135 /// adjacent to a corner of this window.
136 ///
137 void Update(const CWnd& centerWin) {
138 CRect rect;
139 centerWin.GetWindowRect(rect);
140 Update(rect);
141 }
142
143 /// Displays the currently set data in the window.
144 ///
145 /// @param rect [in] The data window is displayed
146 /// adjacent to the corner of this rectangle.
147 ///
148 void Update(const CRect& rect);
149
150 /// Shows the data window, if the window is armed. If the
151 /// window is not armed, calling Show does not display the
152 /// window.
153 ///
154 void Show();
155
156 /// Hides the data window.
157 ///
158 void Hide() { if (m_hWnd != nullptr) ShowWindow(SW_HIDE); }
159
160 /// Flashes the text in the data window once. Typically used
161 /// to indicate that a tool position has been recorded.
162 ///
163 void Strobe();
164
165 /// Sets the opacity of the data window.
166 ///
167 /// @param opacity [in] Opacity of the data window
168 /// where 0 is transparent and 255 is
169 /// opaque.
170 ///
171 void SetOpacity(BYTE opacity);
172
173 /// Arms the data window so that it can be displayed when the
174 /// Show method is called.
175 ///
176 /// @param armed [in] Specifies whether the data window
177 /// is allowed to be displayed.
178 ///
179 static void Arm(bool armed) { m_armed = armed; }
180
181 /// Indicates whether the data window can be displayed.
182 ///
183 /// @return <b>true</b> if the data window is allowed to be displayed
184 /// using the Show method.
185 ///
186 static bool IsArmed() { return m_armed; }
187
188
189 /// Persists the display settings of the data window to the
190 /// specified profile object.
191 ///
192 /// @param profile [in] Profile object into which the
193 /// data window state is persisted.
194 ///
195 static void SaveProfile(MeaProfile& profile);
196
197 /// Restores the display settings of the data window from the
198 /// specified profile object.
199 ///
200 /// @param profile [in] Profile object from which the
201 /// data window state is restored.
202 ///
203 static void LoadProfile(MeaProfile& profile);
204
205 /// Called when the master reset is activated. Causes the
206 /// data window to load its default settings.
207 ///
208 static void MasterReset();
209
210protected:
211 DECLARE_MESSAGE_MAP()
212
213 /// Performs cleanup of objects created in the Create method.
214 ///
215 afx_msg void OnDestroy();
216
217 /// Draws the data window.
218 ///
219 afx_msg void OnPaint();
220
221 /// Called in response to the flash timer so that the data
222 /// window can be strobed.
223 ///
224 /// @param timerId [in] ID of the timer.
225 ///
226 afx_msg void OnTimer(UINT_PTR timerId);
227
228private:
229 /// Specifies how the data window is drawn in support
230 /// of strobing the data window.
231 ///
232 enum DrawState {
233 Normal, ///< Draw the data window in its normal appearance.
234 Inverted ///< Draw the data window with its background and text colors swapped.
235 };
236
237
238 static bool m_armed; ///< Indicates if the data window is allowed to be displayed.
239 static CSize m_margin; ///< Margin between the data text and the window border, in pixels.
240 static UINT m_flashInterval; ///< Window flash interval, in milliseconds.
241
242
243 /// Calculates the width for the data window based on the width

246 ///
247 /// @return Width of the data window, in pixels.
248 ///
249 int CalcWidth();
250

252 ///
253 /// @param dc [in] Device context into which the data window
254 /// is drawn.
255 ///
256 void DrawWin(CDC& dc);
257

260 const MeaUnitsProvider& m_unitsProvider; ///< Units information provider
261 CString m_xLabel; ///< Label for the x coordinate data.
262 CString m_yLabel; ///< Label for the y coordinate data.
263 CString m_wLabel; ///< Label for the width data.
264 CString m_hLabel; ///< Label for the height data.
265 CString m_dLabel; ///< Label for the distance data.

271 CString m_dData; ///< Distance data converted to a string.
272 CString m_aData; ///< Angle data converted to a string.
273 const CWnd* m_parent; ///< Parent window or nullptr if data window is floating.
274 CFont m_font; ///< Font for the data display.
275 LONG m_textHeight; ///< Height of the text, in pixels.
276 LONG m_winHeight; ///< Height of the data window, in pixels.

278 DrawState m_drawState; ///< Indicates how the data window should be drawn.
279 int m_flashCount; ///< Window flash interval, in milliseconds.
280 BYTE m_opacity; ///< Data window opacity, 0 (transparent) to 255 (opaque).
