29/// Abstract base class for tools whose use is mutually exclusive. For
30/// example, a Line tool cannot be used at the same time a Point tool is
31/// being used. Thus, the Line and Point tools are mutually exclusive and
32/// derive from MeaRadioTool.
33///
34class MeaRadioTool: public MeaTool
35{
36public:
37 /// Map of points used to set the position of a tool.
38 /// Typically the position log manager will fill a point
39 /// map and pass it to the tool's SetPosition method.
40 ///
41 typedef std::map<CString, POINT> PointMap;
42
43
44 /// Constructs a new instance of an radio based measurement tool.
45 /// Typically, to use a newly constructed tool, the Enable() method
46 /// must be called.
47 ///
48 /// @param mgr [in] Tool manager parent of the tool.
49 ///
50 explicit MeaRadioTool(MeaToolMgr *mgr);
51
52 /// Destroys the tool.
53 ///
54 virtual ~MeaRadioTool();
55
56
57 /// Indicates that the derived tool is a Radio tool and
58 /// cannot be enabled until any other enabled Radio tool
59 /// is disabled.
60 ///
61 /// @return Always <b>true</b> indicating that this class
62 /// and its derived classes are Radio tools.
63 ///
64 virtual bool IsRadioTool();
65
66
67 /// Returns a resource ID for the string used to label this
68 /// tool on the data display.
69 ///
70 /// @return Resource ID for the label. This base class implementation
71 /// returns the resource ID for an empty string.
72 ///
73 virtual UINT GetLabelId() const;
74
75 /// Indicates whether the derived tool uses crosshairs.
76 ///
77 /// @return This base class implementation always returns <b>true</b>.
78 ///
79 virtual bool HasCrosshairs() const;
80
81
82 /// Displays the tool's crosshairs. If the Enable() method has not
83 /// been previously called, this method has no effect. This base
84 /// class implementation does nothing.
85 ///
86 virtual void EnableCrosshairs();
87
88 /// Hides the tool's crosshairs. This base class implementation
89 /// does nothing.
90 ///
91 virtual void DisableCrosshairs();
92
93
94 /// Visually flashes the tool. Typically, if a tool uses crosshairs,
95 /// they are cycled between a hilite color and their normal color a
96 /// number of times. Flashing the tool helps the user find it on the
97 /// screen. This base class implementation does nothing.
98 ///
99 virtual void Flash();
100
101 /// Visually strobes the tool. Typically, if a tool uses crosshairs,
102 /// they are cycled between a hilite color and their normal color once.
103 /// In addition, if the tool has data windows attached to the crosshairs,
104 /// they are cycled between hidden and visible. Strobing the tool is done
105 /// to visually confirm region capture and position recording. This base
106 /// class implementation does nothing.
107 ///
108 virtual void Strobe();
109
110
111 /// Indicates whether the tool can define a rectangular region that
112 /// can then be read for a screen capture.
113 ///
114 /// @return <b>true</b> if the tool defines a non-empty rectangular
115 /// region. This base class implementation always returns
116 /// <b>false</b>.
117 ///
118 virtual bool HasRegion();
119
120 /// Returns the rectangle defined by the tool. This rectangle can then
121 /// be read for a screen capture.
122 ///
123 /// @return The region rectangle. This base class implementation returns
124 /// an empty rectangle.
125 ///
126 virtual RECT GetRegion();
127
128
129 /// Sets the position of the tool. This method is called by the
130 /// position log manager to play back a tool's position that was
131 /// previously recorded using the GetPosition method. This base
132 /// class implementation does nothing.
133 ///
134 /// @param points [in] Map of positions for the tool's crosshairs.
135 ///
136 virtual void SetPosition(const PointMap& points);
137
138 /// Records the position of the tool. This method is called by the
139 /// position log manager to record a position.
140 ///
141 /// @param position [in] The tool's current position is recorded into
142 /// the position log manager's position object.
143 ///
144 virtual void GetPosition(MeaPositionLogMgr::Position& position) const = 0;
145
146
147 /// Called by the OS when the mouse pointer is moved. This base class
148 /// implementation does nothing.
149 ///
150 /// @param wParam [in] OS Message ID
151 /// @param lParam [in] MOUSEHOOKSTRUCT
152 ///
153 virtual void OnMouseHook(WPARAM wParam, LPARAM lParam);
154};
