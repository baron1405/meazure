32/// Represents a single monitor attached to the system. There is an instance of this class per monitor.
33///
34class MeaPositionScreen {
35
36public:
37 /// Constructs a screen object.
38 ///
39 MeaPositionScreen() : m_primary(false), m_manualRes(false) {}
40
41 /// Constructs deep copy screen object from the specified
42 /// iterator over the list of screens.
43 ///
44 /// @param screenIter [in] Iterator from which to construct a copy of the screen.
45 /// @param unitsProvider [in] Units information and conversion provider
46 /// @param screenProvider [in] Screen information provider
47 ///
48 explicit MeaPositionScreen(const MeaScreenProvider::ScreenIter& screenIter, const MeaUnitsProvider& unitsProvider,
49 const MeaScreenProvider& screenProvider);
50
51 /// Constructs a deep copy of the specified screen.
52 ///
53 /// @param screen [in] Screen object to be copied.
54 ///
55 MeaPositionScreen(const MeaPositionScreen& screen) { Copy(screen); }
56
57 /// Loads screen elements of the log file.
58 ///
59 /// @param screenNode [in] screen node in the DOM.
60 ///
61 void Load(const MeaXMLNode* screenNode);
62
63 /// Saves the screen information
64 ///
65 /// @param writer [in] Provides ability to write a position to the log.
66 /// @param indent [in] Output indentation level.
67 /// @throw CFileException if there was a problem saving the file
68 ///
69 void Save(MeaPositionLogWriter& writer, int indent) const;
70
71 /// Assignment operator for a screen object. Makes
72 /// a deep copy of the object.
73 ///
74 /// @param screen [in] Screen option to assign to this.
75 ///
76 /// @return This screen object.
77 ///
78 MeaPositionScreen& operator=(const MeaPositionScreen& screen) {
79 return Copy(screen);
80 }
81
82 /// Tests for equality between the specified screen object and this.
83 ///
84 /// @param screen [in] Screen object to test against this.
85 ///
86 /// @return <b>true</b> if the specified screen object and this are equal.
87 ///
88 bool operator==(const MeaPositionScreen& screen) const {
89 return ((m_primary == screen.m_primary) &&
90 MeaNumericUtils::IsEqualF(m_rect.top, screen.m_rect.top) &&
91 MeaNumericUtils::IsEqualF(m_rect.bottom, screen.m_rect.bottom) &&
92 MeaNumericUtils::IsEqualF(m_rect.left, screen.m_rect.left) &&
93 MeaNumericUtils::IsEqualF(m_rect.right, screen.m_rect.right) &&
94 MeaNumericUtils::IsEqualF(m_res.cx, screen.m_res.cx) &&
95 MeaNumericUtils::IsEqualF(m_res.cy, screen.m_res.cy) &&
96 (m_manualRes == screen.m_manualRes) &&
97 (m_desc == screen.m_desc));
98 }
99
100 /// Tests for inequality between the specified screen object and this.
101 ///
102 /// @param screen [in] Screen object to test against this.
103 ///
104 /// @return <b>true</b> if the specified screen object and this are not equal.
105 ///
106 bool operator!=(const MeaPositionScreen& screen) const { return !(*this == screen); }
107
108 /// Indicates whether this screen is the primary display.
109 ///
110 /// @return <b>true</b> if this screen is the primary display.
111 ///
112 bool IsPrimary() const { return m_primary; }
113
114 /// Provides the screen rectangle expressed in the units that were in effect at the time the
115 /// screen object was created.
116 ///
117 /// @return Screen rectangle
118 ///
119 const MeaFRect& GetRect() const { return m_rect; }
120
121 /// Provides the screen resolution expressed in the units that were in effect at the time the
122 /// screen object was created.
123 ///
124 /// @return Screen resolution.
125 ///
126 const MeaFSize& GetRes() const { return m_res; }
127
128 /// Indicates whether the resolution calibrated manually.
129 ///
130 /// @return <b>true</b> if the screen resolution has been manually calibrated.
131 ///
132 bool IsManualRes() const { return m_manualRes; }
133
134 /// Description of the screen.
135 ///
136 /// @return Screen description.
137 const CString& GetDesc() const { return m_desc; }
138
139private:
140 /// Makes a deep copy of the specified screen object.
141 ///
142 /// @param screen [in] Screen object to copy.
143 ///
144 /// @return Copy of the specified screen object.
145 ///
146 MeaPositionScreen& Copy(const MeaPositionScreen& screen) {
147 if (&screen != this) {
148 m_primary = screen.m_primary;
149 m_rect = screen.m_rect;
150 m_res = screen.m_res;
151 m_manualRes = screen.m_manualRes;
152 m_desc = screen.m_desc;
153 }
154 return *this;
155 }
156
157
158 bool m_primary; ///< Is this the primary screen
159 MeaFRect m_rect; ///< Screen rectangle expressed in the units that were
160 ///< in effect at the time the screen object was created.
161 MeaFSize m_res; ///< Screen resolution expressed in the units that were
162 ///< in effect at the time the screen object was created.
163 bool m_manualRes; ///< Is the resolution calibrated manually.
164 CString m_desc; ///< Description of the screen
165};
