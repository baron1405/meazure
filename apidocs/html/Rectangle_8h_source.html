29/// A rectangle element. The rectangle is used by the Rectangle measurement
30/// tool. The rectangle is positioned by specifying the location of its
31/// its top left corner (point 1) and its bottom right corner (point 2).
32/// The rectangle is formed by creating a rectangular window and punching
33/// out a smaller rectangle from the window's interior. The remaining
34/// region forms the line of the rectangle.
35///
36class MeaRectangle : public MeaGraphic {
37
38public:
39 /// Constructs a rectangle. Prior to displaying the rectangle with
40 /// the Show() method, the Create() method must be called to create
41 /// the rectangle's window.
42 ///
43 MeaRectangle();
44
45 /// Destroys the rectangle.
46 ///
47 virtual ~MeaRectangle();
48
49 /// Creates the window for the rectangle. This method must be called
50 /// before the rectangle can be displayed and before any operation that
51 /// attempts to manipulate the rectangle's window.
52 ///
53 /// @param thk [in] Thickness of the rectangle's line, in pixels.
54 /// @param shrink [in] Distance, in pixels, to shrink line of the
55 /// rectangle from the top left and bottom right
56 /// corners. The typical use for this parameter is in
57 /// the Rectangle tool. Shrinking the line allows the
58 /// contents under the corner points to be clear for
59 /// viewing and measuring.
60 /// @param parent [in] Parent window or <b>nullptr</b> if the rectangle
61 /// does not have an immediate parent. If a parent
62 /// window is specified, the rectangle is created as
63 /// a child window (<b>WS_CHILD</b>). Otherwise the
64 /// rectangle is created as a popup window
65 /// (<b>WS_POPUP</b>) and is assigned the main window
66 /// as a parent.
67 ///
68 /// @return <b>true</b> if the window is created successfully.
69 ///
70 bool Create(int thk, int shrink, const CWnd* parent = nullptr);
71
72 /// Hides the graphic. Call the Create() method prior
73 /// to calling this method for the first time.
74 ///
75 virtual void Hide() override;

77 /// Sets the position of the rectangle.

82 void SetPosition(const POINT& point1, const POINT& point2);

93 /// Windows GDI was not optimized for doing crazy things like creating
94 /// a window that is a thin rectangle. Because of this, drawing the window
95 /// is very expensive. To minimize visual artifacts when the rectangle is
96 /// repositioned or resized, a timer is used to effectively add damping
97 /// to the drawing process. Even if the user may be making quick changes
98 /// to the rectangle, the time delay reduces the number of redraws. When the
99 /// SetPosition() method is called, it just sets a timer and returns. When
100 /// the timer expires, this method is called and actually redraws the
101 /// window.
102 ///
103 /// @param wParam [in] not used
104 /// @param lParam [in] not used
105 ///
106 /// @return The return is not used so 0 is returned.
107 ///
108 afx_msg LRESULT OnHPTimer(WPARAM wParam, LPARAM lParam);
109
110private:
111 /// Copy constructor is purposely undefined.
112 ///
113 MeaRectangle(const MeaRectangle&);
114
115 /// Assignment operator is purposely undefined
116 ///
117 MeaRectangle& operator=(const MeaRectangle&);
118
119 /// Sets the windows region effectively forming the rectangle. The
120 /// rectangle is formed by first creating a normal rectangular window.
121 /// A smaller rectangle is formed based on the desired thickness for
122 /// the rectangle's line. This smaller rectangle is used to punch out
123 /// the window's region, thereby forming the rectangle. The shrinking
124 /// of the rectangle lines around the top left and bottom right corners,
125 /// if any, is done by using small rectangles the size of the shrink
126 /// to punch out the corners of the window.
127 ///
128 void SetRegion();
129

136 int m_shrink; ///< Number pixels to hang back from the top left and bottom right corners of the rectangle.
137};
