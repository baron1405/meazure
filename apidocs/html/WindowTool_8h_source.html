30/// Window measurement tool. This tool tracks the mouse pointer and attempts
31/// to identify the deepest window in the window hierarchy under the pointer.
32/// the dimensions of that window are displayed. The tool consists of a
33/// rectangle that is used to highlight the window selected for measurement. In
34/// addition, a data window is attached to the rectangle to provide a measurement
35/// information popup.
36///
37class MeaWindowTool : public MeaRadioTool {
38
39public:
40 static const CString kToolName; ///< "WindowTool"
41
42 /// Constructs a new instance of a Window measurement tool. To use
43 /// a newly constructed Window tool, the Enable() method must be called.
44 /// Enable() will call the Create() method the first time it is called
45 /// if Create() has not previously been called.
46 ///
47 /// @param mgr [in] Tool manager parent of the Window tool.
48 ///
49 explicit MeaWindowTool(MeaToolMgr* mgr);
50
51 /// Hides and then destroys the Window tool.
52 //
53 virtual ~MeaWindowTool();
54
55 /// Displays the Window tool. If the Create() method has not been
56 /// previously called, it will be called by Enable.
57 ///
58 virtual void Enable() override;
59
60 /// Hides the Window tool.
61 ///
62 virtual void Disable() override;
63
64 /// Requests that the tool update the display of the window data.
65 ///
66 /// @param reason [in] Reason why the update has been requested
67 ///
68 virtual void Update(MeaUpdateReason reason) override;
69
70 /// Indicates whether the tool uses crosshairs. The Window tool
71 /// does not use crosshairs.
72 ///
73 /// @return For this class this method always returns <b>false</b>.
74 ///
75 virtual bool HasCrosshairs() const override;
76
77 /// Visually strobes the tool. The tool data window attached to the
78 /// window rectangle is cycled between hidden and visible. Strobing
79 /// the tool is done to visually confirm region capture and position
80 /// recording.
81 ///
82 virtual void Strobe() override;
83
84 /// Indicates whether the tool can define a rectangular region that
85 /// can then be read for a screen capture. The Window tool defines
86 /// such a rectangular area.
87 ///
88 /// @return <b>true</b> if the tool defines a non-empty rectangular region.
89 ///
90 virtual bool HasRegion() override;
91
92 /// Returns the rectangle defined by the tool. This rectangle can then
93 /// be read for a screen capture.
94 ///
95 /// @return The rectangle formed by the tool.
96 ///
97 virtual RECT GetRegion() override;
98
99 /// Called by the OS when the mouse pointer is moved. Via this method
100 /// the window under the pointer is identified.
101 ///
102 /// @param wParam [in] OS Message ID
103 /// @param lParam [in] MOUSEHOOKSTRUCT
104 ///
105 virtual void OnMouseHook(WPARAM wParam, LPARAM lParam) override;
106
107 /// Returns the position of the mouse pointer.
108 ///
109 /// @return Position of the mouse pointer.
110 ///
111 virtual const POINT& GetPosition() const override;
112
113 /// Records the position of the window rectangle. This method is called
114 /// by the position log manager to record a position.
115 ///
116 /// @param position [in] The position of the window rectangle is recorded
117 /// into the position log manager's position object.
118 ///
119 virtual void GetPosition(MeaPositionLogMgr::Position& position) const override;
120
121 /// Returns the name of the tool. Each tool has a unique name
122 /// which is used to identify the tool in profiles and position
123 /// logs.
124 ///
125 /// @return Name of the tool ("WindowTool").
126 ///
127 virtual CString GetToolName() const override;
128
129 /// Returns a resource ID for the string used to label this
130 /// tool on the data display.
131 ///
132 /// @return Resource ID for the label.
133 ///
134 virtual UINT GetLabelId() const override;
135
136 /// Called to notify the tool that the colors in the MeaColors class
137 /// have been changed. The window rectangle color and the data window
138 /// opacity are updated.
139 ///
140 virtual void ColorsChanged() override;
141
142private:
143 /// Called by the win32 EnumChildWindows function when it enumerates
144 /// child windows. Also tests to see if the child window is under the
145 /// cursor.
146 ///
147 /// @param hwnd [in] Child window to test
148 /// @param lParam [in] <b>this</b> pointer
149 ///
150 /// @return TRUE if enumeration should continue.
151 ///
152 static BOOL CALLBACK EnumChildProc(HWND hwnd, LPARAM lParam);
153
154 /// Creates the tool's graphical components. The components include
155 /// the window rectangle and the data window. The Enable() method must
156 /// be called to make the tool visible.
157 ///
158 void Create();
159
160 /// Attempts to locate a window under the mouse pointer. The window
161 /// hierarchy is searched to identify the deepest window in the hierarchy.
162 ///
163 /// @return <b>true</b> if the pointer has moved over a new window.
164 ///
165 bool FindWindow();
166
167

172 HWND m_currentWnd; ///< Window currently highlighted
173 HWND m_hiliteWnd; ///< Window to highlight
174 MeaDataWin m_dataWin; ///< Data window tooltip
175};
