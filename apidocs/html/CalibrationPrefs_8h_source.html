32/// Property page for screen resolution calibration preferences. Using this
33/// property page, the screen resolution can be calibrated using a number of
34/// different techniques including direct measurement of the screen.
35///
36class MeaCalibrationPrefs : public CPropertyPage {
37
38public:
39 /// Indicates how the screen resolution is determined.
40 ///
41 enum ResolutionMode {
42 AutoRes = 0, ///< The operating system is supplying the screen resolution.
43 ManualRes = 1 ///< The screen resolution is supplied by the user.
44 };
45
46 /// Indicates which unit system was used to manually calibrate the
47 /// screen resolution.
48 ///
49 enum ResolutionUnits {
50 UseInches = 0, ///< The resolution was calibrated using inches.
51 UseCentimeters = 1 ///< The resolution was calibrated using centimeters.
52 };
53
54
55 /// This structure represents the characteristics of a screen. There
56 /// can be multiple screens connected to a single machine, each with
57 /// its own size, resolution can calibration mode.
58 struct Screen {
59
60 Screen() : m_resMode(AutoRes), m_resUnits(UseInches) {}
61
62 int m_resMode; ///< Indicates whether calibration for the screen resolution is manual or automatic.
63 ///< Must be int due to DDX requirements.
64 int m_resUnits; ///< If manual resolution calibration, indicates if calibration in inches or centimeters.
65 ///< Must be int due to DDX requirements.
66 MeaFSize m_res; ///< Screen resolution.
67 MeaFSize m_size; ///< Screen size, in inches or centimeters.
68 };

116 /// Overrides the base class UpdateData method. The base class asserts if
117 /// a recursive call to UpdateData is attempted. In case there is a recursive

119 /// See CWnd::UpdateData for an explanation of the parameter and return value.
120 ///
121 BOOL UpdateData(BOOL bSaveAndValidate = TRUE);

214 /// Same as the DDX UpdateData method except the specified UI
215 /// object is not updated. This prevents cyclic updating.
216 ///
217 /// @param lockoutId [in] ID of UI object that should not be updated.

219 ///
220 /// @return <b>true</b> if data update was successful.
221 ///
222 bool LockedUpdateData(UINT lockoutId, bool bSaveAndValidate = true);
223
224 /// Indicates whether manual resolution calibration is being done in metrics units.
225 ///
226 /// @return <b>true</b> if the calibration is being done in centimeters.
227 ///
228 bool IsMetric() { return GetScreen().m_resUnits == UseCentimeters; }
229
230 /// Performs DDX handling for the resolution text fields. Internally
231 /// resolution values are maintained in pixels/inch. If calibration is
232 /// being done using centimeters, this method converts to/from inches.
233 ///
234 /// @param pDX [in] DDX object for exchanging data between the page UI
235 /// and the member variables.
236 /// @param nIDC [in] ID of the UI control.
237 /// @param resValue [in, out] Resolution in either pixels/inch or pixels/centimeter.
238 ///
239 void DDX_ResText(CDataExchange* pDX, int nIDC, double& resValue);
240
241 /// Performs DDX handling for the screen dimension text fields. Internally
242 /// dimension values are maintained in inches. If calibration is being done
243 /// using centimeters, this method converts to/from inches.
244 ///
245 /// @param pDX [in] DDX object for exchanging data between the page UI
246 /// and the member variables.
247 /// @param nIDC [in] ID of the UI control.
248 /// @param dimValue [in, out] Dimensions in either inches or centimeters.
249 ///
250 void DDX_DimText(CDataExchange* pDX, int nIDC, double& dimValue);
251
252 /// Performs data validation on all appropriate text field values. For these
253 /// text fields, a value is considered valid if it is greater than zero.
254 ///
255 /// @return <b>true</b> if all appropriate field values are valid.
256 ///
257 bool Validate();
258
259 /// Performs data validation on the specified text field. For these test
260 /// fields, a value is considered valid if it is greater than zero.
261 ///
262 /// @param id [in] ID of the text field whose value is to be validated.
263 ///
264 /// @return <b>true</b> if the field value is valid.
265 ///
266 bool ValidateField(UINT id);
267
268
269 /// Updates the display of all data. This includes updating the text field
270 /// values and slider positions.
271 ///
272 void UpdateView();
273
274 /// Determines the UI based on the current screen information. The information
275 /// displayed on this page is dependent on the screen on which it is shown.
276 ///
277 void UpdateCurrentScreen();
278
279 /// Called when the position of one of the calibration sliders is changed.
280 ///
281 /// @param pos [in] New position of the slider. This is the resolution
282 /// either in pixels/inch or 2 * pixels/centimeter.
283 /// @param id [in] ID of the slider that has been changed.
284 ///
285 /// @return TRUE to indicate successful processing.
286 ///
287 LRESULT OnCaliperChange(WPARAM pos, LPARAM id);
288
289 /// Obtains the screen information for the current screen (i.e. the screen
290 /// on which the page is located).
291 ///
292 /// @return Screen information structure.
293 ///
294 Screen& GetScreen() {
295 ScreenMap::iterator citer = m_screens.find(m_currentIter);
296 assert(citer != m_screens.end());
297 return (*citer).second;
298 }
