34/// Provides a screen magnifier window complete with freeze frame, optional
35/// pixel grid and pixel color capture. A number of color spaces are available
36/// for the display of the captured pixel color. The contents of the screen are
37/// read and displayed in this window using a 1X through 32X zoom factor.
38/// Note that layered windows will not appear in the magnifier. Similarly,
39/// windows rendered using DirectX will appear black since they are drawing
40/// directly to the hardware and displaying using a hardware overlay.
41///
42/// Typically, the magnifier is centered on the active graphic element, usually
43/// a crosshair.
44///
45class MeaMagnifier : public CWnd
46{
47public:
48 /// The magnifier displays the color of the pixel at the center of the
49 /// window. This point typically corresponds to the hot spot of an active
50 /// graphic element such as a crosshair. The color can be displayed in a
51 /// number of different color spaces. This enumeration indicates which
52 /// color space is in use.
53 ///
54 enum ColorFmt {
55 RGBFmt = 0, ///< Red, green and blue as individual values between 0 and 255 inclusive.
56 RGBHexFmt = 1, ///< Red, green and blue as a composite hex number of the form \#RRGGBB.
57 CMYFmt = 2, ///< Cyan, magenta and yellow as individual values between 0 and 255 inclusive.
58 CMYKFmt = 3, ///< Cyan, magenta, yellow and black as individual values.
59 HSLFmt = 4, ///< Hue, saturation and lightness as individual values.
60 YCbCrFmt = 5, ///< Luminance and chrominance as individual values.
61 YIQFmt = 6 ///< Luminance, and inphase and quadrature chrominance channels.
62 };
63
64 /// Indicates whether the magnifier is running or in freeze frame mode.
65 ///
66 enum class RunState {
67 Run = 0, ///< Magnifier is in real time mode.
68 Freeze = 1 ///< Magnifier is in freeze frame mode.
69 };

141 /// Zooms in the magnifier. If the magnifier is already
142 /// at the minimum zoom, calling this method has no effect.
143 ///
144 void ZoomIn() { SetZoomIndex(m_zoomIndex + 1); }
145

147 /// at the maximum zoom, calling this method has no effect.
148 ///

151 /// Indicates whether it is possible to zoom in the magnifier.
152 ///
153 /// @return <b>true</b> if the magnifier can zoom in further.
154 ///
155 bool CanZoomIn() const { return m_zoomIndex < kMaxZoomIndex; }
156
157 /// Indicates whether it is possible to zoom out the magnifier.
158 ///
159 /// @return <b>true</b> if the magnifier can zoom out further.
160 ///
161 bool CanZoomOut() const { return m_zoomIndex > kMinZoomIndex; }

164 /// Sets a specific zoom factor index.
165 ///
166 /// @param zoomIndex [in] Index that selects a zoom factor.
167 ///
168 void SetZoomIndex(int zoomIndex) {
169 if (zoomIndex >= kMinZoomIndex && zoomIndex <= kMaxZoomIndex) {
170 m_zoomIndex = zoomIndex;
171 }
172 m_zoomSlider.SetPos(m_zoomIndex);
173 Update();
174 }
