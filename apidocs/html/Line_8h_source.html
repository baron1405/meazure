30/// A line element. The line is used by many tools including the Line
31/// and Grid tools. The line is positioned by specifying the location
32/// of its two end points. The line is formed by arranging adjacent
33/// single pixel rectangular regions.
34///

100 /// Windows GDI was not optimized for doing crazy things like creating
101 /// a window that is a thin line. Because of this, drawing the window
102 /// is very expensive. To minimize visual artifacts when the line is
103 /// repositioned or resized, a timer is used to effectively add damping
104 /// to the drawing process. Even if the user may be making quick changes
105 /// to the line, the time delay reduces the number of redraws. When the
106 /// SetPosition() method is called, it just sets a timer and returns. When
107 /// the timer expires, this method is called and actually redraws the
108 /// window.
109 ///
110 /// @param wParam [in] not used
111 /// @param lParam [in] not used
112 ///
113 /// @return The return is not used so 0 is returned.
114 ///
115 afx_msg LRESULT OnHPTimer(WPARAM wParam, LPARAM lParam);

141 /// The window region is composed of single pixel rectangles arranged
142 /// in a line from the start point to the end point. The location of
143 /// each rectangle is determined using the Bresenham algorithm adapted
144 /// from "Graphics Gems", Academic Press, 1990, p. 685. The line needs
145 /// to be created in this brute force way because relying on the polygon
146 /// region method produces a horrible looking line.
147 ///
148 void PlotLine();
149
150 /// The window is composed of a series of single pixel rectangles
151 /// arranged in a line. At each point on the line the coordinates of
152 /// a rectangle surrounding that point are added to an array of points.
153 /// This array of points will eventually be handed to the GDI function
154 /// CreatePolyPolygonRgn.
155 ///
156 /// @param x [in] X coordinate of the point on the line
157 /// @param y [in] Y coordinate of the point on the line
158 ///
159 void AddPoint(int x, int y) {
160 CPoint p(x, y);
161 POINT* arr = &m_arr[m_count << 2];
162
163 // (x,y) (x+1, y)
164 // ---------
165 // | |
166 // | |
167 // | |
168 // ---------
169 // (x,y+1) (x+1,y+1)
170
171
172 // (x,y)
173 //
174 *arr++ = p;
175
176 // (x+1,y)
177 //
178 p.x++;
179 *arr++ = p;
180
181 // (x+1,y+1)
182 //
183 p.y++;
184 *arr++ = p;
185
186 // (x, y+1)
187 //
188 p.x--;
189 *arr = p;
190
191 // Keep track of the number of rectangles
192 //
193 m_count++;
194 }
