26/// Interface for objects that need to be informed about changes to the
27/// position log. Classes that wish to be informed about changes to the
28/// position log file inherit from this interface as a mix-in. The class
29/// then implements the methods declared by this interface.
30///
31class MeaPositionLogObserver
32{
33public:
34 /// Interface does nothing in a constructor.
35 ///
36 MeaPositionLogObserver() { }
37
38 /// Interface does nothing in a destructor.
39 ///
40 virtual ~MeaPositionLogObserver() { }
41
42
43 /// Called when a position log file is loaded.
44 ///
45 virtual void LogLoaded() = 0;
46
47 /// Called when a position log file is saved.
48 ///
49 virtual void LogSaved() = 0;
50
51 /// Called when a new position is recorded.
52 /// @param posIndex [in] Index of the new position.
53 virtual void PositionAdded(int posIndex) = 0;
54
55 /// Called when an existing position is replaced with a new position.
56 /// @param posIndex [in] Index of the replaced position.
57 virtual void PositionReplaced(int posIndex) = 0;
58
59 /// Called when a position is deleted.
60 /// @param posIndex [in] Index of the deleted position.
61 virtual void PositionDeleted(int posIndex) = 0;
62
63 /// Called when all recorded positions have been deleted.
64 ///
65 virtual void PositionsDeleted() = 0;
66};
