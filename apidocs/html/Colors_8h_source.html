39 /// Represents a color in the Cyan-Magenta-Yellow color space.
40 ///
41 struct CMY {
42 int cyan;
43 int magenta;
44 int yellow;
45
46 constexpr CMY() : cyan(0), magenta(0), yellow(0) {}
47 constexpr CMY(int c, int m, int y) : cyan(c), magenta(m), yellow(y) {}
48 };

50 /// Represents a color in the Cyan-Magenta-Yellow-Black color space. By convention, the letter "K" represents
51 /// the black component.
52 ///
53 struct CMYK {
54 int cyan;
55 int magenta;
56 int yellow;
57 int black;
58
59 constexpr CMYK() : cyan(0), magenta(0), yellow(0), black(0) {}
60 constexpr CMYK(int c, int m, int y, int k) : cyan(c), magenta(m), yellow(y), black(k) {}
61 };

63 /// Represents a color in the Hue (H), Saturation (S), and Lightness (L) color space.
64 ///
65 struct HSL {
66 int hue;

68 int lightness;
69
70 constexpr HSL() : hue(0), saturation(0), lightness(0) {}
71 constexpr HSL(int h, int s, int l) : hue(h), saturation(s), lightness(l) {}

74 /// Represents a color in the Luminance (Y), Blue-Difference (Cb) and Red-Difference (Cr) chrominance color
75 /// space. See https://en.wikipedia.org/wiki/YCbCr for more information.
76 ///
77 struct YCbCr {
78 int y;
79 int cb;
80 int cr;
81
82 constexpr YCbCr() : y(0), cb(0), cr(0) {}
83 constexpr YCbCr(int luma, int blueChroma, int redChroma) : y(luma), cb(blueChroma), cr(redChroma) {}
84 };

86 /// Represents a color in the Luminance (Y), In-phase (I) and Quadrature (Q) chrominance color space.
87 ///
88 struct YIQ {
89 int y;
90 int i;
91 int q;
92
93 constexpr YIQ() : y(0), i(0), q(0) {}
94 constexpr YIQ(int luma, int inphase, int quadrature) : y(luma), i(inphase), q(quadrature) {}
95 };

97 /// Represents a color in the CIE 1931 XYZ color space. The tristimulus values are the luminance (Y),
98 /// the blue-related (Z) and the RGB mixed (X).
99 ///
100 struct XYZ {
101 double x;
102 double y;
103 double z;
104
105 constexpr XYZ() : x(0.0), y(0.0), z(0.0) {}
106 constexpr XYZ(double xin, double yin, double zin) : x(xin), y(yin), z(zin) {}
107 };
108
109 /// Represents a color in the CIE L*a*b* color space. The values are the lightness (L*), the green-red axis
110 /// chroma (a*) and the blue yellow axis chroma (b*).
111 ///
112 struct Lab {
113 double l;
114 double a;
115 double b;
116
117 constexpr Lab() : l(0.0), a(0.0), b(0.0) {}
118 constexpr Lab(double lstar, double astar, double bstar) : l(lstar), a(astar), b(bstar) {}
119 };

234 /// Calculates the visual difference between the two specified colors. The difference is calculated using the
235 /// CIEDE2000 algorithm. See https://en.wikipedia.org/wiki/Color_difference#CIEDE2000,
236 /// http://www.brucelindbloom.com/index.html?Eqn_DeltaE_CIE2000.html and
237 /// https://hajim.rochester.edu/ece/sites/gsharma/ciede2000/ciede2000noteCRNA.pdf.
238 ///
239 /// @param color1 [in] First color in the difference
240 /// @param color2 [in] Second color in the difference
241 /// @return A value representing the difference between the two specified colors. The smaller the difference,
242 /// the visually closer the two colors. The value is never negative.
243 ///
244 double ColorDifference(const Lab& color1, const Lab& color2);

271 /// Converts from the RGB color space to the Cyan (C), Magenta (M) and Yellow (Y) color space. The conversion is
272 /// done using the algorithm for RGB to CMY conversion presented at http://www.easyrgb.com/en/math.php. The input
273 /// and output range is [0, 255].
274 ///
275 /// @param rgb [in] RGB color
276 /// @return CMY color
277 ///
278 constexpr CMY RGBtoCMY(COLORREF rgb) {
279 return CMY(255 - GetRValue(rgb), 255 - GetGValue(rgb), 255 - GetBValue(rgb));
280 }

282 /// Converts from the RGB color space to the Cyan (C), Magenta (M), Yellow (Y) and Black (K) color space. The
283 /// conversion is done using the algorithm presented at http://www.easyrgb.com/en/math.php. The input and output
284 /// range is [0, 255].
285 ///
286 /// @param rgb [in] RGB color
287 /// @return CMYK color
288 ///
289 CMYK RGBtoCMYK(COLORREF rgb);

291 /// Converts from the RGB color space to the Hue (H), Saturation (S) and Lightness (L) color space. The conversion
292 /// is done using the algorithm for RGB to HSL conversion presented at http://www.easyrgb.com/en/math.php. The
293 /// input range is [0, 255]. The output range is scaled such that H values are in degrees in the range [0, 360),
294 /// and S and L values are percentages in the range [0, 100].
295 ///
296 /// @param rgb [in] RGB color
297 /// @return HSL color
298 ///
299 HSL RGBtoHSL(COLORREF rgb);
300
301 /// Converts from the HSL color space to the RGB color space. The conversion is done using the algorithm for HSL
302 /// to RGB conversion presented at http://www.easyrgb.com/en/math.php. The input range is H in degrees in the
303 /// range [0, 360), and S and L as percentages in the range [0, 100]. The output range is [0, 255].
304 ///
305 /// @param hsl [in] HSL color.
306 /// @return RGB color
307 ///
308 COLORREF HSLtoRGB(const HSL& hsl);
309
310 /// Converts from the RGB color space to the Luminance (Y), Blue-Difference (Cb), Red-Difference (Cr) color space.
311 /// The conversion is done using the ITU-R BT.601 coefficients for digital YCbCr from digital RGB with input
312 /// range [0, 255]. The output range is [16, 235] for Y, and [16, 240] for both Cb and Cr)values. See
313 /// https://en.wikipedia.org/wiki/YCbCr for more information.
314 ///
315 /// @param rgb [in] RGB color
316 /// @return YCbCr color
317 ///
318 YCbCr RGBtoYCbCr(COLORREF rgb);
319
320 /// Converts from the RGB color space to the Luminance (Y), In-phase (I), Quadrature (Q) color space. The
321 /// conversion is done using the NTSC 1953 colorimetry coefficients. The input range is [0, 255]. The output range
322 /// is [0, 255] for Y, [-152, 152] I and [-133, 133] for quadrature. See https://en.wikipedia.org/wiki/YIQ for
323 /// more information.
324 ///
325 /// @param rgb [in] RGB color
326 /// @return YIQ color
327 ///
328 YIQ RGBtoYIQ(COLORREF rgb);
329
330 /// Converts from the RGB color space to the CIE 1931 XYZ color space. The conversion is done assuming inputs in
331 /// the Standard RGB color space and a D65 2 degree standard illuminant. The input range is [0, 255]. The output is
332 /// scaled by 100.0 resulting in the X range [0.0, 95.0470], Y [0.0, 100.0] and Z [0.0, 108.8830]. See
333 /// http://www.brucelindbloom.com/index.html?Math.html and http://www.easyrgb.com/en/math.php for more information.
334 ///
335 /// @param rgb [in] RGB color
336 /// @return XYZ color
337 ///
338 XYZ RGBtoXYZ(COLORREF rgb);
339
340 /// Converts from the CIE 1931 XYZ color space to the CIE L*a*b* color space. The conversion is done assuming
341 /// the Standard RGB color space and a D65 2 degree standard illuminant. The inputs are scaled XYZ with ranges
342 /// X [0.0, 95.0470], Y [0.0, 100.0000] and Z [0.0, 108.8830]. The output range for L is [0.0, 100.0],
343 /// and unbound for a and b. See http://www.brucelindbloom.com/index.html?Math.html and
344 /// http://www.easyrgb.com/en/math.php for more information.
345 ///
346 /// @param xyz [in] XYZ color
347 /// @return L*a*b* color
348 ///
349 Lab XYZtoLab(const XYZ& xyz);
350
351 /// Converts the specified RGB color to the CIE L*a*b* color space. This is a convenience function which converts

353 /// details on these conversions.
354 ///
355 /// @param rgb [in] RGB color
356 /// @return L*a*b* color
357 ///
358 Lab RGBtoLab(COLORREF rgb);
