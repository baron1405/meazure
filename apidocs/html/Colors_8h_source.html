29/// Represents a color in the Hue-Saturation-Lightness color space.
30/// All values are between 0.0 and 1.0.
31///
32typedef struct {
33 double hue;
34 double saturation;
35 double lightness;
36} HSL;

39/// A color style sheet class that provides the colors and opacities used
40/// by the crosshairs, lines and other graphic elements.
41///
42class MeaColors {
43
44public:
45 /// Identifies the item whose color and opacity are maintained by this class.
46 ///
47 enum Item {
48 LineFore, ///< Line foreground color.
49 CrossHairBack, ///< Crosshair background color.
50 CrossHairBorder, ///< Crosshair edge color.
51 CrossHairHilite, ///< Crosshair highlight color.
52 CrossHairOpacity, ///< Opacity of the crosshair.
53 RulerBack, ///< Ruler background color.
54 RulerBorder, ///< Ruler edge and tick mark color.
55 RulerOpacity ///< Ruler opacity.
56 };
57
58
59 /// Initializes the class by establishing the default colors.
60 ///
61 static void Initialize();
62
63 /// Resets all colors to their default values.
64 ///
65 static void Reset() {
66 m_colors = m_defColors;
67 }
68
69 /// Persists the color settings to the specified profile.
70 ///
71 /// @param profile [in] Profile into which the color settings
72 /// are persisted.
73 ///
74 static void SaveProfile(MeaProfile& profile);
75
76 /// Restores the color settings from the specified profile.
77 ///
78 /// @param profile [in] Profile from which the color settings
79 /// are restored.
80 ///
81 static void LoadProfile(MeaProfile& profile);
82
83 /// Called when the master application reset is invoked. The colors
84 /// are reset to their default values.
85 ///
86 static void MasterReset();
87
88 /// Sets the specified item to the specified color.
89 /// @param item [in] Color item to set.
90 /// @param clr [in] Color to set.
91 static void Set(Item item, COLORREF clr) { m_colors[item] = clr; }
92
93 /// Sets the specified item to the specified opacity
94 /// @param item [in] Item whose opacity is to be set.
95 /// @param opacity [in] Opacity to set, where 0 is completely transparent
96 /// and 255 is completely opaque.
97 static void SetA(Item item, BYTE opacity) { m_colors[item] = RGB(opacity, 0, 0); }
98
99 /// Returns the color of the specified item.
100 /// @param item [in] Item whose color is desired.
101 /// @return Color of the specified item.
102 static COLORREF Get(Item item) { return m_colors[item]; }
103
104 /// Returns the value of the red color component of the specified item.
105 /// @param item [in] Item whose color component is desired.
106 /// @return Red color component of the specified item.
107 static BYTE GetR(Item item) { return GetRValue(m_colors[item]); }
108
109 /// Returns the value of the green color component of the specified item.
110 /// @param item [in] Item whose color component is desired.
111 /// @return Green color component of the specified item.
112 static BYTE GetG(Item item) { return GetGValue(m_colors[item]); }
113
114 /// Returns the value of the blue color component of the specified item.
115 /// @param item [in] Item whose color component is desired.
116 /// @return Blue color component of the specified item.
117 static BYTE GetB(Item item) { return GetBValue(m_colors[item]); }
118
119 /// Returns the opacity of the specified item.
120 /// @param item [in] Item whose opacity is desired.
121 /// @return Opacity of the specified item, where 0 is completely
122 /// transparent and 255 is completely opaque.
123 static BYTE GetA(Item item) { return GetRValue(m_colors[item]); }
124
125 /// Returns the default color for the specified item.
126 /// @param item [in] Item whose default color is desired.
127 /// @return Default color for the specified item.
128 static COLORREF GetDef(Item item) {
129 return m_defColors[item];
130 }
131
132 /// Performs linear interpolation between the specified RGB color.
133 /// The interpolation is performed in HSL space, which provides a
134 /// more visually appealing result.
135 ///
136 /// @param startRGB [in] Starting interpolation point.
137 /// @param endRGB [in] Ending interpolation point.
138 /// @param percent [in] Interpolation point between startRGB and endRGB as a percentage.
139 ///
140 /// @return Interpolated color.
141 ///
142 static COLORREF InterpolateColor(COLORREF startRGB, COLORREF endRGB, int percent);
143
144 /// Converts from the RGB color space to the HSL color space.
145 ///
146 /// @param rgb [in] RGB colors as values between 0 and 255.
147 /// @param hsl [out] HSL colors as values between 0.0 and 1.0
148 ///
149 static void RGBtoHSL(COLORREF rgb, HSL& hsl);
150
151 /// Converts from the HSL color space to the RGB color space.
152 ///
153 /// @param hsl [in] HSL colors as values between 0.0 and 1.0.
154 /// @param rgb [out] RGB colors as values between 0 and 255.
155 ///
156 static void HSLtoRGB(const HSL& hsl, COLORREF& rgb);
157
158private:
159 /// All members of this class are static. No instances
160 /// of this class are ever created.
161 ///
162 MeaColors() {}
163
164 /// All members of this class are static. No instances
165 /// of this class are ever created.
166 ///
167 ~MeaColors() {}
168
169 typedef std::map<Item, COLORREF> Colors; ///< Maps items to their colors.
170
171 /// Determines the minimum of two values.
172 /// @return Minimum value.
173 static double Min(double a, double b) { return ((a < b) ? a : b); }
174
175 /// Determines the maximum of two values.
176 /// @return Maximum value.
177 static double Max(double a, double b) { return ((a > b) ? a : b); }
178
179 /// Converts a hue to a RGB component value based on the specified
180 /// weighting factors.
181 /// @param m1 [in] Weighting factor between lightness and saturation.
182 /// @param m2 [in] Weighting factor between lightness and saturation.
183 /// @param h [in] Hue value.
184 /// @return Either an R, G, or B value between 0.0 and 1.0. The
185 /// component returned depends on the weighting factors
186 /// specified.
187 static double HuetoRGB(double m1, double m2, double h);
188
189
190 static Colors m_defColors; ///< Map of default colors.
191 static Colors m_colors; ///< Map of active colors.
192};
