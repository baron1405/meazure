34 /// Represents a color in the Hue-Saturation-Lightness color space.
35 /// All values are between 0.0 and 1.0.
36 ///
37 struct HSL {
38 double hue;
39 double saturation;
40 double lightness;
41
42 HSL() : hue(0.0), saturation(0.0), lightness(0.0) {}
43 HSL(double h, double s, double l) : hue(h), saturation(s), lightness(l) {}
44 };

47 /// Identifies the item whose color and opacity are maintained by this class.

55 RulerBack, ///< Ruler background color.
