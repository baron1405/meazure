35/// Property sheet holding all program preference pages.
36///
37class MeaPreferences : public CPropertySheet {
38 DECLARE_DYNAMIC(MeaPreferences)
39
40public:
41 /// Preference page selection indices. With the
42 /// exception of kAllPages, these indices must be
43 /// consecutive and in the order in which pages are
44 /// to be listed on the property sheet.
45 ///
46 enum PreferencePage {
47 kAllPages = -1, ///< Refers to all property pages.
48 kCalibrationPage = 0, ///< Selects the preference page for screen resolution calibration.
49 kRulerPage = 1, ///< Selects the preference page for ruler appearance.
50 kToolsPage = 2, ///< Selects the preference page for crosshair and tool appearance.
51 kPrecisionPage = 3, ///< Selects the preference page for numerical display precision.
52 kCustomPage = 4, ///< Selects the preference page for defining custom measurement units.
53 kAdvancedPage = 5 ///< Selects the preference page for setting a startup profile.
54 };
55
56
57 /// Constructs the property sheet.
58 ///
59 /// @param nIDCaption [in] String table ID for the title of the preferences dialog.
60 /// @param pParentWnd [in] Parent window for the preferences dialog. If the value is
61 /// nullptr, the dialog is a popup.
62 /// @param iSelectPage [in] Preference page to display when the preferences dialog is shown.
63 /// Default is the first page.
64 ///
65 explicit MeaPreferences(UINT nIDCaption, CWnd* pParentWnd = nullptr, UINT iSelectPage = 0);
66
67 /// Destroys a property sheet instance.
68 ///
69 virtual ~MeaPreferences();
70
71 /// Displays the preference page for defining custom measurement units.
72 ///
73 void ShowCustomPrecision();
74
75
76 MeaCalibrationPrefs m_calibrationPrefs; ///< Preference page for screen resolution calibration.
77 MeaRulerPrefs m_rulerPrefs; ///< Preference page for ruler appearance.
78 MeaToolsPrefs m_toolsPrefs; ///< Preference page for crosshair and tool appearance.
79 MeaPrecisionPrefs m_precisionPrefs; ///< Preference page for numerical display precision.
80 MeaCustomUnitsPrefs m_customPrefs; ///< Preference page for defining custom measurement units.
81 MeaAdvancedPrefs m_advancedPrefs; ///< Preference page for setting a startup profile.
82
83protected:
84 /// Called when the property sheet dialog is dragged. This method calls the
85 /// calibration preference page to allow it to update to reflect the current
86 /// screen's resolution calibration state.
87 ///
88 /// @param x [in] x coordinate of the upper left corner of the dialog window in screen coordinates.
89 /// @param y [in] y coordinate of the upper left corner of the dialog window in screen coordinates.
90 ///
91 afx_msg void OnMove(int x, int y);
92
93 /// Called when the property sheet dialog is about to be created. This method
94 /// changes the tab style of the dialog from stacked tabs to horizontally
95 /// scrolling tabs.
96 ///
97 /// @param lpCreateStruct [in] dialog creation information structure.
98 ///
99 afx_msg int OnCreate(LPCREATESTRUCT lpCreateStruct);
100
101 /// Calls the parent of the property sheet dialog with the apply preferences
102 /// message and the index of the currently active preference page. In addition,
103 /// the modified flag is cleared on all preference pages.
104 ///
105 afx_msg void OnApply();
106
107 /// Works around a design flaw in the CPropertySheet class. The current
108 /// property page is not remembered between invocations of the dialog.
109 /// Overriding this method allows catpuring the current page so it can
110 /// be restored the next time the dialog is displayed.
111 ///
112 /// @param wParam [in] See the CWnd::OnCommand documentation
113 /// @param lParam [in] See the CWnd::OnCommand documentation
114 ///
115 /// @return TRUE if the message has been processed.
116 ///
117 virtual BOOL OnCommand(WPARAM wParam, LPARAM lParam) override;
118
119 DECLARE_MESSAGE_MAP()
120};
