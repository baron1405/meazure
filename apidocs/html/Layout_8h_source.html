35 /// Aligns the specified windows such that their left edges all begin

76 /// Sets the position of the specified window.
77 ///
78 /// @param wnd [in] Window to position.
79 /// @param x [in] X position for the window, in pixels.
80 /// @param y [in] Y position for the window, in pixels.
81 ///
82 /// @return <b>true</b> if the method succeeds.
83 ///
84 inline bool SetWindowPos(const CWnd& wnd, int x, int y) {
85 return ::SetWindowPos(wnd.m_hWnd, nullptr, x, y, 0, 0, SWP_NOSIZE | SWP_NOZORDER) ? true : false;
86 }

88 /// Sets the size of the specified window.
89 ///
90 /// @param wnd [in] Window to size.
91 /// @param cx [in] Width for the window, in pixels.
92 /// @param cy [in] Height for the window, in pixels.
93 ///
94 /// @return <b>true</b> if the method succeeds.
95 ///
96 inline bool SetWindowSize(const CWnd& wnd, int cx, int cy) {

100 /// Returns the height of the specified window.
101 ///
102 /// @param wnd [in] Window whose height is desired.
103 ///
104 /// @return Height of the window in pixels.
105 ///
106 inline int GetWindowHeight(const CWnd& wnd) {
107 CRect rect;
108 wnd.GetWindowRect(rect);
109 return rect.Height();
110 }
111
112 /// Sets the height of the window.
113 ///
114 /// @param wnd [in] Window whose height is to be set.
115 /// @param cy [in] Height for the window, in pixels.
116 ///
117 /// @return <b>true</b> if the method succeeds.
118 ///
119 inline bool SetWindowHeight(const CWnd& wnd, int cy) {
120 CRect rect;
121 wnd.GetWindowRect(rect);
122 return ::SetWindowPos(wnd.m_hWnd, nullptr, 0, 0, rect.Width(), cy, SWP_NOMOVE | SWP_NOZORDER) ? true : false;
123 }
124
125 /// Moves the specified window vertically.
126 ///
127 /// @param wnd [in] Window to be moved.
128 /// @param cy [in] Vertical distance to move window, in pixels.
129 ///
130 /// @return <b>true</b> if the method succeeds.
131 ///
132 inline bool MoveWindowHeight(const CWnd& wnd, int cy) {
133 CRect rect;
134 wnd.GetWindowRect(rect);
135 CWnd* parentWnd = wnd.GetParent();
136 parentWnd->ScreenToClient(rect);
137 return ::SetWindowPos(wnd.m_hWnd, nullptr, rect.left, rect.top + cy, 0, 0,
138 SWP_NOSIZE | SWP_NOZORDER) ? true : false;
139 }

149 /// Converts the x position in screen coordinates to
150 /// an x position in the specified window's coordinate
151 /// system.
152 ///
153 /// @param wnd [in] The x coordinate is converts to this
154 /// window's coordinate system.
155 /// @param x [in, out] The x coordinate to convert.
156 ///
157 /// @return <b>true</b> if the conversion was successful.
158 ///
159 inline bool ScreenToClientX(const CWnd& wnd, int& x) {
160 CPoint pt(x, 0);
161 bool ret = ScreenToClient(wnd.m_hWnd, &pt) == TRUE;
162 x = pt.x;
163 return ret;
164 }
165
166 /// Converts the y position in screen coordinates to
167 /// a y position in the specified window's coordinate
168 /// system.
169 ///
170 /// @param wnd [in] The y coordinate is converts to this
171 /// window's coordinate system.
172 /// @param y [in, out] The y coordinate to convert.
173 ///
174 /// @return <b>true</b> if the conversion was successful.
175 ///
176 inline bool ScreenToClientY(const CWnd& wnd, int& y) {
177 CPoint pt(0, y);
178 bool ret = ScreenToClient(wnd.m_hWnd, &pt) == TRUE;
179 y = pt.y;
180 return ret;
181 }
