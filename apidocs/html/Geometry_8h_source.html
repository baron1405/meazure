30/// Represents a rectangular size. Unlike the Windows SIZE structure, the dimensions of the MeaFSize structure are
31/// double precision values.
32///
33struct MeaFSize {
34 double cx; ///< Length in the x dimension.
35 double cy; ///< Length in the y dimension.
36
37
38 /// Constructs a size object initialized to 0.0 width and height.
39 ///
40 constexpr MeaFSize() : MeaFSize(0.0, 0.0) {}
41
42 /// Constructs a size object initialized to the specified dimensions.
43 ///
44 /// @param x [in] Length in the x dimension
45 /// @param y [in] Length in the y dimension
46 ///
47 constexpr MeaFSize(double x, double y) : cx(x), cy(y) {}
48
49 /// Tests equality between this with the specified size.
50 ///
51 /// @param other [in] Other size to compare against this
52 /// @return <b>true</b> if this and the specified size have the same dimensions.
53 ///
54 bool operator==(const MeaFSize& other) const {
55 return MeaNumericUtils::IsEqualF(cx, other.cx) && MeaNumericUtils::IsEqualF(cy, other.cy);
56 }
57
58 /// Tests inequality between this with the specified size.
59 ///
60 /// @param other [in] Other size to compare against this
61 /// @return <b>true</b> if this and the specified size have the different dimensions.
62 ///
63 bool operator!=(const MeaFSize& other) const {
64 return !(*this == other);
65 }

67 /// Adds this and the specified object and returns a new size object.
68 /// \f[
69 /// \left[ \begin{array}{c}
70 /// f_x \\
71 /// f_y \\
72 /// \end{array} \right]
73 /// =
74 /// \left[ \begin{array}{c}
75 /// this_x \\
76 /// this_y \\
77 /// \end{array} \right]
78 /// +
79 /// \left[ \begin{array}{c}
80 /// fsize_x \\
81 /// fsize_y \\
82 /// \end{array} \right]
83 /// \f]
84 /// @param fsize [in] Size object to add.
85 /// @return New object that is the sum of this and the specified object.
86 ///
87 constexpr MeaFSize operator+(const MeaFSize& fsize) const {
88 return MeaFSize(cx + fsize.cx, cy + fsize.cy);

113 }

253 /// this_x \\

368 double bottom; ///< Bottom of the rectangle.
369 double left; ///< Left side of the rectangle.

371
372 /// Constructs a rectangle object initialized to 0.0 in all coordinates.
373 ///
374 constexpr MeaFRect() : MeaFRect(0.0, 0.0, 0.0, 0.0) {}
375
376 /// Constructs a rectangle object initialized to the specified coordinates.
377 ///
378 /// @param t [in] Initial top coordinate
379 /// @param b [in] Initial bottom coordinate
380 /// @param l [in] Initial left coordinate
381 /// @param r [in] Initial right coordinate
382 ///

394 MeaNumericUtils::IsEqualF(right, other.right);

512 /// Calculates the length corresponding to the specified x and y distances using the formula:
513 /// \f[
514 /// length=\sqrt{dx^2+dy^2}

517 /// @param dy [in] Y distance.

525 /// Calculates the length between the two specified points using the formula:

550 /// Returns the angular sector containing the vector specified by the two points. A circle centered at the
551 /// starting point of the vector is divided into eight 45 degree sectors numbered -4 through 4 inclusive.
552 /// The sectors are numbered according to the following table of angle ranges. Positive angles go clockwise
553 /// from the positive x-axis.
554 ///
555 /// <table>
556 /// <tr>
557 /// <th>Angle Range (deg)</th>
558 /// <th>Sector</th>
559 /// </tr>
560 /// <tr>
561 /// <td>[0, 45)</td>
562 /// <td>1</td>
563 /// </tr>
564 /// <tr>
565 /// <td>[45, 90)</td>
566 /// <td>2</td>
567 /// </tr>
568 /// <tr>
569 /// <td>[90, 135)</td>
570 /// <td>3</td>
571 /// </tr>
572 /// <tr>
573 /// <td>[135, 180)</td>
574 /// <td>4</td>
575 /// </tr>
576 /// <tr>
577 /// <td>[180, 225]</td>
578 /// <td>-4</td>
579 /// </tr>
580 /// <tr>
581 /// <td>(225, 270]</td>
582 /// <td>-3</td>
583 /// </tr>
584 /// <tr>
585 /// <td>(270, 315]</td>
586 /// <td>-2</td>
587 /// </tr>
588 /// <tr>
589 /// <td>(315, 0]</td>
590 /// <td>-1</td>
591 /// </tr>
592 /// <tr>
593 /// <td>Zero vector</td>
594 /// <td>0</td>
595 /// </tr>
596 /// </table>
597 ///
598 /// @param start [in] Starting point for the vector.
599 /// @param end [in] Ending point for the vector.
600 ///
601 /// @return The circular sector corresponding to the vector in the range [-4.0, 4.0].
602 ///
603 inline int CalcSector(const POINT& start, const POINT& end) {
604 int deltax = end.x - start.x;
605 int deltay = end.y - start.y;
606
607 if (deltax == 0 && deltay == 0) {
608 return 0;
609 }
610 if (deltax == 0) {
611 return (deltay < 0) ? -3 : 3;
612 }
613 if (deltay == 0) {
614 return (deltax < 0) ? -4 : 1;
615 }
616
617 // Note: 45 degrees == PI/4 radians
618 double result = atan2(static_cast<double>(deltay), static_cast<double>(deltax)) / MeaNumericUtils::PI4;
619 return static_cast<int>(result + std::copysign(1.0, result));
620 }

622 /// Indicates whether the specified vector is oriented in a vertical direction. A vector is considered
623 /// vertically oriented if it is in sector +/-2 or +/-3. See the documentation for the CalcSector method.
624 ///
625 /// @param start [in] Starting point for the vector
626 /// @param end [in] Ending point for the vector
627 /// @return <b>true</b> if the specified vector is vertically oriented. Returns <b>false</b> if the
628 /// vector is horizontally oriented or is the empty vector.
629 ///
630 inline bool IsVerticallyOriented(const POINT& start, const POINT& end) {
631 switch (CalcSector(start, end)) {
632 case 2:
633 case -2:
634 case 3:
635 case -3:
636 return true;
637 default:
638 return false;
639 }
640 }

642 /// Indicates whether the specified vector is oriented in a horizontal direction. A vector is considered
643 /// horizontally oriented if it is in sector +/-1 or +/-4. See the documentation for the CalcSector method.
644 ///
645 /// @param start [in] Starting point for the vector
646 /// @param end [in] Ending point for the vector
647 /// @return <b>true</b> if the specified vector is horizontally oriented. Returns <b>false</b> if the
648 /// vector is vertically oriented or is the empty vector.
649 ///
650 inline bool IsHorizontallyOriented(const POINT& start, const POINT& end) {
651 switch (CalcSector(start, end)) {
652 case 1:
653 case -1:
654 case 4:
655 case -4:
656 return true;
657 default:
658 return false;
659 }

662 /// Returns the angle of the vector from the specified start point to the specified end point relative to
663 /// the horizontal (x-axis), measured in radians. Using the default coordinate system (i.e. positive y-axis
664 /// pointed down from the origin), positive angles are measured clockwise from the horizontal to the vector.
665 /// The following formula is used:
666 /// \f[
667 /// angle=atan2(end_y-start_y, end_x-start_x)
668 /// \f]
669 /// @param start [in] Starting point for the vector whose angle is to be determined.
670 /// @param end [in] Ending point for the vector whose angle is to be determined.
671 ///
672 /// @return Angle between the x-axis and the line, in radians. Degenerate cases return 0.0.
673 ///
674 inline double CalcAngle(const MeaFPoint& start, const MeaFPoint& end) {
675 double deltax = end.x - start.x;
676 double deltay = end.y - start.y;
677
678 if (MeaNumericUtils::IsZeroF(deltax) && MeaNumericUtils::IsZeroF(deltay)) {
679 return 0.0;
680 }
681 return std::atan2(deltay, deltax);
682 }
683
684 /// Returns the angle, in radians, between the vector from the vertex to point p1 and the vector from the
685 /// vertex to point p2. Using the default coordinate system (i.e. positive y-axis pointed down from the origin),
686 /// positive angles are measured clockwise from the vertex-p1 line to the vertex-p2 line.
687 ///
688 /// @param vertex [in] Intersection point for the two lines.
689 /// @param p1 [in] Forms the end point for the first line from the vertex.
690 /// @param p2 [in] Forms the end point for the second line from the vertex.
691 ///
692 /// @return Angle between the vectors vertex-p1 and vertex-p2.
693 ///
694 inline double CalcAngle(const MeaFPoint& vertex, const MeaFPoint& p1, const MeaFPoint& p2) {
695 double deltax1 = p1.x - vertex.x;
696 double deltax2 = p2.x - vertex.x;
697 double deltay1 = p1.y - vertex.y;
698 double deltay2 = p2.y - vertex.y;
699
700 double numer = deltay2 * deltax1 - deltay1 * deltax2;
701 double denom = deltax2 * deltax1 + deltay1 * deltay2;
702
703 if (MeaNumericUtils::IsZeroF(numer) && MeaNumericUtils::IsZeroF(denom)) {
704 return 0.0;
705 }
706 return std::atan2(numer, denom);
707 }
