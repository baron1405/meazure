35/// Each data text field on the display is specified by one of
36/// these identifiers. Internally each identifier is used as a
37/// mask and as such, must be consecutive powers of 2.
38///
39enum MeaFields {
40 MeaX1Field = 0x0001, ///< X1 coordinate data field.
41 MeaY1Field = 0x0002, ///< Y1 coordinate data field.
42 MeaX2Field = 0x0004, ///< X2 coordinate data field.
43 MeaY2Field = 0x0008, ///< Y2 coordinate data field.
44 MeaXVField = 0x0010, ///< Vertex or center X coordinate data field.
45 MeaYVField = 0x0020, ///< Vertex or center Y coordinate data field.
46 MeaWidthField = 0x0040, ///< Width data field.
47 MeaHeightField = 0x0080, ///< Height data field.
48 MeaDistanceField = 0x0100, ///< Distance data field.
49 MeaAngleField = 0x0200, ///< Angle data field.
50 MeaAreaField = 0x0400, ///< Area data field.
51 MeaScreenWidthField = 0x0800, ///< Screen width data field.
52 MeaScreenHeightField = 0x1000, ///< Screen height data field.
53 MeaResXField = 0x2000, ///< Data field for the screen resolution in the X dimension.
54 MeaResYField = 0x4000 ///< Data field for the screen resolution in the Y dimension.
55};

57/// The data display is divided into two major sections: Region and Screen.
58/// The Region section displays the measurement data from the radio measurement
59/// tools (e.g. Point tool, Line Tool). The Screen section displays data about
60/// the current monitor (e.g. resolution, width, height). These identifiers are
61/// used to identify a data display section for enabling and disabling purposes.
62enum MeaSection {
63 MeaRegionSection, ///< Measurement tool data display section.
64 MeaScreenSection ///< Screen data display section.
65};

151 /// Controls the display of the resolution calibration button. Typically, the

154 /// measurements, a warning button is displayed on the Screen region of the
155 /// data display. Pressing the button displays the resolution calibration

330 /// Each data display field is actually composed of a label, a text field,
331 /// possibly a spin control, and usually a units label. This class packages

333 /// burden and reduces the potential for code duplication for the data
334 /// display class.
