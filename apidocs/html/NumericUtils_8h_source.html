35 /// Tests whether the two specified floating point values are equal.
36 ///
37 /// @tparam T One of <b>float</b>, <b>double</b>, or <b>long double</b>
38 /// @param f1 [in] Value to test
39 /// @param f2 [in] Value to test
40 ///
41 /// @return <b>true</b> if the two values are equal within an epsilon of the floating point type compensated
42 /// for the magnitude of the values. See
43 /// <a href="https://randomascii.wordpress.com/2012/02/25/comparing-floating-point-numbers-2012-edition/">Comparing
44 /// Floating Point Numbers, 2012 Edition</a> and
45 /// <a href="https://embeddeduse.com/2019/08/26/qt-compare-two-floats/">Comparing Two Floating-Point Numbers</a>
46 /// for details on this implementation.
47 ///
48 template<typename T>
49 inline bool IsEqualF(T f1, T f2) {
50 return (std::fabs(f1 - f2) <= std::numeric_limits<T>::epsilon())
51 ? true
52 : std::fabs(f1 - f2) <= (std::numeric_limits<T>::epsilon() * std::fmax(std::fabs(f1), std::fabs(f2)));
53 }
54
55 /// Tests whether the specified floating point value is zero.
56 ///
57 /// @tparam T One of <b>float</b>, <b>double</b>, or <b>long double</b>
58 /// @param f [in] Value to test
59 ///
60 /// @return <b>true</b> if the value is equal to zero within an epsilon of the floating point type.
61 ///
62 template<typename T>
63 inline bool IsZeroF(T f) {
64 return std::fabs(f) <= std::numeric_limits<T>::epsilon();
65 }
66};
