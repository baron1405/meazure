32/// Angle measurement tool. This tool measures the acute or obtuse
33/// angle between two intersecting lines. There are three crosshairs
34/// associated with the tool: one at the intersection of the two
35/// lines (vertex), and one at the end of each line (point 1 and point 2).
36/// Each crosshair can be arbitrarily positioned. An angle bisector
37/// is displayed to indicate on which side of the intersection the
38/// angle is being measured.
39///
40class MeaAngleTool : public MeaRadioTool, public MeaCrossHairCallback {

106 /// Visually strobes the tool. The tool crosshairs are cycled
107 /// between a highlight color and their normal color once. In addition,
108 /// the data windows attached to the crosshairs are cycled between

110 /// position recording.
111 ///

114 /// Sets the position of the tool. This method sets the position
115 /// one location component at a time. For example, the x position

117 /// with the new position and specifying the MeaX1Field for the
118 /// <i>which</i> parameter.
119 ///
120 /// @param which [in] Position component
121 /// @param pixels [in] Position to set for the component, in pixels
122 ///
123 virtual void SetPosition(MeaDataFieldId which, int pixels) override;

195 ///
196 virtual void OnCHLeave(const CrossHairInfo* info) override;
197
198 /// Called when the user attempts to drag one of the tool's
199 /// crosshairs. The crosshair to be moved is identified by the
200 /// MeaCrossHairCallback::CrossHairInfo structure passed to the method.
201 /// The info parameter indicates where the crosshair is to be
202 /// moved and the tool actually moves the crosshair to the new
203 /// position. This allows the tool to perform vertical or horizontal
204 /// locking and move the entire tool depending on which keyboard
205 /// modifiers are pressed during the drag.
206 ///
207 /// @param info [in] Crosshair information structure
208 ///
209 virtual void OnCHMove(const CrossHairInfo* info) override;
