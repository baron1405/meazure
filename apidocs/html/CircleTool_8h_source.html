32/// Circle measurement tool. This tool provides a circle and its radius.
33/// Two crosshairs are provided: one at the center of the circle (vertex)
34/// and one where the radius intersects the perimeter (point 1). Each
35/// crosshair can be arbitrarily positioned, which will result in moving
36/// and resizing the circle.
37///
38class MeaCircleTool : public MeaRadioTool, public MeaCrossHairCallback
39{
40public:
41 static const CString kToolName; ///< "CircleTool"
42
43 /// Constructs a new instance of a circle measurement tool. To use
44 /// a newly constructed Circle tool, the Enable() method must be called.
45 /// Enable() will call the Create() method the first time it is called
46 /// if Create() has not previously been called.
47 ///
48 /// @param mgr [in] Tool manager parent of the Circle tool.
49 ///
50 explicit MeaCircleTool(MeaToolMgr* mgr);
51
52 /// Hides and then destroys the Circle tool.
53 ///
54 virtual ~MeaCircleTool();

56
57 /// Displays the Circle tool. If the Create() method has not been
58 /// previously called, it will be called by Enable.
59 ///
60 virtual void Enable();
61
62 /// Hides the Circle tool.
63 ///
64 virtual void Disable();
65
66
67 /// Requests that the tool update the display of its position data.
68 ///
69 /// @param reason [in] Reason why the update has been requested
70 ///
71 virtual void Update(MeaUpdateReason reason);
72
73
74 /// Indicates whether the tool can define a rectangular region that
75 /// can then be read for a screen capture. For the Circle tool, the
76 /// region is the circle's bounding square.
77 ///
78 /// @return <b>true</b> if the tool defines a non-empty rectangular region.
79 ///
80 virtual bool HasRegion();
81
82 /// Returns the rectangle defined by the tool. This rectangle can then
83 /// be read for a screen capture.
84 ///
85 /// @return The circle's bounding square.
86 ///
87 virtual RECT GetRegion();
88
89
90 /// Persists the state of the tool to the specified profile object.
91 ///
92 /// @param profile [in] The destination for the state information.
93 /// Typically an MeaFileProfile or MeaRegistryProfile
94 /// object.
95 ///
96 virtual void SaveProfile(MeaProfile& profile);
97
98 /// Restores the state of the tool from the specified profile object.
99 ///
100 /// @param profile [in] The source for the state information. Typically
101 /// an MeaFileProfile or MeaRegistryProfile object.
102 ///
103 virtual void LoadProfile(MeaProfile& profile);
104
105
106 /// Displays the tool's crosshairs. If the Enable() method has not
107 /// been previously called, this method has no effect.
108 ///
109 virtual void EnableCrosshairs();
110
111 /// Hides the tool's crosshairs.
112 ///
113 virtual void DisableCrosshairs();
114
115
116 /// Visually flashes the tool. The tool crosshairs are cycled
117 /// between a hilite color and their normal color a number of times.
118 /// Flashing the tool helps the user find it on the screen.
119 ///
120 virtual void Flash();
121
122 /// Visually strobes the tool. The tool crosshairs are cycled
123 /// between a hilite color and their normal color once. In addition,
124 /// the data windows attached to the crosshairs are cycled between
125 /// hidden and visible. Strobing the tool is done to visually confirm
126 /// region capture and position recording.
127 ///
128 virtual void Strobe();
129
130
131 /// Sets the position of the tool. This method sets the position
132 /// one location component at a time. For example, the x position
133 /// of the center crosshair is specified by calling this method
134 /// with the new position and specifying the MeaXVField for the
135 /// <i>which</i> parameter.
136 ///
137 /// @param which [in] Position component
138 /// @param pixels [in] Position to set for the component, in pixels
139 ///
140 virtual void SetPosition(MeaFields which, int pixels);
141
142 /// Sets the position of the tool. This method is called by the
143 /// position log manager to play back a tool's position that was
144 /// previously recorded using the GetPosition method.
145 ///
146 /// @param points [in] Map of positions for the tool's crosshairs.
147 ///
148 virtual void SetPosition(const PointMap& points);
149
150 /// Returns the tool's current position. The current position is
151 /// the position of the last crosshair entered or moved. The default
152 /// position is the location of the center of the circle.
153 ///
154 /// @return Current position
155 ///
156 virtual const POINT& GetPosition() const;
157
158 /// Records the position of the tool. This method is called by the
159 /// position log manager to record a position.
160 ///
161 /// @param position [in] The tool's current position is recorded into
162 /// the position log manager's position object.
163 ///
164 virtual void GetPosition(MeaPositionLogMgr::Position& position) const;
165
166 /// Increments the specified position field.
167 ///
168 /// @param which Field to increment
169 ///
170 virtual void IncPosition(MeaFields which);
171
172 /// Decrements the specified position field.
173 ///
174 /// @param which Field to decrement
175 ///
176 virtual void DecPosition(MeaFields which);
177
178 /// Returns the name of the tool. Each tool has a unique name
179 /// which is used to identify the tool in profiles and position
180 /// logs.
181 ///
182 /// @return Name of the tool ("CircleTool").
183 ///
184 virtual CString GetToolName() const;
185
186 /// Returns a resource ID for the string used to label this
187 /// tool on the data display.
188 ///
189 /// @return Resource ID for the label.
190 ///
191 virtual UINT GetLabelId() const;
192
193 /// Called to notify the tool that the colors in the MeaColors class
194 /// have been changed. The crosshair and line colors are updated as
195 /// well as the crosshair and data window opacities.
196 ///
197 virtual void ColorsChanged();
198
199
200 /// Called when the pointer enters one of the tool's crosshairs.
201 /// The crosshair entered is identified by the MeaCrossHairCallback::CHInfo
202 /// structure passed to the method.
203 ///
204 /// @param info [in] Crosshair information structure
205 ///
206 virtual void OnCHEnter(const CHInfo *info);
207
208 /// Called when the pointer leaves one of the tool's crosshairs.
209 /// The crosshair exited is identified by the MeaCrossHairCallback::CHInfo
210 /// structure passed to the method.
211 ///
212 /// @param info [in] Crosshair information structure
213 ///
214 virtual void OnCHLeave(const CHInfo *info);
215
216 /// Called when the user attempts to drag one of the tool's
217 /// crosshairs. The crosshair to be moved is identified by the
218 /// MeaCrossHairCallback::CHInfo structure passed to the method.
219 /// The info parameter indicates where the crosshair is to be
220 /// moved and the tool actually moves the crosshair to the new
221 /// position. This allows the tool to perform vertical or horizontal
222 /// locking and move the entire tool depending on which keyboard
223 /// modifiers are pressed during the drag.
224 ///
225 /// @param info [in] Crosshair information structure
226 ///
227 virtual void OnCHMove(const CHInfo *info);
228
229private:
230 static const UINT kPerimeterId; ///< ID for the perimeter crosshair
231 static const UINT kCenterId; ///< ID for the center crosshair
232
233
234 /// Creates the tool's graphical components. The components include
235 /// the crosshairs, lines and data windows. The Enable() method must
236 /// be called to make the tool visible.
237 ///
238 /// @return <b>true</b> if the window is successfully created.
239 ///
240 bool Create();
241
242 /// Sets the position of the tool's crosshairs based on the current
243 /// values of #m_center, and #m_perimeter.
244 ///
245 void SetPosition();
246
247 CPoint m_center; ///< Location of the center of the circle
248 CPoint m_perimeter; ///< A point on the perimeter of the circle
249 CPoint *m_curPos; ///< Points to #m_center or #m_perimeter depending on which point the user is moving
250 CPoint m_anchorCenter; ///< Location for vertical / horizontal lock when Shift is held while dragging the center crosshair
251 CPoint m_anchorPerimeter; ///< Location for vertical / horizontal lock when Shift is held while dragging the perimeter crosshair
252 MeaCrossHair m_centerCH; ///< Crosshair for the center point

254 MeaCircle m_circle; ///< Circle graphic object
255 MeaLine m_line; ///< Radial line
256 MeaDataWin m_dataWinCenter; ///< Data window tooltip for the center point
257 MeaDataWin m_dataWinPerimeter; ///< Data window tooltip for the perimeter point
258};
