31/// Ruler drag callback mixin class. When the mouse drags the ruler, the
32/// OnRulerMove method is called. The inheriting class overrides the
33/// OnRulerMove method and actually moves the ruler to the new position.
34///
35class MeaRulerCallback
36{
37public:
38 /// Callback information structure. A callback information structure
39 /// is initialized and passed as the parameter in the callback method.
40 ///
41 struct RulerInfo {
42 MeaRuler *ruler; ///< Pointer to ruler object that initiated the callback.
43 LONG position; ///< Location of the ruler, in pixels. If the ruler
44 ///< orientation is vertical, the x position is reported.
45 ///< If the ruler is horizontal, the y position is
46 ///< reported.
47 UINT id; ///< User defined ID for the ruler.
48 UINT flags; ///< Virtual key flags including modifier keys (e.g. MK_CONTROL, MK_SHIFT).
49 };
50
51
52 /// Constructs a callback object.
53 ///
54 MeaRulerCallback() { }
55
56 /// Destroys a callback object.
57 ///
58 virtual ~MeaRulerCallback() { }
59
60 /// Called when the user presses the left mouse button and
61 /// then moves the mouse pointer while over the ruler
62 /// (i.e. a drag operation). It is the responsibility of
63 /// the callback method to actually moved the ruler to its
64 /// new position as specified in the callback information
65 /// structure.
66 ///
67 /// @param info [in] Ruler information structure
68 ///
69 virtual void OnRulerMove(const RulerInfo *info);
70};

73/// A ruler element. The ruler resembles a classic measurement
74/// straight edge. The ruler can be oriented vertically to
75/// measure along the height of the screen or horizontally to
76/// measure along the width of the screen. A ruler can be
77/// positioned anywhere along the height or width of the screen.
78/// Using the MeaRulerCallback mixin, a class can provide
79/// dragging of the ruler.
80///
81/// Up to three position indicators can be displayed on the face
82/// of the ruler. These indicators are used by the MeaRulerTool
83/// to show the current position of crosshairs and other
84/// measurement points.
85///
86/// The MeaRulerTool uses one vertical and one horizontal ruler
87/// per monitor.
88///
89class MeaRuler : public MeaGraphic
90{
91public:
92 /// A ruler can display up to three position indicators. An
93 /// indicator is a line that can be moved along the ruler to
94 /// show a position of interest. Each indicator is selected
95 /// by an identifier.
96 ///
97 enum IndicatorId {
98 Ind1 = 0, ///< Identifier for indicator 1
99 Ind2 = 1, ///< Identifier for indicator 2
100 Ind3 = 2 ///< Identifier for indicator 3
101 };
102
103 /// A ruler can be oriented vertically to measure along the
104 /// height of the screen, or horizontally to measure along the
105 /// width of the screen.
106 ///
107 enum Orientation {
108 Vertical, ///< Orient the ruler vertically.
109 Horizontal ///< Orient the ruler horizontally.
110 };
111
112 enum {
113 NumIndicators = 3 ///< Maximum number of indicators.
114 };

126 /// Creates the window for the ruler. This method must be called
127 /// before the ruler can be displayed and before any operation that

130 /// @param borderColor [in] The color for the ruler border and tick lines.

132 /// @param opacity [in] Opacity value for the ruler expressed as
133 /// a value between 0 (transparent) and 255 (opaque)
134 /// inclusive.
135 /// @param orientation [in] Orientation of the ruler: vertical or horizontal.
136 /// @param targetRect [in] Containment rectangle for the ruler. The ruler's
137 /// longest dimension will be sized to fit within this
138 /// rectangle.
139 /// @param callback [in] A class derived from MeaRulerCallback that
140 /// wants to receive notification of ruler movement.
141 /// Can be set to NULL if no callback is desired.
142 /// @param parent [in] Parent window or <b>NULL</b> if the ruler does
143 /// not have an immediate parent. If a parent window is
144 /// specified, the ruler is created as a child window
145 /// (<b>WS_CHILD</b>). Otherwise the ruler is created
146 /// as a popup window (<b>WS_POPUP</b>) and is assigned
147 /// the main window as a parent.
148 /// @param id [in] Caller-defined ID for the ruler.
149 ///
150 /// @return <b>true</b> if the window is created successfully.
151 ///
152 bool Create(COLORREF borderColor, COLORREF backColor,
153 BYTE opacity,

282 /// Specifies the location of the ruler number labels. For horizontally
283 /// oriented rulers, the labels may be position on the top or bottom
284 /// relative to the tick marks. For vertical rulers, the labels may be
285 /// positioned on the left or right relative to the tick marks. The
286 /// positioning of the labels is determined automatically based on the
287 /// position of the ruler on the screen. For example, a horizontal ruler
288 /// located in the upper half of the screen has its labels positioned on
289 /// the top, whereas the same ruler located in the lower half of the
290 /// screen has labels on the bottom.
291 ///
292 enum LabelPosition {
293 Top, ///< Position number labels above the ruler tick marks
294 Bottom, ///< Position number labels below the ruler tick marks
295 Left, ///< Position number labels to the left of the ruler tick marks
296 Right ///< Position number labels to the right of the ruler tick marks.
297 };
