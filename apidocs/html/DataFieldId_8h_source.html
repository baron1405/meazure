26/// Each data text field on the display is specified by one of these identifiers. Internally each identifier is used
27/// as a mask and as such, must be consecutive powers of 2.
28///
29enum MeaDataFieldId {
30 MeaX1Field = 0x0001, ///< X1 coordinate data field.
31 MeaY1Field = 0x0002, ///< Y1 coordinate data field.
32 MeaX2Field = 0x0004, ///< X2 coordinate data field.
33 MeaY2Field = 0x0008, ///< Y2 coordinate data field.
34 MeaXVField = 0x0010, ///< Vertex or center X coordinate data field.
35 MeaYVField = 0x0020, ///< Vertex or center Y coordinate data field.
36 MeaWidthField = 0x0040, ///< Width data field.
37 MeaHeightField = 0x0080, ///< Height data field.
38 MeaDistanceField = 0x0100, ///< Distance data field.
39 MeaAngleField = 0x0200, ///< Angle data field.
40 MeaAreaField = 0x0400, ///< Area data field.
41 MeaScreenWidthField = 0x0800, ///< Screen width data field.
42 MeaScreenHeightField = 0x1000, ///< Screen height data field.
43 MeaResXField = 0x2000, ///< Data field for the screen resolution in the X dimension.
44 MeaResYField = 0x4000, ///< Data field for the screen resolution in the Y dimension.
45 MeaAspectField = 0x8000 ///< Aspect ratio field.
46};
