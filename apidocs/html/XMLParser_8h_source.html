41/// The class contains the attributes associated with an XML start element.
42/// In addition to iterating through the attributes, the class provides
43/// searching and other attribute manipulation capabilities.
44///
45class MeaXMLAttributes
46{
47 friend class MeaXMLParser;
48
49protected:
50 /// Represents an attribute value.
51 ///
52 struct AttributeValue {
53 CString value; ///< The value of the attribute
54 bool isDefault; ///< <b>true</b> means that the value is set from the DTD default;
55 ///< <b>false</b> means that the value explicitly specified in the XML file.
56 };
57
58public:
59 /// Constructs an empty instance of the XML attributes class.
60 ///
61 MeaXMLAttributes();
62
63 /// Copy constructor. A deep copy of the specified attribute object
64 /// is performed.
65 ///
66 /// @param attrs [in] XML attribute object instance to copy.
67 ///
68 MeaXMLAttributes(const MeaXMLAttributes& attrs) { Assign(attrs); }
69
70 /// Destroys an instance of the XML attributes class.
71 ///
72 virtual ~MeaXMLAttributes();
73
74 /// Performs a deep assignment of the specified XML attribute object.
75 ///
76 /// @param attrs [in] XML attribute object to assign to this.
77 ///
78 /// @return this
79 ///
80 MeaXMLAttributes& operator=(const MeaXMLAttributes& attrs) { return Assign(attrs); }
81
82
83 /// Returns the value of the specified attribute as a string.
84 /// @param name [in] Attribute name.
85 /// @param value [out] Attribute value as a string.
86 /// @param isDefault [out] <b>true</b> if the value is the DTD default.
87 /// @return <b>true</b> if the attribute is found.
88 bool GetValueStr(LPCTSTR name, CString& value, bool& isDefault) const;
89
90 /// Returns the value of the specified attribute converted to an integer.
91 /// @param name [in] Attribute name.
92 /// @param value [out] Attribute value as an integer.
93 /// @param isDefault [out] <b>true</b> if the value is the DTD default.
94 /// @return <b>true</b> if the attribute is found.
95 bool GetValueInt(LPCTSTR name, int& value, bool& isDefault) const;
96
97 /// Returns the value of the specified attribute converted to a double.
98 /// @param name [in] Attribute name.
99 /// @param value [out] Attribute value as a double.
100 /// @param isDefault [out] <b>true</b> if the value is the DTD default.
101 /// @return <b>true</b> if the attribute is found.
102 bool GetValueDbl(LPCTSTR name, double& value, bool& isDefault) const;
103
104 /// Returns the value of the specified attribute converted to a boolean.
105 /// @param name [in] Attribute name.
106 /// @param value [out] Attribute value as a boolean. This parameter
107 /// is set to <b>true</b> if the attribute contains the
108 /// string "true" or "1". <b>false</b> is returned otherwise.
109 /// @param isDefault [out] <b>true</b> if the value is the DTD default.
110 /// @return <b>true</b> if the attribute is found.
111 bool GetValueBool(LPCTSTR name, bool& value, bool& isDefault) const;
112
113
114 /// Performs a deep copy assignment of the specified attribute object
115 /// to this.
116 ///
117 /// @param attrs [in] Attribute object to assign to this.
118 ///
119 /// @return this
120 ///
121 MeaXMLAttributes& Assign(const MeaXMLAttributes& attrs);
122
123protected:
124 /// Constructs an instance of the class based on the specified
125 /// attributes provided by the expat parser.
126 ///
127 /// @param atts [in] Array of attribute name/value pairs from expat.
128 /// In the array the name is followed by the value which
129 /// is then followed by another name and so on until a
130 /// NULL is encountered.
131 /// @param numSpecified [in] Number of attributes in the array that have been
132 /// explicitly specified versus set by default from the DTD.
133 /// The atts array is organized such that the explicitly
134 /// specified attributes come first.
135 ///
136 MeaXMLAttributes(const XML_Char **atts, int numSpecified);
137
138 std::map<CString, AttributeValue> m_attributeMap; ///< Maps attribute names to their values.
139};

271 MeaXMLNode* m_parent; ///< Parent of this node.

275/// Mix-in class for XML parser callback methods. A class that wants XML
276/// parsing services mix-in inherits from this class, overrides the methods
277/// of this class for the events of interest, creates an instance of the
278/// MeaXMLParser class and points it at the XML file to parse. The parser
279/// calls back using the methods of this class to indicate various XML parsing
280/// events. This is classic SAX parsing behavior. The class contains noop
281/// implementations for the method.
282///
283class MeaXMLParserHandler
284{
285public:
286 /// Empty constructor for the handler class. This class should be
287 /// thought of more as an interface than as an instantiable class.
288 ///
289 MeaXMLParserHandler() { }
290
291 /// Empty destructor for the handler class.
292 ///
293 virtual ~MeaXMLParserHandler() { }
294
295 /// Called when a new element is opened (e.g. &lt;foo&gt; or &lt;foo/&gt;).
296 ///
297 /// @param container [in] Parent element.
298 /// @param elementName [in] Name of the element being opened.
299 /// @param attrs [in] Attributes associated with the element.
300 ///
301 virtual void StartElementHandler(const CString& container,
302 const CString& elementName,
303 const MeaXMLAttributes& attrs);
304
305 /// Called when an element is closed (e.g. &lt;/foo&gt; or &lt;foo/&gt;).
306 ///
307 /// @param container [in] Parent element.
308 /// @param elementName [in] Name of the element being closed.
309 ///
310 virtual void EndElementHandler(const CString& container,
311 const CString& elementName);
312
313 /// Called for all character data encountered between elements.
314 /// Note that this method will be called for all character data
315 /// including whitespace between consecutive elements. The DTD must
316 /// must be used to determine what whitespace is relevant.
317 ///
318 /// @param container [in] Name of the closest open element containing
319 /// the character data.
320 /// @param data [in] Character data.
321 ///
322 virtual void CharacterDataHandler(const CString& container,
323 const CString& data);
324
325 /// Called to parse an external entity.
326 ///
327 /// @param parser [in] Current XML parser.
328 /// @param pathname [in] Pathname of the external entity.
329 ///
330 virtual void ParseEntity(MeaXMLParser& parser, const CString& pathname);
331
332 /// Returns the pathname of the currently parsed file.
333 ///
334 /// @return Pathname of the file currently being parsed.
335 /// This class's implementation of this method returns
336 /// the empty string.
337 ///
338 virtual CString GetFilePathname();
339};

342/// Provides SAX style XML parsing. This class is a wrapper around the
343/// expat parser and adds XML validation. A class that wants XML parsing
344/// services mix-in inherits from the MeaXMLParserHandler class, overrides
345/// the methods of that class for the events of interest, creates an instance
346/// of this class and points it at the XML file to parse. The parser calls
347/// back using the methods of the MeaXMLParserHandler class to indicate various
348/// XML parsing events. This is classic SAX parsing behavior. The class can
349/// also build a primitive DOM.
350///
351class MeaXMLParser: public ev::IValidationHandler
352{
353public:
354 /// Constructs an XML parser.
355 ///
356 /// @param handler [in] Callback object for parsing events.
357 /// @param buildDOM [in] Indicates whether a DOM should be built.
358 ///
359 explicit MeaXMLParser(MeaXMLParserHandler *handler, bool buildDOM = false);
360
361 /// Creates an external entity parser. When an external is encountered,
362 /// the MeaXMLParserHandler::ParseEntity method is called. This method
363 /// should instantiate an XML sub-parser using this constructor and passing
364 /// the parent parser provided by the callback method.
365 ///
366 /// @param parentParser [in] Parent parser for the external entity
367 /// sub-parser.
368 ///
369 MeaXMLParser(const MeaXMLParser& parentParser);
370
371 /// Destroys either a primary or a sub-parser.
372 ///
373 virtual ~MeaXMLParser();
374
375
376 /// Sets the base path for resolving relative external entities.
377 ///
378 /// @param path [in] Base path.
379 ///

381
382
383 /// Obtains a buffer to load with XML data. This method must
384 /// be called prior to each call to ParseBuffer.
385 ///
386 /// @param len [in] Size of the buffer, in bytes.
387 ///
388 /// @return Pointer to the allocated buffer. The class instance
389 /// also holds a pointer to this buffer and owns the memory
390 /// for it. Do not call free or delete on this pointer. A
391 /// subsequent call to the ParseBuffer method will parse the

405
406 /// If a DOM was constructed, this method returns its root node.
407 ///
408 /// @return Root node of the DOM or NULL if none was constructed.

412
