31/// Base class for measurement units labels. The label is
32/// shown on the data display. There is a derived class for
33/// each type of measurement: angles, lengths, areas and
34/// resolutions.
35///
36class MeaUnitsLabel : public MeaLabel {
37
38public:
39 typedef std::map<int, CString> LabelMap; ///< Maps a units identifier to the corresponding label string.
40
41 /// Destroys the base class instance for the label.
42 ///
43 virtual ~MeaUnitsLabel();
44
45 /// Creates the underlying label control.
46 ///
47 /// @param unitsId [in] Units identifier.
48 /// @param style [in] Window style for the label as
49 /// defined by CStatic.
50 /// @param topLeft [in] Location for the top left corner
51 /// of the label, in pixels.
52 /// @param parentWnd [in] Parent window for the label.
53 /// @param id [in] Caller defined identifier.
54 ///
55 /// @return <b>true</b> if the label was created successfully.
56 ///
57 bool Create(UINT unitsId, DWORD style, const POINT& topLeft,
58 CWnd* parentWnd, UINT id = 0xffff);
59
60protected:
61 /// Base class constructor for the label instance.
62 ///
63 /// @param format [in] Format string that defines the
64 /// type of measurement units (e.g. "sq.").
65 /// @param labels [in] Map of measurement identifiers to
66 /// label strings.
67 /// @param width [out] Width of the labels, in pixels.
68 ///
69 MeaUnitsLabel(UINT format, const LabelMap& labels, int& width);
70
71 /// Returns the measurement type format string.
72 ///
73 /// @return Format string identifier.
74 ///
75 UINT GetFormat() const { return m_format; }
76
77 /// Sets the overall width of the label object to the width
78 /// of the widest label string over all the units.
79 ///
80 /// @return <b>true</b> if the size was set successfully.
81 ///
82 bool SetSize();
83
84 /// Formats the label string corresponding to the specified
85 /// units identifier.
86 ///
87 /// @param unitsId [in] Units identifier whose label is
88 /// to be constructed.
89 ///
90 /// @return Formatted label string (e.g. "sq. in").
91 ///
92 CString ConstructLabel(int unitsId) const;
93
94private:
95 const LabelMap& m_labelsRef; ///< Maps units identifiers to label strings.
96 int& m_widthRef; ///< Width of the label object, in pixels.
97 UINT m_format; ///< Label format string identifier.
98};

128/// There are two categories of labels: those for angular
129/// measurements and those for linear measurements. This
130/// class is the base class for linear measurement labels.
131///
132class MeaLinearUnitsLabel : public MeaUnitsLabel {
133
134public:
135 /// Displays the actual text of the label.
136 ///
137 /// @param unitsId [in] Linear units identifier.
138 ///
139 virtual void SetUnits(MeaLinearUnitsId unitsId);
140
141 /// Sets the specified label string for the specified units.
142 ///
143 /// @param unitsId [in] Identifier for the units whose label
144 /// string is to be changed.
145 /// @param label [in] New label string for the units.
146 ///
147 static void ChangeLabel(MeaLinearUnitsId unitsId, const CString& label) {
148 m_labels[unitsId] = label;
149 }
150
151protected:
152 /// Constructs a linear units label.
153 ///
154 /// @param format [in] Format string identifier. The format
155 /// string provides label modifiers for the
156 /// measurement type (e.g. "sq." for "sq. in").
157 /// @param width [out] Width of the label, in pixels.
158 ///
159 MeaLinearUnitsLabel(UINT format, int& width);
160
161 /// Destroys a linear units label.
162 ///
163 virtual ~MeaLinearUnitsLabel();
164
165private:
166 static LabelMap m_labels; ///< Maps linear units identifiers to their corresponding label strings.
167};
