29/// Property page for custom measurement units preferences. Using this
30/// property page, the user can define their own measurement units
31/// (i.e. label and conversion factor).
32///
33class MeaCustomUnitsPrefs : public CPropertyPage {
34 DECLARE_DYNCREATE(MeaCustomUnitsPrefs)
35
36public:
37 enum { IDD = IDD_PREF_CUSTOM };
38
39 /// Constructs an instance of the custom measurement units page.
40 ///
41 MeaCustomUnitsPrefs();
42
43 /// Destroys an instance of the custom measurement units page.
44 ///
45 ~MeaCustomUnitsPrefs();
46
47 /// Obtains the units basis and scale factor for the custom units.
48 ///
49 /// @param scaleBasis [out] Indicates whether the custom units are based
50 /// on pixels, inches or centimeters.
51 /// @param scaleFactor [out] Conversion factor to/from scaleBasis to/from the
52 /// custom units.
53 ///
54 void GetScaleFactor(MeaCustomUnits::ScaleBasis& scaleBasis, double& scaleFactor) const;
55
56 /// Sets the custom units basis and scale factor.
57 ///
58 /// @param scaleBasis [in] Specifies whether the custom units are based on
59 /// pixels, inches or centimeters.
60 /// @param scaleFactor [in] Conversion factor to/from scaleBasis to/from the
61 /// custom units.
62 ///
63 void SetScaleFactor(MeaCustomUnits::ScaleBasis scaleBasis, double scaleFactor);
64
65 /// Called when the property page is created. This method
66 /// subclasses a number of controls on the page, with our
67 /// implementations and obtains pointers to the page controls.
68 ///
69 /// @return TRUE, indicating that the input focus should be
70 /// set to the first control on the page.
71 ///
72 virtual BOOL OnInitDialog() override;
73
74 /// Called when the page becomes inactive. The data on the page
75 /// is validated.
76 ///
77 /// @return TRUE if the data was valid and the page can become
78 /// inactive. FALSE if the data is not valid and the page
79 /// must remain active.
80 ///
81 virtual BOOL OnKillActive() override;
82
83 /// Called when the page becomes active. This method ensures that
84 /// the scale factor basis labels reflect the custom units abbreviation,
85 /// if any.
86 ///
87 /// @return TRUE, indicating that the page is active.
88 ///
89 virtual BOOL OnSetActive() override;
90
91 /// Called when the Apply button is pressed and validates the page
92 /// contents. The page contents are valid if the menu name label and
93 /// abbreviation are either both empty or both non-empty and the scale
94 /// factor is greater than zero.
95 ///
96 /// @return TRUE if the page contents are valid and can be applied. If
97 /// the page contents are not valid, FALSE is returned.
98 ///
99 virtual BOOL OnApply() override;
100
101
102 CString m_abbrev; ///< Abbreviation for the custom units.
103 CString m_name; ///< Full name for the custom units.

105 double m_pxScaleFactor; ///< Scale factor if the custom units are based on pixels.
106 double m_inScaleFactor; ///< Scale factor if the custom units are based on inches.

115 virtual void DoDataExchange(CDataExchange* pDX) override;

121 afx_msg void OnClear();

123 /// Called when a value changes on the UI. This method sets the modified

126 afx_msg void OnChange();

140 /// Called when the units menu name text field value is changed.
141 ///
142 afx_msg void OnNameChange();

144 /// Called when the units abbreviation text field value is changed.

146 afx_msg void OnAbbrevChange();
147
148 /// Called when the Set Display Precision button is pressed. Pressing this

151 ///
152 afx_msg void OnPrecision();
153
154 /// Called when the centimeter-based scale factor radio button is selected.
155 ///
156 afx_msg void OnCentimeterBasis();
157
158 /// Called when the inch-based scale factor radio button is selected.
159 ///
160 afx_msg void OnInchBasis();
161
162 /// Called when the pixel-based scale factor radio button is selected.
163 ///
164 afx_msg void OnPixelBasis();
165
166 DECLARE_MESSAGE_MAP()
167
168
169private:
170 /// Ensures radio button behavior on the scale factor basis radio buttons.
171 /// The specified button is set and all other basis buttons are unset.
172 ///
173 /// @param buttonId [in] ID for the radio button to set.
174 ///
175 void SetCheck(UINT buttonId);
176
177 /// Validates the custom units data on the page. The page data is considered
178 /// valid either if both the name and abbrev. fields are empty, or if they
179 /// both contain text and the scale factor is greater than 0.
180 ///
181 /// @return <b>true</b> if the page data is valid, and <b>false</b> if the
182 /// data is invalid.
183 ///
184 bool Validate();
185
186 /// Enables or disables the scale factor and precision controls depending on
187 /// whether the units name and abbrev. fields contain text or are empty.
188 ///
189 void EnableScaleFactor();

193 CWnd* m_abbrevField; ///< Units abbreviation text field.
194 CWnd* m_precisionButton; ///< Units display precision button.

206 CWnd* m_inAbbrev; ///< Inch-based scale factor abbreviation label.
207 CWnd* m_cmAbbrev; ///< Centimeter-based scale factor abbreviation label.
208};
