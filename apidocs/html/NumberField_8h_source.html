28/// Provides a text field control that allows only numeric entries.
29/// The text field can be configured to allow any floating point value,
30/// only integer values, or only positive integer values.
31///
32class MeaNumberField : public CEdit {
33
34public:
35 /// Configurations for the text field.
36 ///
37 enum ValueType {
38 AllValues = 0, ///< Allows entry of any floating point number.
39 IntValues = 1, ///< Allows entry of positive or negative integers.
40 PosValues = 2 ///< Allows entry of positive integers.
41 };
42
43
44 /// Constructs a field object. Before the field can be used, the
45 /// Create method must be called.
46 ///
47 MeaNumberField();
48
49 /// Destroys a field object.
50 ///
51 virtual ~MeaNumberField();
52
53 /// Creates the window for the field. To make the field visible,
54 /// add the WS_VISIBLE style or call ShowWindow.
55 ///
56 /// @param style [in] Field styles as defined for CEdit.
57 /// @param topLeft [in] Position of the top, left corner for the field, in pixels.
58 /// @param numChars [in] Number of visible characters. For proportional fonts
59 /// the number of characters actually visible may vary from the
60 /// specified value.
61 /// @param parentWnd [in] Parent window for the field.
62 /// @param id [in] ID for the field.
63 ///
64 /// @return <b>true</b> if the field was created successfully.
65 ///
66 bool Create(DWORD style, const POINT& topLeft, int numChars, CWnd* parentWnd, UINT id = 0xFFFF);
67
68 /// Configures the field with the type of value to accept.
69 ///
70 /// @param valueType [in] One of the type defined by MeaNumberField::ValueType.
71 ///
72 void SetValueType(DWORD valueType) { m_valueType = valueType; }
73
74 /// Displays an integer value in the field.
75 ///
76 /// @param value [in] Integer value to display in the field.
77 ///
78 void SetValue(int value) {
79 CString str;
80 str.Format(_T("%d"), value);
81 SetWindowText(str);
82 }
83
84 /// Displays a double precision value in the field.
85 ///
86 /// @param value [in] Double value to display in the field.
87 ///
88 void SetValue(double value) {
89 CString str;
90 str.Format(_T("%f"), value);
91 SetWindowText(str);
92 }
93
94 /// Obtains the value contained in the text field as an integer.
95 ///
96 /// @param value [out] Value contained in the field converted to an integer.
97 ///
98 /// @return <b>true</b> if the field is not empty.
99 ///
100 bool GetValue(int& value) const {
101 CString str;
102 GetWindowText(str);
103 if (str.IsEmpty()) {
104 return false;
105 }
106 value = _ttoi(str);
107 return true;
108 }
109
110 /// Obtains the value contained in the text field as a double precision value.
111 ///
112 /// @param value [out] Value contained in the field converted to a double.
113 ///
114 /// @return <b>true</b> if the field is not empty.
115 ///
116 bool GetValue(double& value) const {
117 CString str;
118 GetWindowText(str);
119 if (str.IsEmpty()) {
120 return false;
121 }
122 value = _tcstod(str, nullptr);
123 return true;
124 }
125
126 /// Indicates if the value in the field is selected.
127 ///
128 /// @return <b>true</b> if the value in the field is selected.
129 ///
130 bool IsTextSelected() const {
131 int start, end;
132 GetSel(start, end);
133 return (start != end);
134 }
135
136 /// Handles the RETURN key, UP and DOWN arrow keys.
137 ///
138 /// @param msg [in] Message structure.
139 ///
140 /// @return TRUE if UP/DOWN and FALSE otherwise.
141 ///
142 virtual BOOL PreTranslateMessage(MSG* msg) override;
143
144protected:
145 DECLARE_MESSAGE_MAP()
146
147 /// Filters out characters not used in a number.
148 ///
149 /// @param nChar [in] Character code.
150 /// @param nRepCnt [in] Character repeat count.
151 /// @param nFlags [in] Low level key codes.
152 ///
153 afx_msg void OnChar(UINT nChar, UINT nRepCnt, UINT nFlags);
154
155 /// Ensures only a valid number is pasted.
156 ///
157 /// @return Always returns TRUE.
158 ///
159 afx_msg LRESULT OnPaste(WPARAM, LPARAM);
160
161 /// Called when the field loses keyboard focus. The parent
162 /// window is sent a MeaFieldFocusMsg message so that it can
163 /// take actions such as reading the value in the field.
164 ///
165 /// @param win [in] Next window to receive focus.
166 ///
167 afx_msg void OnKillFocus(CWnd* win);
168
169 /// Indicates if the specified character is an acceptable sign character.
170 /// Acceptable sign characters are '+' and '-' for a float and integer
171 /// configured field, and only '+' for a positive integer configured field.
172 ///
173 /// @param ch [in] Character to test.
174 ///
175 /// @return <b>true</b> if the specified character represents an acceptable
176 /// sign character.
177 ///
178 bool IsSign(TCHAR ch) const { return ((ch == _T('+')) || ((ch == _T('-') && !PosOnly()))); }
179
180 /// Indicates if the specified character is an acceptable decimal point.
181 /// An acceptable decimal point character is a '.' when the field is
182 /// configured to accept floating point values.
183 ///
184 /// @param ch [in] Character to test.
185 ///
186 /// @return <b>true</b> if the specified character represents an acceptable
187 /// decimal point character.
188 ///
189 bool IsPoint(TCHAR ch) const { return ((ch == _T('.')) && !IntOnly()); }
190
191private:
192 static int m_edgeHeight; ///< Height of the field's 3D sunken edge
193
194
195 /// Tests whether the specified string can be placed in the text field.
196 ///
197 /// @param str [in] String to test.
198 /// @param len [in] Length of string.
199 ///
200 /// @return <b>true</b> if the specified string can be placed in the field.
201 ///
202 bool IsValid(LPCTSTR str, int len) const;
203
204 /// Indicates whether the field has been configured to only accept integer values.
205 ///
206 /// @return <b>true</b> if the field has been configured to only accept integers.
207 ///
208 bool IntOnly() const { return (m_valueType & IntValues) != 0; }
209
210 /// Indicates whether the field has been configured to only accept positive integer values.
211 ///
212 /// @return <b>true</b> if the field has been configured to only accept positive integers.
213 ///
214 bool PosOnly() const { return (m_valueType & PosValues) != 0; }
215
216
217 DWORD m_valueType; ///< Configuration for the field as defined by MeaNumberField::ValueType.
218};
