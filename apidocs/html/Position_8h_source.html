30/// Represents a tool position. A position object references the desktop information that provides its environment.
31/// Since desktop environments do not tend to change often, many position objects typically reference the same
32/// desktop information object. The position object is ultimately persisted to the position log file along with
33/// the desktop information object.
34///
35/// A position consists of one or more (x,y) points. The number of points required for a position depends on the
36/// measurement tool whose position is being recorded. For example, the Point tool requires only one point to
37/// completely describe its position, whereas the Line tool requires two points, one per endpoint of the line.
38/// Each point is named by the tool (e.g. "v", "1", "2").
39///
40class MeaPosition {

236 /// Compares the specified position information object with this to determine equality.
237 ///
238 /// @param position [in] Position information object to compare with this.
239 ///
240 /// @return <b>true</b> if the specified object and this are equal.
241 ///
242 bool operator==(const MeaPosition& position) const {
243 return (m_points == position.m_points) &&
244 MeaNumericUtils::IsEqualF(m_width, position.m_width) &&
245 MeaNumericUtils::IsEqualF(m_height, position.m_height) &&
246 MeaNumericUtils::IsEqualF(m_distance, position.m_distance) &&
247 MeaNumericUtils::IsEqualF(m_area, position.m_area) &&
248 MeaNumericUtils::IsEqualF(m_angle, position.m_angle) &&
249 (m_desktopRef == position.m_desktopRef) &&
250 (m_toolName == position.m_toolName) &&
251 (m_timestamp == position.m_timestamp) &&
252 (m_desc == position.m_desc);
253 }

261 bool operator!=(const MeaPosition& position) const { return !(*this == position); }
