92/// Measurement units are packaged into a class, one for each set
93/// of units (e.g. inches). Each class contains methods for conversion,
94/// display precision, and persistence of the state of the class to a
95/// given profile object. Related to these classes but not contained by
96/// them are the set of unit label classes.
97///
98/// This class serves as the base class for all measurement units.
99///
100class MeaUnits
101{
102public:
103 static const double kPI; ///< The value of pi.
104
105 typedef std::vector<int> DisplayPrecisions; ///< Number of decimal places to display for each type of measurement.
106 typedef std::vector<CString> DisplayPrecisionNames; ///< Strings to identify units precisions in profiles.
107
108
109 /// Persists the state of a units class instance to the specified
110 /// profile object.
111 ///
112 /// @param profile [in] Profile object into which to persist
113 /// the state of the units class.
114 ///
115 virtual void SaveProfile(MeaProfile& profile);
116
117 /// Restores the state of a units class instance from the specified
118 /// profile object.
119 ///
120 /// @param profile [in] Profile object from which to restore
121 /// the state of the units class.
122 ///
123 virtual void LoadProfile(MeaProfile& profile);
124
125
126 /// Persists the measurement precisions to the specified profile object.
127 ///
128 /// @param profile [in] Profile object into which to persist
129 /// the decimal place precisions.
130 ///
131 void SavePrecision(MeaProfile& profile);
132
133 /// Restores the measurement precisions from the specified profile object.
134 ///
135 /// @param profile [in] Profile object from which to restore
136 /// the decimal place precisions.
137 ///
138 void LoadPrecision(MeaProfile& profile);
139
140
141 /// Called to restores the units to their default state.
142 ///
143 virtual void MasterReset();
144
145
146 /// Returns a string describing the units.
147 ///
148 /// @return String describing he units (e.g. "px')
149 ///
150 CString GetUnitsStr() const { return m_unitsStr; }
151
152 /// Sets the decimal place precisions for all measurement types.
153 ///
154 /// @param displayPrecisions [in] Set of precisions for all measurement types.
155 ///
156 void SetDisplayPrecisions(const DisplayPrecisions& displayPrecisions) {
157 m_displayPrecisions = displayPrecisions;
158 }
159
160 /// Returns the current decimal place precisions for all measurement types.
161 ///
162 /// @return Set of precisions for all measurement types.
163 ///
164 const DisplayPrecisions& GetDisplayPrecisions() const {
165 return m_displayPrecisions;
166 }
167
168 /// Returns the default decimal place precisions for all measurement types.
169 ///
170 /// @return Set of default precisions for all measurement types.
171 ///
172 const DisplayPrecisions& GetDefaultPrecisions() const {
173 return m_defaultPrecisions;
174 }
175
176 /// Sets the default display precisions as the current precisions.
177 ///
178 void RestoreDefaultPrecisions() {
179 m_displayPrecisions = m_defaultPrecisions;
180 }
181
182 /// Returns the identifying names for the display precisions.
183 ///
184 /// @return Set of names identifying precisions for all measurement types.
185 ///
186 const DisplayPrecisionNames& GetDisplayPrecisionNames() const {
187 return m_displayPrecisionNames;
188 }
189
190protected:
191 /// Base class constructor.
192 ///
193 /// @param unitsStr [in] Name for the units.
194 ///
195 explicit MeaUnits(LPCTSTR unitsStr);
196
197 /// Base class destructor.
198 ///
199 virtual ~MeaUnits();
200
201 /// Adds the specified identifying name to the list of
202 /// precision names.
203 ///
204 /// @param name [in] Identifying name for the precision.
205 ///
206 void AddPrecisionName(LPCTSTR name) {
207 m_displayPrecisionNames.push_back(name);
208 }
209
210 /// Adds the specified precision value to both the list
211 /// of current precisions and the list of default precisions.
212 ///
213 /// @param precision [in] Number of decimal places to display.
214 ///
215 void AddPrecision(int precision) {
216 m_defaultPrecisions.push_back(precision);
217 m_displayPrecisions.push_back(precision);
218 }
219
220private:
221 DisplayPrecisions m_defaultPrecisions; ///< Default precisions for all measurement types.
222 DisplayPrecisions m_displayPrecisions; ///< Current precisions for all measurement types.
223 DisplayPrecisionNames m_displayPrecisionNames; ///< Names for all precision values.
224 CString m_unitsStr; ///< Name for the units.
225};
