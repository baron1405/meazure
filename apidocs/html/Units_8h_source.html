85/// Measurement units are packaged into a class, one for each set
86/// of units (e.g. inches). Each class contains methods for conversion,
87/// display precision, and persistence of the state of the class to a
88/// given profile object. Related to these classes but not contained by
89/// them are the set of unit label classes.
90///
91/// This class serves as the base class for all measurement units.
92///
93class MeaUnits {
94
95public:
96 typedef std::vector<int> DisplayPrecisions; ///< Number of decimal places to display for each type of measurement.
97 typedef std::vector<CString> DisplayPrecisionNames; ///< Strings to identify units precisions in profiles.
98
99
100 /// Base class destructor.
101 ///
102 virtual ~MeaUnits();
103
104 /// Persists the state of a units class instance to the specified
105 /// profile object.
106 ///
107 /// @param profile [in] Profile object into which to persist
108 /// the state of the units class.
109 ///
110 virtual void SaveProfile(MeaProfile& profile);
111
112 /// Restores the state of a units class instance from the specified
113 /// profile object.
114 ///
115 /// @param profile [in] Profile object from which to restore
116 /// the state of the units class.
117 ///
118 virtual void LoadProfile(MeaProfile& profile);
119
120 /// Persists the measurement precisions to the specified profile object.
121 ///
122 /// @param profile [in] Profile object into which to persist
123 /// the decimal place precisions.
124 ///
125 void SavePrecision(MeaProfile& profile);
126
127 /// Restores the measurement precisions from the specified profile object.
128 ///
129 /// @param profile [in] Profile object from which to restore
130 /// the decimal place precisions.
131 ///
132 void LoadPrecision(MeaProfile& profile);
133
134 /// Called to restores the units to their default state.
135 ///
136 virtual void MasterReset();
137
138 /// Returns a string describing the units.
139 ///
140 /// @return String describing he units (e.g. "px')
141 ///
142 CString GetUnitsStr() const { return m_unitsStr; }
143
144 /// Sets the decimal place precisions for all measurement types.
145 ///
146 /// @param displayPrecisions [in] Set of precisions for all measurement types.
147 ///
148 void SetDisplayPrecisions(const DisplayPrecisions& displayPrecisions) {
149 m_displayPrecisions = displayPrecisions;
150 }
151
152 /// Returns the current decimal place precisions for all measurement types.
153 ///
154 /// @return Set of precisions for all measurement types.
155 ///
156 const DisplayPrecisions& GetDisplayPrecisions() const {
157 return m_displayPrecisions;
158 }
159
160 /// Returns the default decimal place precisions for all measurement types.
161 ///
162 /// @return Set of default precisions for all measurement types.
163 ///
164 const DisplayPrecisions& GetDefaultPrecisions() const {
165 return m_defaultPrecisions;
166 }
167
168 /// Sets the default display precisions as the current precisions.
169 ///
170 void RestoreDefaultPrecisions() {
171 m_displayPrecisions = m_defaultPrecisions;
172 }
173
174 /// Returns the identifying names for the display precisions.
175 ///
176 /// @return Set of names identifying precisions for all measurement types.
177 ///
178 const DisplayPrecisionNames& GetDisplayPrecisionNames() const {
179 return m_displayPrecisionNames;
180 }
181
182protected:
183 /// Base class constructor.
184 ///
185 /// @param unitsStr [in] Name for the units.
186 ///
187 explicit MeaUnits(PCTSTR unitsStr);
188
189 /// Adds the specified identifying name to the list of
190 /// precision names.
191 ///
192 /// @param name [in] Identifying name for the precision.
193 ///
194 void AddPrecisionName(PCTSTR name) {
195 m_displayPrecisionNames.push_back(name);
196 }
197
198 /// Adds the specified precision value to both the list
199 /// of current precisions and the list of default precisions.
200 ///
201 /// @param precision [in] Number of decimal places to display.
202 ///
203 void AddPrecision(int precision) {
204 m_defaultPrecisions.push_back(precision);
205 m_displayPrecisions.push_back(precision);
206 }
207
208private:
209 DisplayPrecisions m_defaultPrecisions; ///< Default precisions for all measurement types.
210 DisplayPrecisions m_displayPrecisions; ///< Current precisions for all measurement types.
211 DisplayPrecisionNames m_displayPrecisionNames; ///< Names for all precision values.
212 CString m_unitsStr; ///< Name for the units.
213};

216/// There are two major categories of units: linear and angular.
217/// The angular units cover the angle measurement units. This is
218/// the base class for all angular units.
219///
220class MeaAngularUnits : public MeaUnits {
221
222public:
223 /// Destroys an angular-based unit.
224 ///
225 virtual ~MeaAngularUnits();
226
227 /// Returns the identifier for the angular units.
228 ///
229 /// @return Angular units ID.
230 ///
231 MeaAngularUnitsId GetUnitsId() const { return m_unitsId; }
232
233 /// Sets whether to show the included angle (default) or the supplemental angle. The supplemental angle is
234 /// defined as:
235 /// \f[
236 /// \theta_{s} =
237 /// \begin{cases}
238 /// 180^\circ - \theta_{i} & \theta_{i} \ge 0\\
239 /// -180^\circ - \theta_{i} & \theta_{i} < 0
240 /// \end{cases}
241 /// \f]
242 ///
243 /// @param showSupplemental [in] true to show supplemental angle instead of included angle
244 ///
245 static void SetSupplementalAngle(boolean showSupplemental) { m_supplementalAngle = showSupplemental; }
246
247 /// Indicates whether the included angle is shown (default) or the supplemental angle.
248 ///
249 /// @return true if the supplemental angle is shown.
250 ///
251 static bool IsSupplementalAngle() { return m_supplementalAngle; }
252
253 /// Formats the specified angular measurement value using the
254 /// precision for the specified measurement ID.
255 ///
256 /// @param id [in] Identifier for the angular units whose precision
257 /// is to be used to format the specified value.
258 /// @param value [in] Measurement value to be formatted.
259 ///
260 /// @return Measurement value formatted with the appropriate precision
261 /// and returned as a string.
262 ///
263 CString Format(MeaAngularMeasurementId id, double value) const;
264
265 /// Converts the specified angle value from its native radians
266 /// to the desired units.
267 ///
268 /// @param angle [in] Value to be converted.
269 ///
270 /// @return Angle value converted to the desired units.
271 ///
272 virtual double ConvertAngle(double angle) const = 0;
273
274protected:
275 /// Constructs an angular-based unit.
276 ///
277 /// @param unitsId [in] Identifier for the angular units.
278 /// @param unitsStr [in] Name for the units.
279 ///
280 MeaAngularUnits(MeaAngularUnitsId unitsId, PCTSTR unitsStr);
281
282private:
283 static bool m_supplementalAngle; ///< Indicates if the supplemental rather than included angle should be shown.
284 MeaAngularUnitsId m_unitsId; ///< Identifier for the units.
285};

336/// There are two major categories of units: linear and angular.
337/// The linear units cover distance related measurements
338/// (e.g. coordinates, distances, resolutions). This is the base
339/// class for all linear units.
340///
341class MeaLinearUnits : public MeaUnits {
342
343public:
344 /// Destroys the linear units.
345 ///
346 virtual ~MeaLinearUnits();
347
348 /// Returns the identifier for the linear units.
349 ///
350 /// @return Linear units ID.
351 ///
352 MeaLinearUnitsId GetUnitsId() const { return m_unitsId; }
353
354
355 /// By default, the Windows coordinate system is inverted from the
356 /// typical Cartesian coordinate system in that the origin is located
357 /// in the upper left corner of the primary screen and the positive
358 /// y-axis points downward. This method sets whether the y-axis should
359 /// be inverted so that the origin is moved to the lower left corner
360 /// of the primary screen and the positive y-axis points up. Note that
361 /// the actual location of the origin is determined by the SetOrigin
362 /// method.
363 ///
364 /// @param invertY [in] <b>true</b> inverts the y-axis, placing
365 /// the origin at the lower left of the primary
366 /// display screen and making positive y pointing
367 /// upward.
368 ///
369 static void SetInvertY(bool invertY) { m_invertY = invertY; }
370
371 /// Returns the orientation of the y-axis.
372 ///
373 /// @return If <b>true</b> is returned, the origin of the coordinate
374 /// system is located at the lower left of the primary screen,
375 /// and the positive y-axis is pointing upward. If <b>false</b>
376 /// is returned, the origin is located at the upper left corner
377 /// and the positive y-axis is pointing down. Note that the
378 /// actual location of the origin is determined by the SetOrigin
379 /// method.
380 ///
381 static bool IsInvertY() { return m_invertY; }
382
383 /// Moves the origin of the coordinate system to the specified point. The
384 /// orientation of the axes is not effected by this method. To change the
385 /// orientation of the y-axis use the SetInvertY method.
386 ///
387 /// @param origin [in] New location for the origin of the coordinate system.
388 /// in pixels.
389 ///
390 static void SetOrigin(const POINT& origin) { m_originOffset = origin; }
391
392 /// Returns the location of the origin of the coordinate system.
393 ///
394 /// @return Location of the origin of the coordinate system, in pixels.
395 ///
396 static const POINT& GetOrigin() { return m_originOffset; }
397
398 /// Internally all measurements are in pixels. Measurement units based
399 /// solely on pixels do not require the use of the screen resolution for
400 /// conversion. Measurement units such as inches, require the screen
401 /// resolution for conversion from pixels. This method indicates whether
402 /// the screen resolution is required for conversion from pixels to the
403 /// desired measurement units and vice versa.
404 ///
405 /// @return <b>true</b> if the screen resolution must be used to convert
406 /// to and from pixels. This base class implementation always
407 /// returns <b>true</b>.
408 ///
409 virtual bool RequiresRes() const;
410
411 /// Formats the specified linear measurement value using the
412 /// precision for the specified measurement ID.
413 ///
414 /// @param id [in] Identifier for the linear units whose precision
415 /// is to be used to format the specified value.
416 /// @param value [in] Measurement value to be formatted.
417 ///
418 /// @return Measurement value formatted with the appropriate precision
419 /// and returned as a string.
420 ///
421 CString Format(MeaLinearMeasurementId id, double value) const;
422
423 /// Converts the specified coordinate from pixels to the desired units.
424 /// This conversion takes into account the location of the origin and the
425 /// orientation of the y-axis.
426 ///
427 /// @param pos [in] Coordinates in pixels to convert to the desired units.
428 ///
429 /// @return Coordinates converted from pixels to the desired units. The
430 /// conversion takes into account the location of the origin and
431 /// the orientation of the y-axis.
432 ///
433 MeaFPoint ConvertCoord(const POINT& pos) const;
434
435 /// Converts the specified position from pixels to the desired units.
436 /// This conversion does not take into account the location of the origin
437 /// nor does it compensate for the orientation of the y-axis.
438 ///
439 /// @param pos [in] Position in pixels to convert to the desired units.
440 ///
441 /// @return Position converted from pixels to the desired units. The
442 /// conversion does not take into account the location of the
443 /// origin nor does it compensate for the orientation of the y-axis.
444 ///
445 MeaFPoint ConvertPos(const POINT& pos) const;
446
447 /// Converts the specified resolution in pixels/inch to the desired units.
448 ///
449 /// @param res [in] Resolution in pixels/inch to convert to the desired units.
450 ///
451 /// @return Resolution converted from pixels/inch to the desired units.
452 ///
453 MeaFSize ConvertRes(const MeaFSize& res) const;
454
455 /// Converts from the current units to pixels. The conversion takes into account
456 /// the location of the origin and the orientation of the y-axis. The conversion
457 /// is performed for the specified coordinate located on the specified axis.
458 ///
459 /// @param dir [in] Axis on which the conversion should take place.
460 /// @param wnd [in] Window used to determine the screen resolution.
461 /// @param pos [in] Coordinate to convert to pixels.
462 ///
463 /// @return Coordinate converted from the current units to pixels. The conversion
464 /// takes into account the location of the origin and the orientation of
465 /// the y-axis.
466 ///
467 double UnconvertCoord(MeaConvertDir dir, const CWnd* wnd, double pos) const;
468
469 /// Converts from the current units to pixels. The conversion takes into account
470 /// the location of the origin and the orientation of the y-axis. The conversion
471 /// is performed for the specified coordinate located on the specified axis. Since
472 /// the specified coordinate might not convert to an integral number of pixels,
473 /// this method returns the bounding integral pixel values and indicates whether
474 /// an exact integral conversion could be performed.
475 ///
476 /// @param dir [in] Axis on which the conversion should take place.
477 /// @param wnd [in] Window used to determine the screen resolution.
478 /// @param pos [in] Coordinate to convert to pixels.
479 /// @param c1 [out] Closest integral pixel value below the converted
480 /// coordinate.
481 /// @param c2 [out] Closest integral pixel value above the converted
482 /// coordinate.
483 ///
484 /// @return <b>true</b> if the conversion to pixels was exact.
485 ///
486 bool UnconvertCoord(MeaConvertDir dir, const CWnd* wnd, double pos, int& c1, int& c2) const;
487
488 /// Converts from the current units to pixels. The conversion takes into account
489 /// the location of the origin and the orientation of the y-axis.
490 ///
491 /// @param pos [in] Coordinates to convert to pixels.
492 ///
493 /// @return Coordinates converted from the current units to pixels. The conversion
494 /// takes into account the location of the origin and the orientation of
495 /// the y-axis.
496 ///
497 POINT UnconvertCoord(const MeaFPoint& pos) const;
498
499 /// Converts from the current units to pixels. The conversion does not take into
500 /// account the location of the origin nor the orientation of the y-axis.
501 ///
502 /// @param pos [in] Position to convert to pixels.
503 ///
504 /// @return Position converted from the current units to pixels. The conversion
505 /// does not take into account the location of the origin nor the
506 /// orientation of the y-axis.
507 ///
508 POINT UnconvertPos(const MeaFPoint& pos) const;
509
510 /// Converts the specified value from the current units to pixels. A minimum
511 /// pixel value is specified in case the resolution is such that the conversion
512 /// to pixels results in a value that is too small.
513 ///
514 /// @param res [in] Screen resolution to use to perform the conversion.
515 /// @param value [in] Value to be converted to pixels.
516 /// @param minPixels [in] If the converted pixel value in less than this
517 /// minimum value, the minimum value is returned.
518 ///
519 /// @return X and Y pixel values.
520 ///
521 SIZE ConvertToPixels(const MeaFSize& res, double value, int minPixels) const;
522
523 /// Returns the X and Y factors to convert from pixels to the
524 /// current units. In other words, multiplying the values returned
525 /// from this method by pixels results in a value expressed in the
526 /// current units.
527 ///
528 /// @param res [in] Screen resolution, in pixels/inch.
529 ///
530 /// @return X and Y conversion factors, in units/pixels.
531 ///
532 virtual MeaFSize FromPixels(const MeaFSize& res) const = 0;
533
534protected:
535 /// Constructs the linear units.
536 ///
537 /// @param unitsId [in] Identifier for the linear units.
538 /// @param unitsStr [in] Name for the units.
539 /// @param screenProvider [in] Screen information provider
540 ///
541 MeaLinearUnits(MeaLinearUnitsId unitsId, PCTSTR unitsStr, const MeaScreenProvider& screenProvider);
542
543 /// Returns a resolution conversion factor. For most units this is
544 /// the same as the values returned by the FromPixels method. However,
545 /// since internally the resolution is expressed as pixels per inch,
546 /// the pixel units need to provide their own conversion factor.
547 ///
548 /// @param res [in] Resolution to convert.
549 ///
550 /// @return Resolution conversion factor.
551 ///
552 virtual MeaFSize GetResFromPixels(const MeaFSize& res) const;
553
554 /// Converts from the pixels to the current units. The conversion takes into account
555 /// the location of the origin and the orientation of the y-axis. The conversion
556 /// is performed for the specified coordinate located on the specified axis.
557 ///
558 /// @param dir [in] Axis on which the conversion should take place.
559 /// @param wnd [in] Window used to determine the screen resolution.
560 /// @param pos [in] Coordinate to convert from pixels.
561 ///
562 /// @return Coordinate converted from pixels to the current units. The conversion
563 /// takes into account the location of the origin and the orientation of
564 /// the y-axis.
565 ///
566 double ConvertCoord(MeaConvertDir dir, const CWnd* wnd, int pos) const;
567
568 /// In a multiple monitor environment there are multiple screen resolutions,
569 /// one set per monitor. Therefore, to determine the a resolution, a screen must
570 /// be determined. This method uses the specified position to determined a screen
571 /// and thus a resolution. The method compensates for the location of the origin
572 /// and the orientation of the y-axis.
573 ///
574 /// @param pos [in] Position used to determine a screen and thus a resolution,
575 /// in the current units.
576 ///
577 /// @return Screen resolution, in the current units.
578 ///
579 const MeaFSize& FindResFromCoord(const MeaFPoint& pos) const;
580
581 /// In a multiple monitor environment there are multiple screen resolutions,
582 /// one set per monitor. Therefore, to determine the a resolution, a screen must
583 /// be determined. This method uses the specified position to determined a screen
584 /// and thus a resolution. The method does not compensate for the location of the
585 /// origin nor the orientation of the y-axis.
586 ///
587 /// @param pos [in] Position used to determine a screen and thus a resolution,
588 /// in the current units.
589 ///
590 /// @return Screen resolution, in the current units.
591 ///
592 const MeaFSize& FindResFromPos(const MeaFPoint& pos) const;
593
594protected:
595 const MeaScreenProvider& m_screenProvider; ///< Screen information provider
596
597private:
598 static CPoint m_originOffset; ///< Offset of the origin from the system origin, in pixels.
599 static bool m_invertY; ///< Indicates if the y-axis direction is inverted.
600 MeaLinearUnitsId m_unitsId; ///< Linear units identifier.
601 int m_majorTickCount; ///< Number of minor ruler tick marks between major tick marks.
602};

648 /// @param res [in] Resolution to convert.
649 ///
650 /// @return Resolution conversion factor.
651 ///
652 virtual MeaFSize GetResFromPixels(const MeaFSize& res) const override;
