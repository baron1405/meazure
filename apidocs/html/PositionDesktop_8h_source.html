36/// Represents all information of interest about the system at the time a position is recorded. This includes the

144 /// Compares the specified desktop information object with this to determine equality.

151 return (MeaNumericUtils::IsEqualF(m_origin.x, desktop.m_origin.x) &&

153 (m_invertY == desktop.m_invertY) &&
154 MeaNumericUtils::IsEqualF(m_size.cx, desktop.m_size.cx) &&

156 (m_linearUnits == desktop.m_linearUnits) &&
157 (m_angularUnits == desktop.m_angularUnits) &&
158 (m_screens == desktop.m_screens) &&
159 (m_customName == desktop.m_customName) &&
160 (m_customAbbrev == desktop.m_customAbbrev) &&
161 (m_customBasisStr == desktop.m_customBasisStr) &&
162 MeaNumericUtils::IsEqualF(m_customFactor, desktop.m_customFactor) &&
163 (m_customPrecisions == desktop.m_customPrecisions));

262/// References a desktop information object. Position objects hold a reference to the desktop on which they
263/// were recorded. References to desktop information object are reference counted so that unreferenced
264/// desktop information is not written to the log file. This class hides the reference counting details from
265/// the position object.
266///
267class MeaPositionDesktopRef {
268public:
269
270 /// Constructs a reference to the specified desktop information object.
271 ///
272 /// @param counter [in] Reference count manager
273 /// @param desktop [in] Desktop information object to reference
274 ///
275 MeaPositionDesktopRef(MeaPositionDesktopRefCounter* counter, const MeaPositionDesktop& desktop) :
276 m_counter(counter), m_id(desktop.GetId()) {
277 m_counter->AddDesktopRef(m_id);
278 }
279
280 /// Constructs a reference to the specified desktop information object. This constructor is used during
281 /// deserialization of the position log file.
282 ///
283 /// @param counter [in] Reference count manager
284 /// @param idStr [in] GUID identifier for a desktop information object
285 ///
286 MeaPositionDesktopRef(MeaPositionDesktopRefCounter* counter, const CString& idStr) :
287 m_counter(counter), m_id(idStr) {
288 m_counter->AddDesktopRef(m_id);
289 }
290
291 /// Makes a copy of the specified desktop information reference.
292 ///
293 /// @param ref [in] Desktop information object to copy
294 ///
295 MeaPositionDesktopRef(const MeaPositionDesktopRef& ref) : m_counter(ref.m_counter), m_id(ref.m_id) {
296 m_counter->AddDesktopRef(m_id);
297 }
298
299 ~MeaPositionDesktopRef() {
300 try {
301 m_counter->ReleaseDesktopRef(m_id);
302 } catch (...) {
303 assert(false);
304 }
305 }
306
307 MeaPositionDesktopRef& operator=(const MeaPositionDesktopRef& ref) {
308 if (&ref != this) {
309 m_counter->ReleaseDesktopRef(m_id);
310
311 m_counter = ref.m_counter;
312 m_id = ref.m_id;
313
314 m_counter->AddDesktopRef(m_id);
315 }
316
317 return *this;
318 }
319
320 bool operator==(const MeaPositionDesktopRef& ref) const { return m_id == ref.m_id; }
321 bool operator!=(const MeaPositionDesktopRef& ref) const { return m_id != ref.m_id; }
322
323 const MeaGUID& GetId() const { return m_id; }
324
325 CString ToString() const { return m_id.ToString(); }
326
327private:
328 MeaPositionDesktopRefCounter* m_counter;
329 MeaGUID m_id;
330};
