107 /// Used by the STL to perform ordering of MeaGUID objects in collections.
108 ///
109 struct less {
110 /// Compares two MeaGUID objects.
111 ///
112 /// @param lhs [in] Left hand side of the comparison.
113 /// @param rhs [in] Right hand side of the comparison.
114 ///
115 /// @return <b>true</b> if lhs < rhs lexically.
116 ///
117 bool operator()(const MeaGUID& lhs, const MeaGUID& rhs) const

148 /// Tests the specified MeaGUID object for equality with this.
149 /// @param guid [in] MeaGUID object to test.
150 /// @return <b>true</b> if the specified GUID object equals this object.
151 bool IsEqual(const MeaGUID& guid) const { return IsEqual(guid.m_guid); }
152
153 /// Tests the specified GUID structure for equality with this.
154 /// @param guid [in] Operating system defined GUID structure.
155 /// @return <b>true</b> if the specified GUID structure equals this object.
156 bool IsEqual(const GUID& guid) const {
157 return ((m_guid.Data1 == guid.Data1) &&
158 (m_guid.Data2 == guid.Data2) &&
159 (m_guid.Data3 == guid.Data3) &&
160 (m_guid.Data4[0] == guid.Data4[0]) &&
161 (m_guid.Data4[1] == guid.Data4[1]) &&
162 (m_guid.Data4[2] == guid.Data4[2]) &&
163 (m_guid.Data4[3] == guid.Data4[3]) &&
164 (m_guid.Data4[4] == guid.Data4[4]) &&
165 (m_guid.Data4[5] == guid.Data4[5]) &&
166 (m_guid.Data4[6] == guid.Data4[6]) &&
167 (m_guid.Data4[7] == guid.Data4[7]));
168 }
