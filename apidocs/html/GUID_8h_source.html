73 /// Tests for equality between two MeaGUID objects.
74 /// @param guid [in] MeaGUID object to test against this object.
75 /// @return <b>true</b> if the specified object equals this object.
76 bool operator==(const MeaGUID& guid) const { return this->m_guid == guid.m_guid; }
77
78 /// Tests for inequality between two MeaGUID objects.
79 /// @param guid [in] MeaGUID object to test against this object.
80 /// @return <b>true</b> if the specified object does not equal this object.
81 bool operator!=(const MeaGUID& guid) const { return !(*this == guid); }
82
83 /// Tests for equality between two this object and a GUID structure.
84 /// @param guid [in] Operating system defined GUID structure to test against this object.
85 /// @return <b>true</b> if the specified GUID equals this object.
86 bool operator==(const GUID& guid) const {
87 return ((m_guid.Data1 == guid.Data1) &&
88 (m_guid.Data2 == guid.Data2) &&
89 (m_guid.Data3 == guid.Data3) &&
90 (m_guid.Data4[0] == guid.Data4[0]) &&
91 (m_guid.Data4[1] == guid.Data4[1]) &&
92 (m_guid.Data4[2] == guid.Data4[2]) &&
93 (m_guid.Data4[3] == guid.Data4[3]) &&
94 (m_guid.Data4[4] == guid.Data4[4]) &&
95 (m_guid.Data4[5] == guid.Data4[5]) &&
96 (m_guid.Data4[6] == guid.Data4[6]) &&
97 (m_guid.Data4[7] == guid.Data4[7]));
98 }
99
100 /// Tests for inequality between two this object and a GUID structure.
101 /// @param guid [in] Operating system defined GUID structure to test against this object.
102 /// @return <b>true</b> if the specified GUID does not equal this object.
103 bool operator!=(const GUID& guid) const { return !(*this == guid); }
104
105 /// Casts to an operating system defined GUID structure.

110 /// Used by the STL to perform ordering of MeaGUID objects in collections.
111 ///
112 struct less {
113 /// Compares two MeaGUID objects.
114 ///
115 /// @param lhs [in] Left hand side of the comparison.
116 /// @param rhs [in] Right hand side of the comparison.
117 ///
118 /// @return <b>true</b> if lhs < rhs lexically.
119 ///
120 bool operator()(const MeaGUID& lhs, const MeaGUID& rhs) const {
121 return lhs.ToString() < rhs.ToString();
122 }
123 };

126 /// Assign the specified GUID structure to this.

130 m_guid = guid;
131 return *this;
132 }
133
134 /// Assign the specified GUID string to this.

138
139 /// Assign the specified MeaGUID object to this.
140 /// @param guid [in] MeaGUID object to assign.
141 /// @return This object.
142 MeaGUID& Assign(const MeaGUID& guid) {
143 if (this != &guid) {
144 Assign(guid.m_guid);
145 }
146 return *this;
147 }

160 GUID m_guid; ///< Underlying GUID for the object.
