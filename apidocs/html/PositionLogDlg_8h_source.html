28/// A non-modal dialog for recording new positions and managing positions
29/// that have already been recorded. Besides recording positions, the
30/// dialog allows individual positions to be deleted, replaced, and
31/// annotated.
32///
33/// This class is an observer of the position log manager. This allows
34/// the dialog to be called when positions are recorded, deleted,
35/// loaded or saved.
36///
37class MeaPositionLogDlg : public CDialog, public MeaPositionLogObserver {
38
39public:
40 enum { IDD = IDD_POSITION_MGR };
41
42 /// Constructs the position manager dialog. Since this is
43 /// a non-modal dialog, its Create method is called followed
44 /// by a call to ShowWindow.
45 ///
46 MeaPositionLogDlg();
47
48 /// Destroys the dialog.
49 ///
50 virtual ~MeaPositionLogDlg();
51
52 /// Creates the window for the dialog.
53 ///
54 /// @return TRUE if the window was created successfully.
55 ///
56 BOOL Create() {
57 return CDialog::Create(IDD_POSITION_MGR, nullptr);
58 }
59
60 // Position log observer support
61
62 /// Called when a position log file is loaded.
63 ///
64 virtual void LogLoaded() override;
65
66 /// Called when a position log file is saved.
67 ///
68 virtual void LogSaved() override;
69
70 /// Called when a new position is recorded.
71 /// @param posIndex [in] Index of the new position.
72 virtual void PositionAdded(int posIndex) override;
73
74 /// Called when an existing position is replaced with a new position.
75 /// @param posIndex [in] Index of the replaced position.
76 virtual void PositionReplaced(int posIndex) override;
77
78 /// Called when a position is deleted.
79 /// @param posIndex [in] Index of the deleted position.
80 virtual void PositionDeleted(int posIndex) override;
81
82 /// Called when all recorded positions have been deleted.
83 ///
84 virtual void PositionsDeleted() override;
85
86 /// Called right before the dialog is displayed. Sets the
87 /// state of all UI controls.
88 /// @return TRUE to continue the display of the dialog.
89 ///
90 virtual BOOL OnInitDialog() override;
91
92protected:
93 /// Called when the Add position button is pressed. Records
94 /// the current tool positions.
95 ///
96 afx_msg void OnAddPosition();
97
98 /// Called when the Delete All positions button is pressed.
99 /// Deletes all positions.
100 ///
101 afx_msg void OnDeletePositions();
102
103 /// Called when the Delete position button is pressed. Deletes
104 /// the position corresponding to the position scrollbar.
105 ///
106 afx_msg void OnDeletePosition();
107
108 /// Called when the Replace button is pressed. The current
109 /// scrollbar position is replaced with the current radio tool
110 /// position.
111 ///
112 afx_msg void OnReplacePosition();
113
114 /// Called when the position scrollbar is moved.
115 /// @param nSBCode [in] Trackbar code.
116 /// @param nPos [in] Position value.
117 /// @param pScrollBar [in] Scrollbar control object.
118 ///
119 afx_msg void OnHScroll(UINT nSBCode, UINT nPos, CScrollBar* pScrollBar);
120
121 /// Called when the position description field loses the keyboard
122 /// focus. If the text in the field differs from that in the
123 /// current position, it is entered into the position object.
124 ///
125 afx_msg void OnKillfocusPositionDesc();
126
127 /// Called when the Load button is pressed. Queries the user
128 /// for the position log file to load.
129 ///
130 afx_msg void OnLoadPositions();
131
132 /// Called when the Save button is pressed. Saves the position
133 /// log file to the current pathname or queries for a pathname.
134 ///
135 afx_msg void OnSavePositions();
136
137 /// Called when the Save As button is pressed. Queries for a
138 /// pathname and saves the position log file to that pathname.
139 ///
140 afx_msg void OnSavePositionsAs();
141
142 DECLARE_MESSAGE_MAP()
143
144
145 /// Called when the dialog Close button is pressed. This method
146 /// destroys the dialog's window.
147 ///
148 virtual void OnCancel() override;
149
150 /// Called after the non-client area of the dialog is destroyed.
151 /// This is the last step in dialog destruction and is the place
152 /// where a non-modal dialog class instance can be deleted.
153 ///
154 virtual void PostNcDestroy() override;
155
156 /// Sets the dialog title with the currently opened position filename.
157 ///
158 void SetDlgTitle();
159
160 /// Returns the scrollbar value which is the position list index.
161 /// @return Scrollbar position.
162 int GetScrollPos() const;
163
164 /// Sets the scrollbar value which is the position list index.
165 /// @param pos [in] Position index.
166 void SetScrollPos(int pos) const;
167
168 /// Sets the range for the position list scrollbar.
169 void SetScrollRange() const;
170
171 /// Update the position number field, the recorded date field
172 /// and the description field.
173 ///
174 /// @param posIndex [in] Position index or -1 if the scrollbar
175 /// should be read to determine the current position.
176 ///
177 void UpdatePositionInfo(int posIndex = -1) const;
178
179 /// Enables/disables the dialog controls based on the current
180 /// state of the position manager.
181 ///
182 void UpdateEnable() const;
183};
